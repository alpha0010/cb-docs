123 void DebuggerDriver::RunQueue()
124 {
125  if (m_QueueBusy || !m_DCmds.GetCount() || !IsProgramStopped())
126  return;
127 
128  DebuggerCmd *command = CurrentCommand();
129 
130 // Log(_T("Running command: ") + CurrentCommand()->m_Cmd);
131  // don't send a command if empty; most debuggers repeat the last command this way...
132  if (!command->m_Cmd.IsEmpty())
133  {
134  m_QueueBusy = true;
135  m_pDBG->DoSendCommand(command->m_Cmd);
136  if (command->IsContinueCommand())
137  m_ProgramIsStopped = false;
138  }
139 
140  // Call Action()
141  command->Action();
142 
143  // If the command was an action (i.e. no command specified,
144  // remove it from the queue and run the next command.
145  // For other commands, this happens in driver's ParseOutput().
146  if (command->m_Cmd.IsEmpty())
147  {
148  RemoveTopCommand(true);
149  RunQueue();
150  }
151 }
