77 inline int DetectRepeatingSymbols(wxString const &str, int pos)
78 {
79  int newPos = -1, currPos = pos;
80  while (1)
81  {
82  if (currPos + 4 >= static_cast<int>(str.length()))
83  break;
84  if (str[currPos + 1] != wxT(','))
85  break;
86  if (str[currPos + 3] == wxT('\''))
87  {
88  const wxString &s = str.substr(currPos + 3, str.length() - (currPos + 3));
89  if (regexRepeatedChars.Matches(s))
90  {
91  size_t start, length;
92  regexRepeatedChars.GetMatch(&start, &length, 0);
93  newPos = currPos + 3 + length;
94  if ((newPos + 4 < static_cast<int>(str.length()))
95  && str[newPos] == wxT(',') && str[newPos + 2] == wxT('"'))
96  {
97  newPos += 3;
98  while (newPos < static_cast<int>(str.length()) && str[newPos] != wxT('"'))
99  ++newPos;
100  if (newPos + 1 < static_cast<int>(str.length()) && str[newPos] == wxT('"'))
101  ++newPos;
102  }
103  currPos = newPos;
104  }
105  else
106  break;
107  }
108  else
109  break;
110 
111  // move the current position to point at the '"' character
112  currPos--;
113  }

117 inline bool GetNextToken(wxString const &str, int pos, Token &token)
118 {
119  token.hasRepeatedChar = false;
120  while (pos < static_cast<int>(str.length())
121  && (str[pos] == _T(' ') || str[pos] == _T('\t') || str[pos] == _T('\n')))
122  ++pos;
123 
124  if (pos >= static_cast<int>(str.length()))
125  return false;
126 
127  token.start = -1;
128  bool in_quote = false, in_char = false;
129  int open_braces = 0;
130  struct BraceType { enum Enum { None, Angle, Square, Normal }; };
131  BraceType::Enum brace_type = BraceType::None;
132 
133  switch (static_cast<wxChar>(str[pos]))
134  {
135  case _T('='):
136  token = Token(pos, pos + 1, Token::Equal);
137  return true;
138  case _T(','):
139  token = Token(pos, pos + 1, Token::Comma);
140  return true;
141  case _T('{'):
142  token = Token(pos, pos + 1, Token::OpenBrace);
143  return true;
144  case _T('}'):
145  token = Token(pos, pos + 1, Token::CloseBrace);
146  return true;
147 
148  case _T('"'):
149  in_quote = true;
150  token.type = Token::String;
151  token.start = pos;
152  break;
153  case _T('\''):
154  in_char = true;
155  token.type = Token::String;
156  token.start = pos;
157  break;
158  case _T('<'):
159  token.type = Token::String;
160  token.start = pos;
161  open_braces = 1;
162  brace_type = BraceType::Angle;
163  break;
164  case wxT('['):
165  token.type = Token::String;
166  token.start = pos;
167  open_braces = 1;
168  brace_type = BraceType::Square;
169  break;
170  case wxT('('):
171  token.type = Token::String;
172  open_braces = 1;
173  brace_type = BraceType::Normal;
174  token.start = pos;
175  break;
176  default:
177  token.type = Token::String;
178  token.start = pos;
179  }
180  ++pos;
181 
182  bool escape_next = false;
183  while (pos < static_cast<int>(str.length()))
184  {
185  if (open_braces == 0)
186  {
187  if (str[pos] == _T(',') && !in_quote)
188  {
189  token.end = pos;
190  return true;
191  }
192  else if ((str[pos] == _T('=') || str[pos] == _T('{') || str[pos] == _T('}')) && !in_quote && !in_char)
193  {
194  token.end = pos;
195  return true;
196  }
197  else if (str[pos] == _T('"'))
198  {
199  if (in_quote)
200  {
201  if (!escape_next)
202  {
203  int newPos = DetectRepeatingSymbols(str, pos);
204  if (newPos != -1)
205  {
206  token.hasRepeatedChar = true;
207  token.end = newPos;
208  return true;
209  }
210  else
211  {
212  token.end = pos + 1;
213  return true;
214  }
215  }
216  else
217  escape_next = false;
218  }
219  else
220  {
221  if (escape_next)
222  return false;
223  in_quote = true;
224  }
225  }
226  else if (str[pos] == _T('\''))
227  {
228  if (!escape_next)
229  in_char = !in_char;
230  escape_next = false;
231  }
232  else if (str[pos] == _T('\\'))
233  escape_next = true;
234  else
235  escape_next = false;
236 
237  switch (brace_type)
238  {
239  case BraceType::Angle:
240  if (str[pos] == wxT('<'))
241  open_braces++;
242  break;
243  case BraceType::Square:
244  if (str[pos] == wxT('['))
245  open_braces++;
246  break;
247  case BraceType::None:
248  default:
249  break;
250  }
251  }
252  else
253  {
254  switch (brace_type)
255  {
256  case BraceType::Angle:
257  if (str[pos] == wxT('<'))
258  open_braces++;
259  else if (str[pos] == wxT('>'))
260  --open_braces;
261  break;
262  case BraceType::Square:
263  if (str[pos] == wxT('['))
264  open_braces++;
265  else if (str[pos] == wxT(']'))
266  --open_braces;
267  break;
268  case BraceType::Normal:
269  if (str[pos] == wxT('('))
270  open_braces++;
271  else if (str[pos] == wxT(')'))
272  --open_braces;
273  break;
274  case BraceType::None:
275  default:
276  break;
277  }
278  }
279  ++pos;
280  }
281 
282  if (in_quote)
283  {
284  token.end = -1;
285  return false;
286  }
287  else
288  {
289  token.end = pos;
290  return true;
291  }

338  int token_real_end = 0;
339  while (GetNextToken(value, position, token))

342  token.Trim(value);

351  wxString::size_type pos_val = value.find(wxT('\n'), token_real_end);
352  if (pos_val == wxString::npos)
353  return false;
354  position = pos_val+1;
355  if (length > 0 && position >= start + length)
356  break;
357  continue;
358  }
359  else
360  {
361  // If we have the '\n' in the token, then we have the next valid token, too,
362  // so we correct the current token to be the correct one.
363  if (str.find_last_of(wxT(':'), pos) == wxString::npos)
364  return false;

387  token_real_end = comma_end;
388  break;

391  if (regexRepeatedChar.Matches(expanded_str))
392  continue;
393  token_real_end = expanded_token.end;
394  }
395  }
396  else if (expanded_token.end == static_cast<int>(value.length()) || value[expanded_token.end] == wxT('}'))
397  {
398  token.end = expanded_token.end;
399  token_real_end = expanded_token.end;
400  }
401  break;
402  }

493  last_was_closing_brace = false;
