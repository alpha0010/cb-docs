138 void SelectTargetDlg::EndModal(int retCode)
139 {
140  if (retCode == wxID_OK)
141  {
142  m_Selected = XRCCTRL(*this, "lstItems", wxListBox)->GetSelection();
143  ProjectBuildTarget* target = m_pProject->GetBuildTarget(m_Selected);
144  if (target)
145  {
146  // Search all '\n' in the program-parameters and replace them by
147  // ' '. This is necessary because a multiline text control may add
148  // '\n' to the text but these characters must not be part of the
149  // parameters when executing the program.
150  wxString execution_parameters = XRCCTRL(*this, "txtParams", wxTextCtrl)->GetValue();
151  wxString::size_type pos = 0;
152 
153  while ((pos = execution_parameters.find('\n', pos)) != wxString::npos)
154  {
155  execution_parameters[pos] = ' ';
156  }
157  target->SetExecutionParameters(execution_parameters);
158  target->SetHostApplication(XRCCTRL(*this, "txtHostApp", wxTextCtrl)->GetValue());
159  target->SetRunHostApplicationInTerminal(XRCCTRL(*this, "chkHostInTerminal", wxCheckBox)->IsChecked());
160  }
161  }
162  wxScrollingDialog::EndModal(retCode);
163 } // end of EndModal
