210 cbProject* ProjectManager::LoadProject(const wxString& filename, bool activateIt)
211 {
212  cbProject* result = nullptr;
213  if (!wxFileExists(filename) || !BeginLoadingProject())
214  {
215  return nullptr;
216  }
217 
218  // "Try" block (loop which only gets executed once)
219  // These blocks are extremely useful in constructs that need
220  // premature exits. Instead of having multiple return points,
221  // multiple return values and/or gotos,
222  // you just break out of the loop (which only gets executed once) to exit.
223  do
224  {
225  cbProject* project = IsOpen(filename);
226  if (project)
227  {
228  // already open
229  result = project;
230  break;
231  }
232 
233  if (FileTypeOf(filename) == ftCodeBlocksProject)
234  {
235  project = new cbProject(filename);
236 
237  // We need to do this because creating cbProject allows the app to be
238  // closed in the middle of the operation. So the class destructor gets
239  // called in the middle of a method call.
240 
241  if (!project->IsLoaded())
242  {
243  delete project;
244  break;
245  }
246 
247  result = project;
248  }
249  else // !ftCodeBlocksProject
250  {
251  // the plugin handler should call begin/end on its own...
252  EndLoadingProject(nullptr);
253 
254  cbMimePlugin* plugin = Manager::Get()->GetPluginManager()->GetMIMEHandlerForFile(filename);
255  if (plugin)
256  plugin->OpenFile(filename);
257  }
258 
259  break;
260  } while (false);
261  // we 're done
262 
263  EndLoadingProject(result);
264  if (activateIt)
265  {
266  if (m_IsLoadingWorkspace)
267  // postpone the call to SetProject() until EndLoadingWorkspace() is called
268  // (we must call RebuildTree() before SetProject() is called)
269  m_pProjectToActivate = result;
270  else
271  SetProject(result, true);
272  }
273 
274  return result;
275 }

277 void ProjectManager::ReloadProject(cbProject *project)
278 {
279  m_ui->FreezeTree();
280 
281  bool workspaceModified = m_pWorkspace ? m_pWorkspace->GetModified() : false;
282  wxString name = project->GetFilename();
283  wxString activeProjectName = m_pActiveProject ? m_pActiveProject->GetFilename() : wxString(wxEmptyString);
284  ProjectsArray projectDependencies; // all projects that the reloaded project depends on
285  ProjectsArray projectsDependingOnReloaded; // all projects that depend on the reloaded project
286 
287  for (DepsMap::iterator it = m_ProjectDeps.begin(); it != m_ProjectDeps.end(); ++it)
288  {
289  if (!it->second)
290  continue;
291 
292  if (it->first == project)
293  projectDependencies = *(it->second);
294  else
295  {
296  if (it->second->Index(project) != wxNOT_FOUND)
297  projectsDependingOnReloaded.push_back(it->first);
298  }
299  }
300 
301 
302  int originalPosition = m_pProjects->Index(project);
303 
304  CloseProject(project);
305  cbProject *loadedProject = LoadProject(name);
306 
307  if (loadedProject)
308  {
309  if (!projectDependencies.empty())
310  {
311  for (ProjectsArray::iterator it = projectDependencies.begin(); it != projectDependencies.end(); ++it)
312  AddProjectDependency(loadedProject, *it);
313  }
314  if (!projectsDependingOnReloaded.empty())
315  {
316  for (ProjectsArray::iterator it = projectsDependingOnReloaded.begin();
317  it != projectsDependingOnReloaded.end(); ++it)
318  {
319  AddProjectDependency(*it, loadedProject);
320  }
321  }
322 
323  int loadedPosition = -1;
324  int index = 0;
325  cbProject *projectToActivate = nullptr;
326  for (ProjectsArray::iterator it = m_pProjects->begin(); it != m_pProjects->end(); ++it, ++index)
327  {
328  if (*it == loadedProject)
329  loadedPosition = index;
330 
331  if ((*it)->GetFilename() == activeProjectName)
332  projectToActivate = *it;
333  }
334 
335  m_pProjects->RemoveAt(loadedPosition);
336  m_pProjects->Insert(loadedProject, originalPosition);
337 
338  if (projectToActivate)
339  m_pActiveProject = projectToActivate;
340 
341  m_ui->RebuildTree();
342 
343  if (m_pWorkspace)
344  m_pWorkspace->SetModified(workspaceModified);
345  }
346 
347  m_ui->UnfreezeTree();
348 }

412 bool ProjectManager::CloseProject(cbProject* project, bool dontsave, bool refresh)
413 {
414  if (!project)
415  return true;
416  if (project->GetCurrentlyCompilingTarget())
417  return false;
418  if (!dontsave)

420  if (!m_ui->QueryCloseProject(project))
421  return false;

423 
424  bool wasActive = project == m_pActiveProject;
425  if (wasActive)
426  m_pActiveProject = nullptr;
427 
428  int index = m_pProjects->Index(project);
429  if (index == wxNOT_FOUND)
430  return false;

440  m_pWorkspace->SetModified(true);

442  RemoveProjectFromAllDependencies(project);

445  // moved here from cbProject's destructor, because by then
446  // the list of project files was already emptied...
447  CodeBlocksEvent event(cbEVT_PROJECT_CLOSE);
448  event.SetProject(project);
449  Manager::Get()->GetPluginManager()->NotifyPlugins(event);

451  project->CloseAllFiles(true);
452  if (refresh)
453  m_ui->RemoveProject(project);
454  if (wasActive && m_pProjects->GetCount())
455  SetProject(m_pProjects->Item(0), refresh);
456  delete project;
457  if (!m_InitialDir.IsEmpty()) // Restore the working directory
458  wxFileName::SetCwd(m_InitialDir);
459  m_IsClosingProject = isClosingOtherProjects;

461  return true;

656 {

663  // if project has only one target, use this

665  targets.Add(0);

686  for (size_t i = 0; i < targets.GetCount(); ++i)
687  {
688  ProjectBuildTarget* target = project->GetBuildTarget(targets[i]);
