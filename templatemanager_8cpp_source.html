62 void TemplateManager::LoadUserTemplates()
63 {
64  m_UserTemplates.Clear();
65  wxString baseDir = ConfigManager::GetConfigFolder() + wxFILE_SEP_PATH + _T("UserTemplates");
66 
67  if (!wxDirExists(baseDir)) // avoid warnings in debug builds
68  return;
69 
70  wxDir dir(baseDir);
71  if (!dir.IsOpened())
72  return;
73 
74  wxString filename;
75  bool ok = dir.GetFirst(&filename, _T("*"), wxDIR_DIRS);
76  while (ok)
77  {
78  m_UserTemplates.Add(filename);
79  ok = dir.GetNext(&filename);
80  }
81 
82  Manager::Get()->GetLogManager()->DebugLog(F(_T("%d user templates loaded"), m_UserTemplates.GetCount()));
83 }

123 cbProject* TemplateManager::NewProjectFromUserTemplate(NewFromTemplateDlg& dlg, wxString* pFilename)
124 {
125  cbProject* prj = NULL;
126  if (!dlg.SelectedUserTemplate())
127  {
128  Manager::Get()->GetLogManager()->DebugLog(_T("TemplateManager::NewProjectFromUserTemplate() called when no user template was selected ?!?"));
129  return NULL;
130  }
131 
132  wxString path = Manager::Get()->GetConfigManager(_T("template_manager"))->Read(_T("/projects_path"));
133  wxString sep = wxFileName::GetPathSeparator();
134  // select directory to copy user template files
135  path = ChooseDirectory(nullptr, _("Choose a directory to create the new project"),
136  path, _T(""), false, true);
137  if (path.IsEmpty())
138  return NULL;
139  else if (path.Mid(path.Length() - 1) == wxFILE_SEP_PATH)
140  path.RemoveLast();
141 
142  // check for existing files; if found, notify about overwriting them
143  if (wxDirExists(path))
144  {
145  wxDir dir(path);
146  if (dir.HasFiles() || dir.HasSubDirs())
147  {
148  if (cbMessageBox(path + _(" already contains other files.\n"
149  "If you continue, files with the same names WILL BE OVERWRITTEN.\n"
150  "Are you sure you want to continue?"),
151  _("Files exist in directory"), wxICON_EXCLAMATION | wxYES_NO | wxNO_DEFAULT) != wxID_YES)
152  {
153  return nullptr;
154  }
155  }
156  }
157 
158  wxBusyCursor busy;
159 
160  wxString templ = ConfigManager::GetConfigFolder() + wxFILE_SEP_PATH + _T("UserTemplates");
161  templ << sep << dlg.GetSelectedUserTemplate();
162  if (!wxDirExists(templ))
163  {
164  Manager::Get()->GetLogManager()->DebugLog(F(_T("Cannot open user-template source path '%s'!"), templ.wx_str()));
165  return NULL;
166  }
167 
168  // copy files
169  wxString project_filename;
170  wxArrayString files;
171  wxDir::GetAllFiles(templ, &files);
172  int count = 0;
173  int total_count = files.GetCount();
174  for (size_t i = 0; i < files.GetCount(); ++i)
175  {
176  wxFileName dstname(files[i]);
177  dstname.MakeRelativeTo(templ + sep);
178  wxString src = files[i];
179  wxString dst = path + sep + dstname.GetFullPath();
180 // Manager::Get()->GetLogManager()->DebugLog("dst=%s, dstname=%s", dst.c_str(), dstname.GetFullPath().c_str());
181  if (!CreateDirRecursively(dst))
182  Manager::Get()->GetLogManager()->DebugLog(_T("Failed creating directory for ") + dst);
183  if (wxCopyFile(src, dst, true))
184  {
185  if (FileTypeOf(dst) == ftCodeBlocksProject)
186  project_filename = dst;
187  ++count;
188  }
189  else
190  #if wxCHECK_VERSION(3, 0, 0)
191  Manager::Get()->GetLogManager()->DebugLog(F(_T("Failed copying %s to %s"), src.wx_str(), dst.wx_str()));
192  #else
193  Manager::Get()->GetLogManager()->DebugLog(F(_T("Failed copying %s to %s"), src.c_str(), dst.c_str()));
194  #endif
195  }
196  if (count != total_count)
197  cbMessageBox(_("Some files could not be loaded with the template..."), _("Error"), wxICON_ERROR);
198  else
199  {
200  // open new project
201  if (project_filename.IsEmpty())
202  cbMessageBox(_("User-template saved successfully but no project file exists in it!"));
203  else
204  {
205  // ask to rename the project file, if need be
206  wxFileName fname(project_filename);
207  wxString newname = cbGetTextFromUser(_("If you want, you can change the project's filename here (without extension):"), _("Change project's filename"), fname.GetName());
208  if (!newname.IsEmpty() && newname != fname.GetName())
209  {
210  fname.SetName(newname);
211  wxRenameFile(project_filename, fname.GetFullPath());
212  project_filename = fname.GetFullPath();
213  }
214  prj = Manager::Get()->GetProjectManager()->LoadProject(project_filename);
215  if (prj && !newname.IsEmpty())
216  {
217  prj->SetTitle(newname);
218  for (int i = 0; i < prj->GetBuildTargetsCount(); ++i)
219  {
220  ProjectBuildTarget* bt = prj->GetBuildTarget(i);
221  TargetFilenameGenerationPolicy tgfpPrefix, tgfpExtension;
222  bt->GetTargetFilenameGenerationPolicy(tgfpPrefix, tgfpExtension);
223  wxString outputFileName = bt->GetOutputFilename();
224  wxFileName outFname(outputFileName);
225  if (tgfpPrefix == tgfpPlatformDefault && bt->GetTargetType() == ttStaticLib)
226  {
227  Compiler* projectCompiler = CompilerFactory::GetCompiler(bt->GetCompilerID());
228  if (projectCompiler)
229  newname.Prepend(projectCompiler->GetSwitches().libPrefix);
230  }
231  outFname.SetName(newname);
232  bt->SetOutputFilename(outFname.GetFullPath());
233  }
234  Manager::Get()->GetProjectManager()->GetUI().RebuildTree(); // so the tree shows the new name
235  CodeBlocksEvent evt(cbEVT_PROJECT_OPEN, 0, prj);
236  Manager::Get()->ProcessEvent(evt);
237  }
238  }
239  }
240  if (prj && pFilename)
241  *pFilename = prj->GetFilename();
242  return prj;
243 }

245 void TemplateManager::SaveUserTemplate(cbProject* prj)
246 {
247  if (!prj)
248  return;
249 
250  // save project & all files
251  if (!prj->SaveAllFiles() ||
252  !prj->Save())
253  {
254  cbMessageBox(_("Could not save project and/or all its files. Aborting..."), _("Error"), wxICON_ERROR);
255  return;
256  }
257 
258  // create destination dir
259  wxString templ = ConfigManager::GetConfigFolder() + wxFILE_SEP_PATH + _T("UserTemplates");
260  if (!CreateDirRecursively(templ, 0755))
261  {
262  cbMessageBox(_("Couldn't create directory for user templates:\n") + templ, _("Error"), wxICON_ERROR);
263  return;
264  }
265 
266  // get default template title
267  wxString title = prj->GetTitle();
268 
269  // filter title, removing all illegal filename characters
270  wxFileName titleFileName(title) ;
271  wxString forbidden = titleFileName.GetForbiddenChars();
272  for (size_t i=0; i<forbidden.Length(); ++i)
273  title.Replace(wxString(forbidden[i]), wxT(""), true);
274 
275  // check if it exists and ask a different title
276  while (true)
277  {
278  // ask for template title (unique)
279  wxTextEntryDialog dlg(nullptr, _("Enter a title for this template"), _("Enter title"), title);
280  PlaceWindow(&dlg);
281  if (dlg.ShowModal() != wxID_OK)
282  return;
283 
284  title = dlg.GetValue();
285  if (!wxDirExists(templ + wxFILE_SEP_PATH + title))
286  {
287  templ << wxFILE_SEP_PATH << title;
288  wxMkdir(templ, 0755);
289  break;
290  }
291  else
292  cbMessageBox(_("You have another template with the same title.\nPlease choose another title..."));
293  }
294 
295  wxBusyCursor busy;
296 
297  // copy project and all files to destination dir
298  int count = 0;
299  int total_count = prj->GetFilesCount();
300  templ << wxFILE_SEP_PATH;
301  wxFileName fname;
302 
303  for (FilesList::iterator it = prj->GetFilesList().begin(); it != prj->GetFilesList().end(); ++it)
304  {
305  wxString src = (*it)->file.GetFullPath();
306  wxString dst = templ + (*it)->relativeToCommonTopLevelPath;
307  #if wxCHECK_VERSION(3, 0, 0)
308  Manager::Get()->GetLogManager()->DebugLog(F(_T("Copying %s to %s"), src.wx_str(), dst.wx_str()));
309  #else
310  Manager::Get()->GetLogManager()->DebugLog(F(_T("Copying %s to %s"), src.c_str(), dst.c_str()));
311  #endif
312  if (!CreateDirRecursively(dst))
313  Manager::Get()->GetLogManager()->DebugLog(_T("Failed creating directory for ") + dst);
314  if (wxCopyFile(src, dst, true))
315  ++count;
316  else
317  #if wxCHECK_VERSION(3, 0, 0)
318  Manager::Get()->GetLogManager()->DebugLog(F(_T("Failed copying %s to %s"), src.wx_str(), dst.wx_str()));
319  #else
320  Manager::Get()->GetLogManager()->DebugLog(F(_T("Failed copying %s to %s"), src.c_str(), dst.c_str()));
321  #endif
322  }
323 
324  // cbProject doesn't have a GetRelativeToCommonTopLevelPath() function, so we simulate it here
325  // to find out the real destination file to create...
326  wxString topLevelPath = prj->GetCommonTopLevelPath();
327  fname.Assign(prj->GetFilename());
328  fname.MakeRelativeTo(topLevelPath);
329  fname.Assign(templ + fname.GetFullPath());
330  if (!CreateDirRecursively(fname.GetPath(wxPATH_GET_VOLUME | wxPATH_GET_SEPARATOR), 0755))
331  {
332  cbMessageBox(_("Failed to create the directory for the project file!"), _("Error"), wxICON_ERROR);
333  ++count;
334  }
335  else
336  {
337  if (!wxCopyFile(prj->GetFilename(), fname.GetFullPath()))
338  {
339  Manager::Get()->GetLogManager()->DebugLog(_T("Failed to copy the project file: ") + fname.GetFullPath());
340  cbMessageBox(_("Failed to copy the project file!"), _("Error"), wxICON_ERROR);
341  ++count;
342  }
343  }
344 
345  if (count == total_count)
346  cbMessageBox(_("User-template saved successfully"), _("Information"), wxICON_INFORMATION | wxOK);
347  else
348  cbMessageBox(_("Some files could not be saved with the template..."), _("Error"), wxICON_ERROR);
349 }
