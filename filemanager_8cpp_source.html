136 LoaderBase* FileManager::Load(const wxString& file, bool reuseEditors)
137 {
138  if (reuseEditors)
139  {
140  // if a file is opened in the editor, and the file get modified, we use the content of the
141  // editor, otherwise, we still use the original file
142  EditorManager* em = Manager::Get()->GetEditorManager();
143  if (em)
144  {
145  wxFileName fileName(file);
146  for (int i = 0; i < em->GetEditorsCount(); ++i)
147  {
148  cbEditor* ed = em->GetBuiltinEditor(em->GetEditor(i));
149  if (ed && fileName == ed->GetFilename())
150  {
151  if (!ed->GetModified())
152  break;
153  EditorReuser *nl = new EditorReuser(file, ed->GetControl()->GetText());
154  return nl;
155  }
156  }
157  }
158  }
159 
160  if (file.StartsWith(_T("http://")))
161  {
162  URLLoader *ul = new URLLoader(file);
163  urlLoaderThread.Queue(ul);
164  return ul;
165  }
166 
167  FileLoader *fl = new FileLoader(file);
168 
169  if (file.length() > 2 && file[0] == _T('\\') && file[1] == _T('\\'))
170  {
171  // UNC files behave like "normal" files, but since we know they are served over the network,
172  // we can run them independently from local filesystem files for higher concurrency
173  uncLoaderThread.Queue(fl);
174  return fl;
175  }
176 
177  fileLoaderThread.Queue(fl);
178  return fl;

203 // FileManager::SaveUTF8 (formerly const char* overload of FileManager::Save) uses wxFile::Save without any conversions.
204 // It is only suitable e.g. for saving a TiXmlDocument (UTF8 data), but not for editors or projects, which contain wide char data.
205 // Therefore it is now a private member and has been renamed to prevent accidential use.
206 //
207 // FileManager::Save uses FileManager::WriteWxStringToFile to write data, its operation is otherwise exactly identical.
208 //
209 // 1. If file does not exist, create in exclusive mode, save, return success or failure. Exclusive mode is presumably "safer".
210 // 2. Otherwise (file exists), save to temporary.
211 // 3. Attempt to atomically replace original with temporary.
212 // 4. If this fails, rename temporary to document failure.
213 // 5. No more delayed deletes -- This requires special paths for when the application shuts down
214 // and only lends to race conditions and a possible crash-on-exit.
215 // If you can't trust your computer to sync data to disk properly, you already lost.
216 
217 bool FileManager::SaveUTF8(const wxString& name, const char* data, size_t len)
218 {
219  if (wxFileExists(name) == false)
220  {
221  return wxFile(name, wxFile::write_excl).Write(data, len) == len;
222  }
223  else
224  {
225  if (!wxFile::Access(name, wxFile::write))
226  return false;
227 
228  wxString temp(name);
229  temp.append(wxT(".temp"));
230 
231  wxStructStat buff;
232  wxLstat( name, &buff );
233 
234  wxFile f;
235  f.Create(temp, true, buff.st_mode);
236 
237  if(f.Write(data, len) == len)
238  {
239  f.Close();
240  if(platform::move(temp, name))
241  {
242  return true;
243  }
244  else
245  {
246  wxString failed(name);
247  failed.append(wxT(".save-failed"));
248  platform::move(temp, failed);
249  }
250  }
251  return false;
252  }

295 bool FileManager::WriteWxStringToFile(wxFile& f, const wxString& data, wxFontEncoding encoding, bool bom)
296 {
297  const char* mark = nullptr;
298  size_t mark_length = 0;
299  if (bom)
300  {
301  switch (encoding)
302  {
303  case wxFONTENCODING_UTF8:
304  mark = "\xEF\xBB\xBF";
305  mark_length = 3;
306  break;
307  case wxFONTENCODING_UTF16BE:
308  mark = "\xFE\xFF";
309  mark_length = 2;
310  break;
311  case wxFONTENCODING_UTF16LE:
312  mark = "\xFF\xFE";
313  mark_length = 2;
314  break;
315  case wxFONTENCODING_UTF32BE:
316  mark = "\x00\x00\xFE\xFF";
317  mark_length = 4;
318  break;
319  case wxFONTENCODING_UTF32LE:
320  mark = "\xFF\xFE\x00\x00";
321  mark_length = 4;
322  break;
323  case wxFONTENCODING_SYSTEM:
324  default:
325  break;
326  }
327 
328  if (f.Write(mark, mark_length) != mark_length)
329  return false;
330  }
331 
332  if (data.length() == 0)
333  return true;
334 
335 #if defined(UNICODE) || defined(_UNICODE)
336 
337  size_t inlen = data.Len(), outlen = 0;
338  wxCharBuffer mbBuff;
339  if ( encoding == wxFONTENCODING_UTF7 )
340  {
341  wxMBConvUTF7 conv;
342  mbBuff = conv.cWC2MB(data.c_str(), inlen, &outlen);
343  }
344  else if ( encoding == wxFONTENCODING_UTF8 )
345  {
346  wxMBConvUTF8 conv;
347  mbBuff = conv.cWC2MB(data.c_str(), inlen, &outlen);
348  }
349  else if ( encoding == wxFONTENCODING_UTF16BE )
350  {
351  wxMBConvUTF16BE conv;
352  mbBuff = conv.cWC2MB(data.c_str(), inlen, &outlen);
353  }
354  else if ( encoding == wxFONTENCODING_UTF16LE )
355  {
356  wxMBConvUTF16LE conv;
357  mbBuff = conv.cWC2MB(data.c_str(), inlen, &outlen);
358  }
359  else if ( encoding == wxFONTENCODING_UTF32BE )
360  {
361  wxMBConvUTF32BE conv;
362  mbBuff = conv.cWC2MB(data.c_str(), inlen, &outlen);
363  }
364  else if ( encoding == wxFONTENCODING_UTF32LE )
365  {
366  wxMBConvUTF32LE conv;
367  mbBuff = conv.cWC2MB(data.c_str(), inlen, &outlen);
368  }
369  else
370  {
371  // try wxEncodingConverter first, even it it only works for
372  // wxFONTENCODING_ISO8859_1..15, wxFONTENCODING_CP1250..1257 and wxFONTENCODING_KOI8
373  // but it's much, much faster than wxCSConv (at least on linux)
374  wxEncodingConverter conv;
375  // should be long enough
376  char* tmp = new char[2*inlen];
377 
378  if (conv.Init(wxFONTENCODING_UNICODE, encoding) && conv.Convert(data.wx_str(), tmp))
379  {
380  mbBuff = tmp;
381  outlen = strlen(mbBuff); // should be correct, because Convert has returned true
382  }
383  else
384  {
385  // try wxCSConv, if nothing else works
386  wxCSConv csconv(encoding);
387  mbBuff = csconv.cWC2MB(data.c_str(), inlen, &outlen);

392  if (outlen > 0)

399
