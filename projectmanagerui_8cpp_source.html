297 void ProjectManagerUI::RebuildTree()
298 {
299  if (Manager::IsAppShuttingDown()) // saves a lot of time at startup for large projects
300  return;
301 
302  FreezeTree();
303  ProjectManager* pm = Manager::Get()->GetProjectManager();
304  ProjectsArray* pa = pm->GetProjects();
305  int count = pa->GetCount();
306  for (int i = 0; i < count; ++i)
307  {
308  if ( cbProject* prj = pa->Item(i) )
309  prj->SaveTreeState(m_pTree);
310  }
311  m_pTree->DeleteAllItems();
312  wxString title;
313  bool read_only = false;
314  cbWorkspace* wkspc = pm->GetWorkspace();
315  if (wkspc)
316  {
317  title = wkspc->GetTitle();
318  wxString ws_file = wkspc->GetFilename();
319  read_only = ( !ws_file.IsEmpty()
320  && wxFile::Exists(ws_file.c_str())
321  && !wxFile::Access(ws_file.c_str(), wxFile::write) );
322  }
323  if (title.IsEmpty())
324  title = _("Workspace");
325  m_TreeRoot = m_pTree->AddRoot(title, cbProjectTreeImages::WorkspaceIconIndex(read_only), cbProjectTreeImages::WorkspaceIconIndex(read_only));
326  for (int i = 0; i < count; ++i)
327  {
328  if ( cbProject* prj = pa->Item(i) )
329  {
330  BuildProjectTree(prj, m_pTree, m_TreeRoot, m_TreeVisualState, pm->GetFilesGroupsAndMasks());
331  m_pTree->SetItemBold(prj->GetProjectNode(), prj == pm->GetActiveProject());
332  }
333  }
334  m_pTree->Expand(m_TreeRoot);
335 
336  for (int i = 0; i < count; ++i)
337  {
338  if ( cbProject* prj = pa->Item(i) )
339  prj->RestoreTreeState(m_pTree);
340  }
341  UnfreezeTree();
342 }

755 void ProjectManagerUI::DoOpenFile(ProjectFile* pf, const wxString& filename)
756 {
757  // Basic stuff: We can only open files that are still present
758  wxFileName the_file(filename);
759  if (!the_file.FileExists())
760  {
761  wxString msg;
762  msg.Printf(_("Could not open the file '%s'.\nThe file does not exist."), filename.c_str());
763  cbMessageBox(msg, _("Error"));
764  pf->SetFileState(fvsMissing);
765  Manager::Get()->GetLogManager()->LogError(msg);
766  return;
767  }
768 
769  FileType ft = FileTypeOf(filename);
770  if (ft == ftHeader || ft == ftSource || ft == ftTemplateSource)
771  {
772  // C/C++ header/source files, always get opened inside Code::Blocks
773  if ( cbEditor* ed = Manager::Get()->GetEditorManager()->Open(filename) )
774  {
775  ed->SetProjectFile(pf);
776  ed->Activate();
777  }
778  else
779  {
780  wxString msg;
781  msg.Printf(_("Failed to open '%s'."), filename.c_str());
782  Manager::Get()->GetLogManager()->LogError(msg);
783  }
784  }
785  else
786  {
787  // first look for custom editors
788  // if that fails, try MIME handlers
789  EditorBase* eb = Manager::Get()->GetEditorManager()->IsOpen(filename);
790  if (eb && !eb->IsBuiltinEditor())
791  {
792  // custom editors just get activated
793  eb->Activate();
794  return;
795  }
796 
797  // not a recognized file type
798  cbMimePlugin* plugin = Manager::Get()->GetPluginManager()->GetMIMEHandlerForFile(filename);
799  if (!plugin)
800  {
801  wxString msg;
802  msg.Printf(_("Could not open file '%s'.\nNo handler registered for this type of file."), filename.c_str());
803  Manager::Get()->GetLogManager()->LogError(msg);
804  }
805  else if (plugin->OpenFile(filename) != 0)
806  {
807  const PluginInfo* info = Manager::Get()->GetPluginManager()->GetPluginInfo(plugin);
808  wxString msg;
809  msg.Printf(_("Could not open file '%s'.\nThe registered handler (%s) could not open it."), filename.c_str(), info ? info->title.c_str() : wxString(_("<Unknown plugin>")).c_str());
810  Manager::Get()->GetLogManager()->LogError(msg);
811  }
812  }
