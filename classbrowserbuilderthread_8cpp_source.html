88 void ClassBrowserBuilderThread::Init(NativeParser* np,
89  CCTreeCtrl* treeTop,
90  CCTreeCtrl* treeBottom,
91  const wxString& active_filename,
92  void* user_data, // active project
93  const BrowserOptions& bo,
94  TokenTree* tt,
95  int idThreadEvent)
96 {
97  TRACE(_T("ClassBrowserBuilderThread::Init"));
98 
99  CC_LOCKER_TRACK_CBBT_MTX_LOCK(m_ClassBrowserBuilderThreadMutex);
100 
101  m_NativeParser = np;
102  m_CCTreeCtrlTop = treeTop;
103  m_CCTreeCtrlBottom = treeBottom;
104  m_ActiveFilename = active_filename;
105  m_UserData = user_data;
106  m_BrowserOptions = bo;
107  m_TokenTree = tt;
108  m_idThreadEvent = idThreadEvent;
109 
110  m_CurrentFileSet.clear();
111  m_CurrentTokenSet.clear();
112 
113  TokenTree* tree = m_NativeParser->GetParser().GetTokenTree();
114 
115  // fill filter set for current-file-filter
116  if ( m_BrowserOptions.displayFilter == bdfFile
117  && !m_ActiveFilename.IsEmpty() )
118  {
119  // m_ActiveFilename is the full filename up to the extension dot. No extension though.
120  // get all filenames' indices matching our mask
121  wxArrayString paths = m_NativeParser->GetAllPathsByFilename(m_ActiveFilename);
122 
123  // Should add locker after called m_NativeParser->GetAllPathsByFilename
124  CC_LOCKER_TRACK_TT_MTX_LOCK(s_TokenTreeMutex)
125 
126  TokenFileSet result;
127  for (size_t i = 0; i < paths.GetCount(); ++i)
128  {
129  tree->GetFileMatches(paths[i], result, true, true);
130  for (TokenFileSet::const_iterator tfs_it = result.begin(); tfs_it != result.end(); ++tfs_it)
131  m_CurrentFileSet.insert(*tfs_it);
132  }
133 
134  CC_LOCKER_TRACK_TT_MTX_UNLOCK(s_TokenTreeMutex)
135  }
136  else if ( m_BrowserOptions.displayFilter == bdfProject
137  && m_UserData )
138  {
139  CC_LOCKER_TRACK_TT_MTX_LOCK(s_TokenTreeMutex)
140 
141  cbProject* prj = static_cast<cbProject*>(m_UserData);
142  for (FilesList::const_iterator fl_it = prj->GetFilesList().begin();
143  fl_it != prj->GetFilesList().end(); ++fl_it)
144  {
145  ProjectFile* curFile = *fl_it;
146  if (!curFile)
147  continue;
148 
149  const size_t fileIdx = tree->GetFileIndex(curFile->file.GetFullPath());
150  if (fileIdx)
151  m_CurrentFileSet.insert(fileIdx);
152  }
153 
154  CC_LOCKER_TRACK_TT_MTX_UNLOCK(s_TokenTreeMutex)
155  }
156 
157  // our token tree has an internal map file -> tokens, so, we can collect all the tokens if we
158  // are given the file set.
159  // the tokens are stored in m_CurrentTokenSet, and the special global scope tokens are stored
160  // in the m_CurrentGlobalTokensSet.
161  if (!m_CurrentFileSet.empty())
162  {
163  CC_LOCKER_TRACK_TT_MTX_LOCK(s_TokenTreeMutex)
164 
165  m_CurrentTokenSet.clear();
166  m_CurrentGlobalTokensSet.clear();
167  for (TokenFileSet::const_iterator itf = m_CurrentFileSet.begin(); itf != m_CurrentFileSet.end(); ++itf)
168  {
169  const TokenIdxSet* tokens = tree->GetTokensBelongToFile(*itf);
170  if (!tokens)
171  continue;
172 
173  // loop tokens in file
174  for (TokenIdxSet::const_iterator its = tokens->begin(); its != tokens->end(); ++its)
175  {
176  Token* curToken = tree->at(*its);
177  if (curToken)
178  {
179  m_CurrentTokenSet.insert(*its);
180  if (curToken->m_ParentIndex == -1)
181  m_CurrentGlobalTokensSet.insert(*its);
182  }
183  }
184  }
185 
186  CC_LOCKER_TRACK_TT_MTX_UNLOCK(s_TokenTreeMutex)
187  }
188 
189  CC_LOCKER_TRACK_CBBT_MTX_UNLOCK(m_ClassBrowserBuilderThreadMutex)
190 }

194 void* ClassBrowserBuilderThread::Entry()
195 {
196  while (!m_TerminationRequested && !Manager::IsAppShuttingDown() )
197  {
198  // waits here, until the ClassBrowser unlocks
199  // we put a semaphore wait function in the while loop, so the first time if
200  // the semaphore is 1, we can call BuildTree() in the loop, in the meanwhile
201  // the semaphore becomes 0. We will be blocked by semaphore's wait function
202  // in the next while loop. The semaphore post function will be called in the
203  // GUI thread once a new BuildTree() call is needed.
204  m_ClassBrowserSemaphore.Wait();
205 
206  if (m_TerminationRequested || Manager::IsAppShuttingDown() )
207  break;
208 
209  if (platform::gtk || platform::macosx)
210  {
211  // this code (PART 1/2) seems to be good on linux
212  // because of it the libcairo crash on dualcore processors
213  // is gone, but on windows it has very bad influence,
214  // henceforth the ifdef guard
215  // the questions remains if it is the correct solution
216  if (!::wxIsMainThread())
217  ::wxMutexGuiEnter();
218  }
219 
220  BuildTree();
221 
222  if (platform::gtk || platform::macosx)
223  {
224  // this code (PART 2/2) seems to be good on linux
225  // because of it the libcairo crash on dualcore processors
226  // is gone, but on windows it has very bad influence,
227  // henceforth the ifdef guard
228  // the questions remains if it is the correct solution
229  if (!::wxIsMainThread())
230  ::wxMutexGuiLeave();
231  }
232  }
233 
234  m_NativeParser = 0;
235  m_CCTreeCtrlTop = 0;
236  m_CCTreeCtrlBottom = 0;
237 
238  return 0;

310  case tkEnum:
311  case tkTypedef:
312  case tkConstructor:
313  case tkDestructor:
314  case tkFunction:
315  case tkVariable:

317  case tkMacroDef:
318  case tkMacroUse:
319  case tkAnyContainer:
320  case tkAnyFunction:
321  case tkUndefined:
322  default:
323  break;
324  }
325  if (kind != 0)
326  AddChildrenOf(m_CCTreeCtrlTop, item, data->m_Token->m_Index, kind);

328  }

336  }

340  AddMembersOf(m_CCTreeCtrlTop, item);
341 #ifdef CC_BUILDTREE_MEASURING
342  CCLogger::Get()->DebugLog(F(_T("ExpandItems (internally) took : %ld ms for %u items."),sw.Time(),m_CCTreeCtrlTop->GetCount()));

415 void ClassBrowserBuilderThread::BuildTree()
416 {
417  if (CBBT_SANITY_CHECK || !m_CCTreeCtrlTop || !m_CCTreeCtrlBottom || !m_NativeParser)
418  return; // Called before UI tree construction completed?!
419 
420  wxCommandEvent e1(wxEVT_COMMAND_ENTER, m_idThreadEvent);
421  e1.SetInt(buildTreeStart);
422  m_Parent->AddPendingEvent(e1);
423 
424 #ifdef CC_BUILDTREE_MEASURING
425  wxStopWatch sw;
426  wxStopWatch sw_total;
427 #endif
428  // 1.) Registration of images
429  m_CCTreeCtrlTop->SetImageList(m_NativeParser->GetImageList());
430  m_CCTreeCtrlBottom->SetImageList(m_NativeParser->GetImageList());
431 
432  // 2.) Create initial root node, if not already there
433  wxTreeItemId root = m_CCTreeCtrlTop->GetRootItem();
434  if (!root.IsOk())
435  {
436  root = m_CCTreeCtrlTop->AddRoot(_("Symbols"), PARSER_IMG_SYMBOLS_FOLDER, PARSER_IMG_SYMBOLS_FOLDER, new CCTreeCtrlData(sfRoot));
437  m_CCTreeCtrlTop->SetItemHasChildren(root);

439 
440  // 3.) Update compare functions
441  m_CCTreeCtrlTop->SetCompareFunction(m_BrowserOptions.sortType);
442  m_CCTreeCtrlBottom->SetCompareFunction(m_BrowserOptions.sortType);
443 
444  // 4.) Save expanded items to restore later
445  m_ExpandedVect.clear();
446  SaveExpandedItems(m_CCTreeCtrlTop, root, 0);
447 #ifdef CC_BUILDTREE_MEASURING
448  CCLogger::Get()->DebugLog(F(_T("Saving expanded items took : %ld ms"),sw.Time()));
449  sw.Start();
450 #endif
451 
452  // 5.) Save selected item to restore later
453  SaveSelectedItem();
454 #ifdef CC_BUILDTREE_MEASURING
455  CCLogger::Get()->DebugLog(F(_T("Saving selected items took : %ld ms"),sw.Time()));
456  sw.Start();
457 #endif
458

460  if (m_BrowserOptions.treeMembers && m_CCTreeCtrlBottom)
461  {
462  m_CCTreeCtrlBottom->Hide();
463  m_CCTreeCtrlBottom->Freeze();
464  }

485 #endif

489  if (CBBT_SANITY_CHECK)
490  return;

590 void ClassBrowserBuilderThread::RemoveInvalidNodes(CCTreeCtrl* tree, wxTreeItemId parent)
591 {
592  TRACE(_T("ClassBrowserBuilderThread::RemoveInvalidNodes"));
593 
594  if (CBBT_SANITY_CHECK || !parent.IsOk())
595  return;
596 
597  // recursively enters all existing nodes and deletes the node if the token it references
598  // is invalid (i.e. m_TokenTree->at() != token_in_data)
599 
600  // we 'll loop backwards so we can delete nodes without problems
601  wxTreeItemId existing = tree->GetLastChild(parent);
602  while (parent.IsOk() && existing.IsOk())
603  {
604  bool removeCurrent = false;
605  bool hasChildren = tree->ItemHasChildren(existing);
606  CCTreeCtrlData* data = static_cast<CCTreeCtrlData*>(tree->GetItemData(existing));
607 
608  if (tree == m_CCTreeCtrlBottom)
609  removeCurrent = true;
610  else if (data && data->m_Token)
611  {
612  const Token* token = nullptr;
613  {
614  CC_LOCKER_TRACK_TT_MTX_LOCK(s_TokenTreeMutex)
615 
616  token = m_TokenTree->at(data->m_TokenIndex);
617 
618  CC_LOCKER_TRACK_TT_MTX_UNLOCK(s_TokenTreeMutex)
619  }
620  if ( token != data->m_Token
621  || (data->m_Ticket && data->m_Ticket != data->m_Token->GetTicket())
622  || !TokenMatchesFilter(data->m_Token) )
623  {
624  removeCurrent = true;
625  }
626  }
627 
628  if (removeCurrent)
629  {
630  if (hasChildren)
631  tree->DeleteChildren(existing);
632 
633  wxTreeItemId next = tree->GetPrevSibling(existing);
