331

333  {

335  panel->SetParentDialog(this);
336  lb->AddPage(panel, panel->GetTitle());
337 
338  wxString onFile = ConfigManager::LocateDataFile(base + panel->GetBitmapBaseName() + _T(".png"), sdDataGlobal | sdDataUser);
339  if (onFile.IsEmpty())
