69  {
70  wxString fname;
71  wxFileDialog dlg (Manager::Get()->GetAppWindow(),
72  fileDesc,
73  _T(""),
74  _T(""),
75  _T("Text files (*.txt)|*.txt|Any file (*)|*"),
76  wxFD_SAVE | wxFD_OVERWRITE_PROMPT);
77  PlaceWindow(&dlg);
78  if (dlg.ShowModal() != wxID_OK)
79  return;
80 
81  // Opening the file migth have failed, verify:
82  wxFile f(dlg.GetPath(), wxFile::write);
83  if (f.IsOpened())
84  {
85  f.Write(content); // write buffer to file
86  f.Close(); // release file handle
87  }
88  else
89  cbMessageBox(_("Cannot create file ") + fname, _("CC Debug Info"));
90  }

578 void CCDebugInfo::OnFindClick(cb_unused wxCommandEvent& event)
579 {
580  TokenTree* tree = m_Parser->GetTokenTree();
581  if (!tree) return;
582 
583  wxString search = txtFilter->GetValue();
584 
585  m_Token = 0;
586 
587  // first determine if the user entered an ID or a search mask
588  long unsigned id;
589  if (search.ToULong(&id, 10))
590  {
591  // easy; ID
592  m_Token = tree->at(id);
593  }
594  else
595  {
596  // find all matching tokens
597  TokenIdxSet result;
598  for (size_t i = 0; i < tree->size(); ++i)
599  {
600  const Token* token = tree->at(i);
601  if (token && token->m_Name.Matches(search))
602  result.insert(i);
603  }
604 
605  // a single result?
606  if (result.size() == 1)
607  m_Token = tree->at(*(result.begin()));
608  else
609  {
610  // fill a list and ask the user which token to display
611  wxArrayString arr;
612  wxArrayInt intarr;
613  for (TokenIdxSet::const_iterator it = result.begin(); it != result.end(); ++it)
614  {
615  const Token* token = tree->at(*it);
616  arr.Add(token->DisplayName());
617  intarr.Add(*it);
618  }
619  int sel = cbGetSingleChoiceIndex(_("Please make a selection:"), _("Multiple matches"), arr, this,
620  wxSize(400, 400));
621  if (sel == -1)
622  return;
623 
624  m_Token = tree->at(intarr[sel]);
625  }
626  }
627 
628  DisplayTokenInfo();
629 }

764  dirs += dir + _T("\r\n");

773  wxWindowDisabler disableAll;

799  }
