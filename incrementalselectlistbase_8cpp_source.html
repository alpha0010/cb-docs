67 void IncrementalSelectListBase::FilterItems()
68 {
69  m_Result.Empty();
70  m_Indexes.Empty();
71 
72  // We put a star before and after pattern to find search expression everywhere in path
73  // that is: if user enter "a", it will match "123a", "12a" or "a12".
74  wxString search(wxT("*") + m_Text->GetValue().Lower() + wxT("*"));
75 
76  for (int i = 0; i < m_Iterator.GetCount(); ++i)
77  {
78  wxString const &item = m_Iterator.GetItem(i);
79  // 2 for before and after stars =~ empty string
80  if ((search.Length()==2) || item.Lower().Matches(search.c_str()))
81  {
82  m_Result.Add(m_Iterator.GetDisplayItem(i));
83  m_Indexes.Add(i);
84  }
85  }
86 
87  // if only alphabetical, pull word boundaries to the top
88  if (search.Length() > 2)
89  {
90  wxString prefix;
91  for (size_t i = 0; i < search.Length(); ++i)
92  {
93  if (wxIsalpha(search[i]))
94  prefix += search[i];
95  }
96  if (prefix.Length() == search.Length() - 2)
97  {
98  std::vector<size_t> promoteIdxs;
99  wxArrayString newRes;
100  wxArrayLong newIndx;
101  for (size_t i = 0; i < m_Result.Count(); ++i)
102  {
103  wxString cur = m_Result[i].Lower();
104  bool promote = false;
105  if (cur.StartsWith(prefix))
106  promote = true;
107  else
108  {
109  int maxLn = cur.Length() - prefix.Length();
110  for (int j = 0; j < maxLn; ++j)
111  {
112  if (!wxIsalpha(cur[j]) && cur.Mid(j + 1).StartsWith(prefix))
113  {
114  promote = true;
115  break;
116  }
117  }
118  }
119  if (promote)
120  {
121  promoteIdxs.push_back(i);
122  newRes.Add(m_Result[i]);
123  newIndx.Add(m_Indexes[i]);
124  }
125  }
126  if (!promoteIdxs.empty())
127  {
128  for (size_t i = 0; i < m_Result.Count(); ++i)
129  {
130  if (!std::binary_search(promoteIdxs.begin(), promoteIdxs.end(), i))
131  {
132  newRes.Add(m_Result[i]);
133  newIndx.Add(m_Indexes[i]);
134  }
135  }
136  m_Result = newRes;
137  m_Indexes = newIndx;
138  }
139  }
140  }
141  m_Promoted = search.Length()>2;
142 
143 }

168 void IncrementalSelectListBase::KeyDownAction(wxKeyEvent& event, int &selected, int selectedMax)
169 {
170  // now, adjust position from key input
171  switch (event.GetKeyCode())
172  {
173  case WXK_RETURN:
174  case WXK_NUMPAD_ENTER:
175  EndModal(wxID_OK);
176  return;
177  break;
178 
179  case WXK_ESCAPE:
180  EndModal(wxID_CANCEL);
181  return;
182  break;
183 
184  case WXK_UP:
185  case WXK_NUMPAD_UP:
186  if (selected)
187  selected--;
188  break;
189 
190  case WXK_DOWN:
191  case WXK_NUMPAD_DOWN:
192  selected++;
193  break;
194 
195  case WXK_PAGEUP:
196  case WXK_NUMPAD_PAGEUP:
197  selected -= 10;
198  break;
199 
200  case WXK_PAGEDOWN:
201  case WXK_NUMPAD_PAGEDOWN:
202  selected += 10;
203  break;
204 
205  case WXK_HOME:
206  if (wxGetKeyState(WXK_CONTROL))
207  selected = 0;
208  else
209  event.Skip();
210  break;
211 
212  case WXK_END:
213  if (wxGetKeyState(WXK_CONTROL))
214  selected = selectedMax;
215  else
216  event.Skip();
217  break;
218 
219  default:
220  event.Skip();
221  break;
222  }
223 
224  // Clamp value below 0 and above Max
225  if (selected < 0)
226  selected = 0;
227  else
228  {
229  if (selected > selectedMax)
230  selected = selectedMax;
231  }
232 }
