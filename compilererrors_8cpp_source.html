70  break;
71  }

74  }

134 void CompilerErrors::DoGotoError(const CompileError& error)
135 {
136  if (error.line <= 0)
137  return;
138  DoClearErrorMarkFromAllEditors();
139  cbEditor* ed = 0;
140  cbProject* project = error.project ? error.project : Manager::Get()->GetProjectManager()->GetActiveProject();
141  if (project && Manager::Get()->GetProjectManager()->IsProjectStillOpen(project))
142  {
143  wxString filename = error.filename;
144  bool isAbsolute = (filename.Length() > 1 && filename.GetChar(1) == ':') ||
145  filename.StartsWith(_T("/")) ||
146  filename.StartsWith(_T("\\"));
147  ProjectFile* f = project->GetFileByFilename(error.filename, !isAbsolute, true);
148  if (f)
149  {
150  ed = Manager::Get()->GetEditorManager()->Open(f->file.GetFullPath());
151  if (ed)
152  {
153  ed->SetProjectFile(f);
154  }
155  }
156  else
157  {
158  if(!isAbsolute) // this is always the case, except for system headers
159  filename.Prepend(project->GetCommonTopLevelPath());
160 
161  ed = Manager::Get()->GetEditorManager()->Open(filename);
162  }
163  }
164 
165  // if we reached here and ed is NULL, the filename in the output isn't relative
166  // to the project root directory or doesn't belong to the project
167 
168  // first check if we can open it directly...
169  if (!ed)
170  ed = Manager::Get()->GetEditorManager()->Open(error.filename);
171 
172  // check if we find the file among opened files (highly probable for error
173  // messages since we are getting error for something we have just screwed up)
174  if (!ed)
175  {
176  for (int i = 0; i < Manager::Get()->GetEditorManager()->GetEditorsCount(); ++i)
177  {
178  cbEditor* edit = Manager::Get()->GetEditorManager()->GetBuiltinEditor(i);
179  if (!edit)
180  continue;
181 
182  ProjectFile* pf = edit->GetProjectFile();
183  if (!pf)
184  continue;
185 
186  if (IsSuffixOfPath(error.filename, pf->file.GetFullPath()))
187  {
188  ed = Manager::Get()->GetEditorManager()->Open(pf->file.GetFullPath());
189  break;
190  }
191  }
192  }
193 
194  // finally go through the project files and try to find the file there
195  if (!ed && project)
196  {
197  for (FilesList::iterator it = project->GetFilesList().begin(); it != project->GetFilesList().end(); ++it)
198  {
199  ProjectFile* pf = *it;
200  if (!pf)
201  continue;
202 
203  if (IsSuffixOfPath(error.filename, pf->file.GetFullPath()))
204  {
205  ed = Manager::Get()->GetEditorManager()->Open(pf->file.GetFullPath());
206  break;
207  }
208  }
209  }
210 
211  if (ed)
212  {
213  ed->Activate();
214  ed->UnfoldBlockFromLine(error.line - 1);
215  ed->GotoLine(error.line - 1);
216  ed->SetErrorLine(error.line - 1);
217  }
218 }
