397 {
398  public:
400  CdbCmd_Backtrace(DebuggerDriver* driver, bool switchToFirst)
401  : DebuggerCmd(driver),
402  m_SwitchToFirst(switchToFirst)
403  {
404  m_Cmd << _T("k n");

406  void ParseOutput(const wxString& output)
407  {
408  // output is:
409  // # ChildEBP RetAddr
410  // 00 0012fe98 00401426 Win32GUI!WinMain+0x89 [c:\devel\tmp\win32 test\main.cpp @ 55]
411  // 00 0012fe98 00401426 Win32GUI!WinMain+0x89
412  //
413  // so we have a two-steps process:
414  // 1) Get match for the second version (without file/line info)
415  // 2) See if we have file/line info and read it
416  m_pDriver->GetStackFrames().clear();
417 
418  wxArrayString lines = GetArrayFromString(output, _T('\n'));
419  if (!lines.GetCount() || !lines[0].Contains(_T("ChildEBP")))
420  return;
421 
422  bool firstValid = true;
423  cbStackFrame frameToSwitch;
424 
425  // start from line 1
426  for (unsigned int i = 1; i < lines.GetCount(); ++i)
427  {
428  if (reBT1.Matches(lines[i]))
429  {
430  cbStackFrame sf;
431  sf.MakeValid(true);
432 
433  #if defined(_WIN64)
434  size_t number, address;
435  reBT1.GetMatch(lines[i], 1).ToULongLong(&number);
436  reBT1.GetMatch(lines[i], 2).ToULongLong(&address, 16); // match 2 or 3 ???
437  #else
438  unsigned long int number, address;
439  reBT1.GetMatch(lines[i], 1).ToULong(&number);
440  reBT1.GetMatch(lines[i], 2).ToULong(&address, 16); // match 2 or 3 ???
441  #endif
442 
443  sf.SetNumber(number);
444  sf.SetAddress(address);
445  sf.SetSymbol(reBT1.GetMatch(lines[i], 4));
446  // do we have file/line info?
447  if (reBT2.Matches(lines[i]))
448  {
449  sf.SetFile(reBT2.GetMatch(lines[i], 1) + reBT2.GetMatch(lines[i], 2),
450  reBT2.GetMatch(lines[i], 3));
451  }
452  m_pDriver->GetStackFrames().push_back(cb::shared_ptr<cbStackFrame>(new cbStackFrame(sf)));
453 
454  if (m_SwitchToFirst && sf.IsValid() && firstValid)
455  {
456  firstValid = false;
457  frameToSwitch = sf;
458  }
459  }
460  }
461  Manager::Get()->GetDebuggerManager()->GetBacktraceDialog()->Reload();
462  if (!firstValid)
463  {
464  Cursor cursor;
465  cursor.file = frameToSwitch.GetFilename();
466  frameToSwitch.GetLine().ToLong(&cursor.line);
467  cursor.address = wxString::Format(wxT("0x%X"), frameToSwitch.GetAddress());
468  cursor.changed = true;
469  m_pDriver->SetCursor(cursor);
470  m_pDriver->NotifyCursorChanged();
471  }
472  }
473  private:
474  bool m_SwitchToFirst;
