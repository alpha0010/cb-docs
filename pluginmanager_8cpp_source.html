56 {
57  wxString majorS = version.BeforeFirst(_T('.')); // 6.3.2 -> 6

69 // returns -1 if new is less then old, 0 if equal and 1 if new is greater than old
70 inline int CompareVersions(const wxString& new_version, const wxString& old_version)
71 {
72  long new_major, new_minor, new_release;
73  long old_major, old_minor, old_release;
74 
75  VersionStringToNumbers(new_version, &new_major, &new_minor, &new_release);
76  VersionStringToNumbers(old_version, &old_major, &old_minor, &old_release);
77 
78 #define SIGN(a) (a>0?1:(a<0?-1:0))
79  int result = 0;
80  result += SIGN(new_major - old_major) << 2;
81  result += SIGN(new_minor - old_minor) << 1;
82  result += SIGN(new_release - old_release) << 0;
83 #undef SIGN
84 
85  if (result < 0) return -1;
86  else if (result > 0) return 1;
87  return 0;
88 }

122  {
123  RefCountedLib& rcl = it->second;

595 bool PluginManager::ExtractFile(const wxString& bundlename,
596  const wxString& src_filename,
597  const wxString& dst_filename,

604 
605  // check if the destination file already exists
606  if (wxFileExists(dst_filename) && !wxFile::Access(dst_filename, wxFile::write))
607  {
608 // Manager::Get()->GetLogManager()->DebugLog(F(_T("Destination file in use")));
609  cbMessageBox(_("The destination file is in use.\nAborting..."), _("Warning"), wxICON_WARNING);
610  return false;
611  }
612 
613  // make sure destination dir exists
614  CreateDirRecursively(wxFileName(dst_filename).GetPath(wxPATH_GET_VOLUME | wxPATH_GET_SEPARATOR));
615 
616  // actually extract file
617 // Manager::Get()->GetLogManager()->DebugLog(F(_T("Extracting...")));
618  wxFileSystem* fs = new wxFileSystem;
619  wxFSFile* f = fs->OpenFile(bundlename + _T("#zip:") + src_filename);
620  if (f)
621  {
622  // open output file for writing
623  wxFile output(dst_filename, wxFile::write);
624  if (!output.IsOpened())
625  {
626 // Manager::Get()->GetLogManager()->DebugLog(F(_T("Can't open destination file for writing")));

630  delete f;
631  delete fs;
632  return false;
633  }
634 
635  // copy file
636  wxInputStream* is = f->GetStream();
637  char tmp[1025] = {};
638  while (!is->Eof() && is->CanRead())
639  {
640  memset(tmp, 0, sizeof(tmp));
641  is->Read(tmp, sizeof(tmp) - 1);
642  output.Write(tmp, is->LastRead());
643  }
644  delete f;
645 // Manager::Get()->GetLogManager()->DebugLog(F(_T("Extracted")));
646  }
647  else
648  {
649 // Manager::Get()->GetLogManager()->DebugLog(F(_T("File not found in plugin")));
650  if (isMandatory)
651  {
652  wxString msg = wxString::Format(_("File '%s' not found in plugin '%s'"),
653  src_filename.c_str(), bundlename.c_str());
654  cbMessageBox(msg, _("Error"), wxICON_ERROR);
655  delete fs;
656  return false;
657  }
658  }

665  FreePluginProc freeProc,
666  PluginSDKVersionProc versionProc)
667 {

669  if (name.IsEmpty() || !createProc || !freeProc || !versionProc)
670  return;
671 
672  // first check to see it's not already loaded
673  if (FindPluginByName(name))
674  return; // yes, already loaded
675 
676  // read manifest file for plugin
677  PluginInfo info;
678  if (!ReadManifestFile(m_CurrentlyLoadingFilename, name, &info) ||
679  info.name.IsEmpty())
680  {
681  Manager::Get()->GetLogManager()->LogError(_T("Invalid manifest file for: ") + name);
682  return;
683  }
684 
685  // now get the SDK version number (extra check)
686  int major;
687  int minor;
688  int release;
689  versionProc(&major, &minor, &release);
690  if (major != PLUGIN_SDK_VERSION_MAJOR ||
691  minor != PLUGIN_SDK_VERSION_MINOR ||
692  release != PLUGIN_SDK_VERSION_RELEASE)
693  {
694  // wrong version: in this case, inform the user...
695  wxString fmt;
696  fmt.Printf(_("SDK version mismatch for %s (%d.%d.%d). Expecting %d.%d.%d"),
697  name.c_str(),
698  major,
699  minor,
700  release,
701  PLUGIN_SDK_VERSION_MAJOR,
702  PLUGIN_SDK_VERSION_MINOR,
703  PLUGIN_SDK_VERSION_RELEASE);
704  Manager::Get()->GetLogManager()->LogError(fmt);
705  return;
706  }
707 
708  // all done
709  // add this plugin in the temporary registration vector to be loaded
710  // by LoadPlugin() (which triggered the call to this function).
711  PluginRegistration pr;
712  pr.name = name;
713  pr.createProc = createProc;
714  pr.freeProc = freeProc;

716  pr.info = info;
717  m_RegisteredPlugins.push_back(pr);

1002 bool PluginManager::LoadPlugin(const wxString& pluginName)
1003 {
1004  // clear registration temporary vector
1005  m_RegisteredPlugins.clear();
1006 
1007  // load library
1008  m_CurrentlyLoadingFilename = pluginName;
1009  m_pCurrentlyLoadingLib = LibLoader::LoadLibrary(pluginName);
1010  if (!m_pCurrentlyLoadingLib->IsLoaded())
1011  {
1012  Manager::Get()->GetLogManager()->LogError(F(_T("%s: not loaded (missing symbols?)"), pluginName.wx_str()));
1013  LibLoader::RemoveLibrary(m_pCurrentlyLoadingLib);
1014  m_pCurrentlyLoadingLib = nullptr;
1015  m_CurrentlyLoadingFilename.Clear();
1016  return false;
1017  }
1018 
1019  // by now, the library has loaded and its global variables are initialized.
1020  // this means it has already called RegisterPlugin()
1021  // now we can actually create the plugin(s) instance(s) :)
1022 
1023  // try to load the plugin(s)
1024  std::vector<PluginRegistration>::iterator it;
1025  for (it = m_RegisteredPlugins.begin(); it != m_RegisteredPlugins.end(); ++it)
1026  {
1027  PluginRegistration& pr = *it;
1028  cbPlugin* plug = nullptr;
1029  try
1030  {
1031  plug = pr.createProc();
1032  }
1033  catch (cbException& exception)
1034  {
1035  exception.ShowErrorMessage(false);
1036  continue;
1037  }
1038 
1039  // all done; add it to our list
1040  PluginElement* plugElem = new PluginElement;
1041  plugElem->fileName = m_CurrentlyLoadingFilename;
1042  plugElem->info = pr.info;
1043  plugElem->library = m_pCurrentlyLoadingLib;
1044  plugElem->freeProc = pr.freeProc;
1045  plugElem->plugin = plug;
1046  m_Plugins.Add(plugElem);
1047 
1048  SetupLocaleDomain(pr.name);
1049 
1050  Manager::Get()->GetLogManager()->DebugLog(F(_T("%s: loaded"), pr.name.wx_str()));
1051  }
1052 
1053  if (m_RegisteredPlugins.empty())
1054  {
1055  // no plugins loaded from this library, but it's not an error
1056  LibLoader::RemoveLibrary(m_pCurrentlyLoadingLib);
1057  }
1058  m_pCurrentlyLoadingLib = nullptr;
1059  m_CurrentlyLoadingFilename.Clear();
1060  return true;
1061 }
