467  bool operator()(const cbCodeCompletionPlugin::CCToken& a, const cbCodeCompletionPlugin::CCToken& b)
468  {

483 // cbEVT_COMPLETE_CODE
484 void CCManager::OnCompleteCode(CodeBlocksEvent& event)
485 {
486  event.Skip();
487  ConfigManager* cfg = Manager::Get()->GetConfigManager(wxT("ccmanager"));
488  if (!cfg->ReadBool(wxT("/code_completion"), true))
489  return;
490 
491  cbEditor* ed = Manager::Get()->GetEditorManager()->GetBuiltinActiveEditor();
492  if (!ed)
493  return;
494  cbCodeCompletionPlugin* ccPlugin = GetProviderFor(ed);
495  if (!ccPlugin)
496  return;
497 
498  cbStyledTextCtrl* stc = ed->GetControl();
499  int tknEnd = stc->GetCurrentPos();
500  if (tknEnd == m_LastACLaunchState[lsCaretStart] && !m_AutocompTokens.empty())
501  {
502  DoBufferedCC(stc);
503  return;
504  }
505  int tknStart = stc->WordStartPosition(tknEnd, true);
506 
507  m_AutocompTokens = ccPlugin->GetAutocompList(event.GetInt() == FROM_TIMER,
508  ed, tknStart, tknEnd);
509  if (m_AutocompTokens.empty())
510  return;
511 
512  bool isPureAlphabetical = true;
513  TokenSorter sortFunctor(isPureAlphabetical);
514  std::sort(m_AutocompTokens.begin(), m_AutocompTokens.end(), sortFunctor);
515  if (isPureAlphabetical)
516  stc->AutoCompSetOrder(wxSCI_ORDER_PRESORTED);
517  else
518  stc->AutoCompSetOrder(wxSCI_ORDER_CUSTOM);
519  wxString items;
520  // experimentally, the average length per token seems to be 23 for the main CC plugin
521  items.Alloc(m_AutocompTokens.size() * 20); // TODO: measure performance
522  for (size_t i = 0; i < m_AutocompTokens.size(); ++i)
523  {
524  items += m_AutocompTokens[i].displayName;
525  if (m_AutocompTokens[i].category == -1)
526  items += wxT("\r");
527  else
528  items += F(wxT("\n%d\r"), m_AutocompTokens[i].category);
529  }
530  items.RemoveLast();
531 
532  if (!stc->CallTipActive() && !stc->AutoCompActive())
533  m_CallTipActive = wxSCI_INVALID_POSITION;
534 
535  stc->AutoCompSetIgnoreCase(!cfg->ReadBool(wxT("/case_sensitive"), false));
536  stc->AutoCompSetChooseSingle(cfg->ReadBool(wxT("/auto_select_single"), false));
537  stc->AutoCompSetMaxHeight(14);
538  stc->AutoCompSetTypeSeparator(wxT('\n'));
539  stc->AutoCompSetSeparator(wxT('\r'));
540  stc->AutoCompShow(tknEnd - tknStart, items);
541  m_OwnsAutocomp = true;
542  if (isPureAlphabetical)
543  {
544  const wxString& contextStr = stc->GetTextRange(tknStart, stc->WordEndPosition(tknEnd, true));
545  std::vector<cbCodeCompletionPlugin::CCToken>::const_iterator tknIt
546  = std::lower_bound(m_AutocompTokens.begin(), m_AutocompTokens.end(),
547  cbCodeCompletionPlugin::CCToken(-1, contextStr),
548  sortFunctor);
549  if (tknIt != m_AutocompTokens.end() && tknIt->displayName.StartsWith(contextStr))
550  stc->AutoCompSelect(tknIt->displayName);
551  }
552  m_LastACLaunchState[lsTknStart] = tknStart;
553  m_LastACLaunchState[lsCaretStart] = tknEnd;
554 }

642 // cbEVT_EDITOR_TOOLTIP
643 void CCManager::OnEditorTooltip(CodeBlocksEvent& event)
644 {
645  event.Skip();
646 
647  int tooltipMode = Manager::Get()->GetConfigManager(wxT("ccmanager"))->ReadInt(wxT("/tooltip_mode"), 1);
648  if (tooltipMode == 0) // disabled
649  return;
650 
651  // if the event comes from user menu click, then its String field isn't empty
652  // @see CCManager::OnMenuSelect() for details
653  bool fromMouseDwell = event.GetString().IsEmpty();
654  if (wxGetKeyState(WXK_CONTROL) && fromMouseDwell)
655  return;
656  if (tooltipMode == 3 && fromMouseDwell) // keybound only
657  return;
658 
659  EditorBase* base = event.GetEditor();
660  cbEditor* ed = base && base->IsBuiltinEditor() ? static_cast<cbEditor*>(base) : nullptr;
661  if (!ed || ed->IsContextMenuOpened())
662  return;
663 
664  cbStyledTextCtrl* stc = ed->GetControl();
665  cbCodeCompletionPlugin* ccPlugin = GetProviderFor(ed);
666  int pos = stc->PositionFromPointClose(event.GetX(), event.GetY());
667  if (!ccPlugin || pos < 0 || pos >= stc->GetLength())
668  {
669  if (stc->CallTipActive() && event.GetExtraLong() == 0 && m_CallTipActive == wxSCI_INVALID_POSITION)
670  static_cast<wxScintilla*>(stc)->CallTipCancel();
671  return;
672  }
673 
674  int hlStart, hlEnd, argsPos;
675  hlStart = hlEnd = argsPos = wxSCI_INVALID_POSITION;
676  bool allowCallTip = true;
677  const std::vector<cbCodeCompletionPlugin::CCToken>& tokens = ccPlugin->GetTokenAt(pos, ed, allowCallTip);
678  std::set<wxString> uniqueTips;
679  for (size_t i = 0; i < tokens.size(); ++i)
680  uniqueTips.insert(tokens[i].displayName);
681  wxStringVec tips(uniqueTips.begin(), uniqueTips.end());
682 
683  const int style = event.GetInt();
684  if (!tips.empty())
685  {
686  const int tknStart = stc->WordStartPosition(pos, true);
687  const int tknEnd = stc->WordEndPosition(pos, true);
688  if (tknEnd - tknStart > 2)
689  {
690  for (size_t i = 0; i < tips[0].Length(); ++i)
691  {
692  size_t hlLoc = tips[0].find(stc->GetTextRange(tknStart, tknEnd), i);
693  if (hlLoc == wxString::npos)
694  break;
695  hlStart = hlLoc;
696  hlEnd = hlStart + tknEnd - tknStart;
697  if ( (hlStart > 0 && (tips[0][hlStart - 1] == wxT('_') || wxIsalpha(tips[0][hlStart - 1])))
698  || (hlEnd < static_cast<int>(tips[0].Length()) - 1 && (tips[0][hlEnd] == wxT('_') || wxIsalpha(tips[0][hlEnd]))) )
699  {
700  i = hlEnd;
701  hlStart = hlEnd = wxSCI_INVALID_POSITION;
702  }
703  else
704  break;
705  }
706  }
707  }
708  else if ( allowCallTip
709  && !( stc->IsString(style)
710  || stc->IsComment(style)
711  || stc->IsCharacter(style)
712  || stc->IsPreprocessor(style) ) )
713  {
714  const int line = stc->LineFromPosition(pos);
715  if (pos + 4 > stc->PositionFromLine(line) + (int)ed->GetLineIndentString(line).Length())
716  {
717  const CallTipVec& cTips = ccPlugin->GetCallTips(pos, style, ed, argsPos);
718  for (size_t i = 0; i < cTips.size(); ++i)
719  tips.push_back(cTips[i].tip);
720  if (!tips.empty())
721  {
722  hlStart = cTips[0].hlStart;
723  hlEnd = cTips[0].hlEnd;
724  }
725  }
726  }
727  if (tips.empty())
728  {
729  if (stc->CallTipActive() && event.GetExtraLong() == 0 && m_CallTipActive == wxSCI_INVALID_POSITION)
730  static_cast<wxScintilla*>(stc)->CallTipCancel();
731  }
732  else
733  {
734  DoShowTips(tips, stc, pos, argsPos, hlStart, hlEnd);
735  event.SetExtraLong(1);
736  }
737  m_CallTipActive = wxSCI_INVALID_POSITION;
738 }

899 // cbEVT_SHOW_CALL_TIP
900 // There are some caches used in this function
901 // see the comment "search long term recall" and "search short term recall"
902 // They remember which page on the calltip was last selected, and show that again first next time it
903 // is requested.
904 // Short term recall caches the current calltip, so it can be displayed exactly if a calltip is
905 // re-requested in the same location.
906 // Long term recall uses two dictionaries, the first one based on the tip's content, and will
907 // display the last shown page when a new calltip is requested that has the same content.
908 // The second (fuzzy) dictionary remembers which page was shown based on the typed word prefix
909 // (this is to support FortranProject, which uses more dynamic calltips).
910 void CCManager::OnShowCallTip(CodeBlocksEvent& event)
911 {
912  event.Skip();
913 
914  int tooltipMode = Manager::Get()->GetConfigManager(wxT("ccmanager"))->ReadInt(wxT("/tooltip_mode"), 1);
915  // 0 - disable
916  // 1 - enable
917  // 2 - force single page
918  // 3 - keybound only
919  if (tooltipMode == 0)
920  return;
921 
922  cbEditor* ed = Manager::Get()->GetEditorManager()->GetBuiltinActiveEditor();
923  if (!ed)
924  return;
925 
926  cbCodeCompletionPlugin* ccPlugin = GetProviderFor(ed);
927  if (!ccPlugin)
928  return;
929 
930  cbStyledTextCtrl* stc = ed->GetControl();
931  if (!stc)
932  return;
933 
934  int pos = stc->GetCurrentPos();
935  int argsPos = wxSCI_INVALID_POSITION;
936  // save the current tip shown text for later recalling
937  wxString curTip;
938  // check whether m_CurCallTip is invalid(point to the end of m_CallTips)
939  if (!m_CallTips.empty() && m_CurCallTip != m_CallTips.end())
940  curTip = m_CurCallTip->tip;
941 
942  m_CallTips = ccPlugin->GetCallTips(pos, stc->GetStyleAt(pos), ed, argsPos);
943  // since m_CallTips get updated, we should update the m_CurCallTip, they are done in
944  // the following if/else statement.
945  if (!m_CallTips.empty() && (event.GetInt() != FROM_TIMER || argsPos == m_CallTipActive))
946  {
947  int lnStart = stc->PositionFromLine(stc->LineFromPosition(pos));
948  while (wxIsspace(stc->GetCharAt(lnStart)))
949  ++lnStart; // do not show too far left on multi-line call tips
950  if ( m_CallTips.size() > 1
951  && tooltipMode == 2 ) // force single page
952  {
953  wxString tip;
954  int hlStart, hlEnd;
955  hlStart = hlEnd = wxSCI_INVALID_POSITION;
956  for (CallTipVec::const_iterator itr = m_CallTips.begin();
957  itr != m_CallTips.end(); ++itr)
958  {
959  if (hlStart == hlEnd && itr->hlStart != itr->hlEnd)
960  {
961  hlStart = tip.Length() + itr->hlStart;
962  hlEnd = tip.Length() + itr->hlEnd;
963  }
964  tip += itr->tip + wxT('\n');
965  }
966  m_CallTips.clear();
967  m_CallTips.push_back(cbCodeCompletionPlugin::CCCallTip(tip.RemoveLast(), hlStart, hlEnd));
968  }
969  m_CurCallTip = m_CallTips.begin();
970  if (m_CallTips.size() > 1)
971  {
972  // search long term recall
973  // convert the wxString to a hash value, then search the hash value map cache
974  // thus we can avoid directly lookup the cache by wxString
975  std::map<int, size_t>::const_iterator choiceItr =
976  m_CallTipChoiceDict.find(CCManagerHelper::CallTipToInt(m_CurCallTip->tip, m_CallTips.size()));
977 
978  // choiceItr is an iterator in the Dict, and choiceItr->second is the zero based index in m_CallTips
979  if (choiceItr != m_CallTipChoiceDict.end() && choiceItr->second < m_CallTips.size())
980  m_CurCallTip = m_CallTips.begin() + choiceItr->second;
981 
982  if (choiceItr == m_CallTipChoiceDict.end() || argsPos == m_CallTipActive)
983  {
984  int prefixEndPos = argsPos;
985  while (prefixEndPos > 0 && wxIsspace(stc->GetCharAt(prefixEndPos - 1)))
986  --prefixEndPos;
987 
988  // convert the prefix string to hash, check to see whether the hash is in the cache
989  const wxString& prefix = stc->GetTextRange(stc->WordStartPosition(prefixEndPos, true), prefixEndPos);
990  choiceItr = m_CallTipFuzzyChoiceDict.find(CCManagerHelper::CallTipToInt(prefix, m_CallTips.size()));
991  if (choiceItr != m_CallTipFuzzyChoiceDict.end() && choiceItr->second < m_CallTips.size())
992  m_CurCallTip = m_CallTips.begin() + choiceItr->second;
993  }
994  // search short term recall
995  for (CallTipVec::const_iterator itr = m_CallTips.begin();
996  itr != m_CallTips.end(); ++itr)
997  {
998  if (itr->tip == curTip)
999  {
1000  m_CurCallTip = itr;
1001  break;
1002  }
1003  }
1004  }
1005  m_CallTipActive = argsPos;
1006  DoUpdateCallTip(ed);
1007  }
1008  else if (m_CallTipActive != wxSCI_INVALID_POSITION)
1009  {
1010  static_cast<wxScintilla*>(stc)->CallTipCancel();
1011  m_CallTipActive = wxSCI_INVALID_POSITION;
1012  // make m_CurCallTip invalid
1013  m_CurCallTip = m_CallTips.end();
1014  }
1015 }

1317 void CCManager::DoShowTips(const wxStringVec& tips, cbStyledTextCtrl* stc, int pos, int argsPos, int hlStart, int hlEnd)
1318 {

1320  int marginWidth = stc->GetMarginWidth(wxSCI_MARGIN_SYMBOL) + stc->GetMarginWidth(wxSCI_MARGIN_NUMBER);
1321  int maxWidth = (stc->GetSize().x - marginWidth) / stc->TextWidth(wxSCI_STYLE_LINENUMBER, wxT("W")) - 1;
1322  maxWidth = std::min(std::max(60, maxWidth), 135);
1323  wxString tip;
1324  int lineCount = 0;
1325  wxString lineBreak = wxT('\n');
1326  if (!tips.front().IsEmpty() && tips.front()[0] <= wxT('\002'))
1327  {
1328  lineBreak += wxT(' ');
1329  if (tips.front().Length() > 1 && tips.front()[1] <= wxT('\002'))
1330  lineBreak += wxT(" ");
1331  }
1332 
1333  for (size_t i = 0; i < tips.size() && lineCount < maxLines; ++i)
1334  {
1335  if (tips[i].Length() > (size_t)maxWidth + 6) // line is too long, try breaking it
1336  {
1337  wxString tipLn = tips[i];
1338  while (!tipLn.IsEmpty())
1339  {
1340  wxString segment = tipLn.Mid(0, maxWidth);
1341  int index = segment.Find(wxT(' '), true); // break on a space
1342  if (index < 20) // no reasonable break?
1343  {
1344  segment = tipLn.Mid(0, maxWidth * 6 / 5); // increase search width a bit
1345  index = segment.Find(wxT(' '), true);
1346  }
1347  for (int commaIdx = index - 1; commaIdx > maxWidth / 2; --commaIdx) // check back for a comma
1348  {
1349  if (segment[commaIdx] == wxT(',') && segment[commaIdx + 1] == wxT(' '))
1350  {
1351  index = commaIdx + 1; // prefer splitting on a comma, if that does not set us back too far
1352  break;
1353  }
1354  }
1355  if (index < 20 || segment == tipLn) // end of string, or cannot split
1356  {
1357  tip += tipLn + lineBreak;
1358  CCManagerHelper::RipplePts(hlStart, hlEnd, tip.Length(), lineBreak.Length());
1359  tipLn.Clear();
1360  }
1361  else // continue splitting
1362  {
1363  tip += segment.Mid(0, index) + lineBreak + wxT(' ');
1364  CCManagerHelper::RipplePts(hlStart, hlEnd, tip.Length(), lineBreak.Length() + 1);
1365  // already starts with a space, so all subsequent lines are prefixed by two spaces
1366  tipLn = tipLn.Mid(index);
1367  }
1368  ++lineCount;
1369  }
1370  }
1371  else // just add the line
1372  {
1373  tip += tips[i] + lineBreak;
1374  CCManagerHelper::RipplePts(hlStart, hlEnd, tip.Length(), lineBreak.Length());
1375  ++lineCount;
1376  }
1377  }
1378  tip.Trim(); // trailing linefeed
1379 
1380  // try to show the tip at the start of the token/arguments, or at the margin if we are scrolled right
1381  // an offset of 2 helps deal with the width of the folding bar (TODO: does an actual calculation exist?)
1382  int line = stc->LineFromPosition(pos);
1383  if (argsPos == wxSCI_INVALID_POSITION)
1384  argsPos = stc->WordStartPosition(pos, true);
1385  else
1386  argsPos = std::min(CCManagerHelper::FindColumn(line, stc->GetColumn(argsPos), stc), stc->WordStartPosition(pos, true));
