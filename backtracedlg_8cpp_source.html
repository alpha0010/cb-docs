78 void BacktraceDlg::Reload()
79 {
80  cbDebuggerPlugin *plugin = Manager::Get()->GetDebuggerManager()->GetActiveDebugger();
81  if (!plugin)
82  return;
83 
84  ColourManager &colours = *Manager::Get()->GetColourManager();
85  const wxColour &activeBackground = colours.GetColour(wxT("dbg_backtrace_active_background"));
86  const wxColour &activeForeground = colours.GetColour(wxT("dbg_backtrace_active_foreground"));
87 
88  m_list->Freeze();
89  m_list->DeleteAllItems();
90 
91  int active_frame = plugin->GetActiveStackFrame();
92  int active_frame_index = 0;
93  for (int ii = 0; ii < plugin->GetStackFrameCount(); ++ii)
94  {
95  cb::shared_ptr<const cbStackFrame> frame = plugin->GetStackFrame(ii);
96  wxString addr = frame->GetAddressAsString();
97  wxString num;
98  num.Printf(wxT("%d"), frame->GetNumber());
99  int idx = m_list->InsertItem(m_list->GetItemCount(), num);
100  m_list->SetItem(idx, 1, addr);
101  m_list->SetItem(idx, 2, frame->GetSymbol());
102  m_list->SetItem(idx, 3, frame->GetFilename());
103  m_list->SetItem(idx, 4, frame->GetLine());
104 
105  if (active_frame == frame->GetNumber())
106  {
107  active_frame_index = ii;
108  m_list->SetItemBackgroundColour(ii, activeBackground);
109  m_list->SetItemTextColour(ii, activeForeground);
110  }
111  }
112 
113  if (active_frame_index < m_list->GetItemCount())
114  m_list->EnsureVisible(active_frame_index);
115  m_list->Thaw();
116  m_list->SetColumnWidth(0, 32);
117 
118  for (int i = 1; i < 4; ++i)
119  m_list->SetColumnWidth(i, wxLIST_AUTOSIZE);
120 }

149 void BacktraceDlg::OnJump(cb_unused wxCommandEvent& event)
150 {
151  if (m_list->GetSelectedItemCount() == 0)
152  return;
153 
154  // find selected item index
155  int index = m_list->GetNextItem(-1, wxLIST_NEXT_ALL, wxLIST_STATE_SELECTED);
156 
157  wxListItem info;
158  info.m_itemId = index;
159  info.m_mask = wxLIST_MASK_TEXT;
160  info.m_col = 3;
161  wxString file = m_list->GetItem(info) ? info.m_text : _T("");
162  info.m_col = 4;
163  wxString line = m_list->GetItem(info) ? info.m_text : _T("");
164  if (!file.IsEmpty() && !line.IsEmpty())
165  {
166  long line_number;
167  line.ToLong(&line_number, 10);
168 
169 
170  cbDebuggerPlugin *plugin = Manager::Get()->GetDebuggerManager()->GetActiveDebugger();
171  if(plugin)
172  plugin->SyncEditor(file, line_number, false);
173  }
174 }

176 void BacktraceDlg::OnSwitchFrame(cb_unused wxCommandEvent& event)
177 {
178  if (!IsSwitchFrameEnabled())
179  return;
180 
181  if (m_list->GetSelectedItemCount() == 0)
182  return;
183 
184  // find selected item index
185  int index = m_list->GetNextItem(-1, wxLIST_NEXT_ALL, wxLIST_STATE_SELECTED);
186  // read the frame number from the first column
187  long realFrameNr;
188  if (m_list->GetItemText(index).ToLong(&realFrameNr))
189  {
190  // switch to this frame
191  cbDebuggerPlugin *plugin = Manager::Get()->GetDebuggerManager()->GetActiveDebugger();
192  if (plugin)
193  plugin->SwitchToFrame(realFrameNr);
194  }
195  else
196  cbMessageBox(_("Couldn't find out the frame number!"), _("Error"), wxICON_ERROR);
197 }

209 void BacktraceDlg::OnSave(cb_unused wxCommandEvent& event)
210 {

219

221  {
222  wxListItem info;
223  info.m_itemId = ii;
224  info.m_col = 1;
225  info.m_mask = wxLIST_MASK_TEXT;
226  wxString addr = m_list->GetItem(info) && !info.m_text.IsEmpty() ? info.m_text : _T("??");
227  info.m_col = 2;
228  wxString func = m_list->GetItem(info) && !info.m_text.IsEmpty() ? info.m_text : _T("??");
229  info.m_col = 3;
230  wxString file = m_list->GetItem(info) && !info.m_text.IsEmpty() ? info.m_text : _T("??");
231  info.m_col = 4;
232  wxString line = m_list->GetItem(info) && !info.m_text.IsEmpty() ? info.m_text : _T("??");
233 
234  text << _T('#') << m_list->GetItemText(ii) << _T(' ')
235  << addr << _T('\t')
236  << func << _T(' ')
237  << _T('(') << file << _T(':') << line << _T(')')
