64 int ParserThreadedTask::Execute()
65 {
66  TRACE(_T("ParserThreadedTask::Execute(): Enter"));
67  if (!m_Parser) return 0;
68 
69  CC_LOCKER_TRACK_P_MTX_LOCK(m_ParserMutex)
70 
71  wxString preDefs(m_Parser->m_PredefinedMacros);
72  StringList batchFiles(m_Parser->m_BatchParseFiles);
73 
74  CC_LOCKER_TRACK_P_MTX_UNLOCK(m_ParserMutex);
75 
76  TRACE(_T("ParserThreadedTask::Execute(): Parse predefined macros(in buffer)"));
77  if (!preDefs.IsEmpty())
78  m_Parser->ParseBuffer(preDefs, false, false);
79 
80  m_Parser->ClearPredefinedMacros();
81 
82  if (m_Parser->m_IgnoreThreadEvents)
83  m_Parser->m_IsFirstBatch = true;
84 
85  TRACE(_T("ParserThreadedTask::Execute(): Parse source files"));
86  while (!batchFiles.empty())
87  {
88  TRACE(_T("-ParserThreadedTask::Execute(): Parse %s"), batchFiles.front().wx_str());
89  m_Parser->Parse(batchFiles.front()); // bool isLocal = true, bool locked = false
90  batchFiles.pop_front();
91  }
92 
93  CC_LOCKER_TRACK_P_MTX_LOCK(m_ParserMutex)
94 
95  m_Parser->m_BatchParseFiles.clear();
96 
97  if (m_Parser->m_IgnoreThreadEvents)
98  {
99  m_Parser->m_IgnoreThreadEvents = false; // we need to hear the pool finish event
100  m_Parser->m_IsParsing = true;
101  }
102 
103  CC_LOCKER_TRACK_P_MTX_UNLOCK(m_ParserMutex);
104  TRACE(_T("ParserThreadedTask::Execute(): Leave"));
105 
106  return 0;
107 }
