84 void EditPathDlg::OnBrowse(cb_unused wxCommandEvent& event)
85 {
86  wxFileName path;
87  wxArrayString multi;
88 
89  wxString val = XRCCTRL(*this, "txtPath", wxTextCtrl)->GetValue();
90  int idx = val.Find(DEFAULT_ARRAY_SEP);
91  if (idx != -1)
92  val.Remove(idx);
93  wxFileName fname(val);
94 
95  if (m_WantDir)
96  {
97  // try to "decode" custom var
98  wxString initial_val = val;
99  Manager::Get()->GetMacrosManager()->ReplaceEnvVars(val);
100  fname = val;
101  fname.MakeAbsolute(m_Basepath);
102  m_Path = fname.GetFullPath();
103 
104  path = ChooseDirectory(this, m_Message, (m_Path.IsEmpty() ? s_LastPath : m_Path),
105  m_Basepath, false, m_ShowCreateDirButton);
106 
107  if (path.GetFullPath().IsEmpty())
108  return;
109 
110  // if it was a custom var, see if we can re-insert it
111  if (initial_val != val)
112  {
113  wxString tmp = path.GetFullPath();
114  if (tmp.Replace(val, initial_val) != 0)
115  {
116  // done here
117  XRCCTRL(*this, "txtPath", wxTextCtrl)->SetValue(tmp);
118  return;
119  }
120  }
121  }
122  else
123  {
124  wxFileDialog dlg(this, m_Message, (fname.GetPath().IsEmpty() ? s_LastPath : fname.GetPath()),
125  fname.GetFullName(), m_Filter, wxFD_CHANGE_DIR | (m_AllowMultiSel ? wxFD_MULTIPLE : 0) );
126 
127  PlaceWindow(&dlg);
128  if (dlg.ShowModal() == wxID_OK)
129  {
130  if (m_AllowMultiSel)
131  dlg.GetPaths(multi);
132  else
133  path = dlg.GetPath();
134  }
135  else
136  return;
137  }
138 
139  if (m_AllowMultiSel && multi.GetCount() != 0 && !multi[0].IsEmpty())
140  s_LastPath = multi[0];
141  else if (!path.GetFullPath().IsEmpty())
142  s_LastPath = path.GetFullPath();
143 
144  wxString result;
145  if (m_AskMakeRelative && !m_Basepath.IsEmpty())
146  {
147  // ask the user if he wants it to be kept as relative
148  if (cbMessageBox(_("Keep this as a relative path?"),
149  _("Question"),
150  wxICON_QUESTION | wxYES_NO, this) == wxID_YES)
151  {
152  if (m_AllowMultiSel)
153  {
154  for (unsigned int i = 0; i < multi.GetCount(); ++i)
155  {
156  path = multi[i];
157  path.MakeRelativeTo(m_Basepath);
158  multi[i] = path.GetFullPath();
159  }
160  result = GetStringFromArray(multi);
161  }
162  else
163  {
164  path.MakeRelativeTo(m_Basepath);
165  result = path.GetFullPath();
166  }
167  }
168  else
169  {
170  if (m_AllowMultiSel)
171  result = GetStringFromArray(multi);
172  else
173  result = path.GetFullPath();
174  }
175  }
176  else // always absolute path
177  {
178  if (m_AllowMultiSel)
179  result = GetStringFromArray(multi);
180  else
181  result = path.GetFullPath();
182  }
183 
184  // finally set the path
185  XRCCTRL(*this, "txtPath", wxTextCtrl)->SetValue(result);
186 }
