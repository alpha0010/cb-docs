936 // Checks whether 'suffix' could be a suffix of 'path' and therefore represents
937 // the same path. This is used to check whether a relative path could represent
938 // the same path as absolute path. For instance, for
939 // suffix = sdk/globals.cpp
940 // path = /home/user/codeblocks/trunk/src/sdk/globals.cpp
941 // it returns true. The function expects that 'path' is normalized and compares
942 // 'path' with 'suffix' starting from the end of the path. When it reaches .. in
943 // 'suffix' it gives up (there is no way to check relative filename names
944 // exactly) and if the path compared so far is identical, it returns true
945 bool IsSuffixOfPath(wxFileName const & suffix, wxFileName const & path)
946 {
947  if (path.GetFullName() != suffix.GetFullName())
948  {
949  return false;
950  }
951 
952  wxArrayString suffixDirArray = suffix.GetDirs();
953  wxArrayString pathDirArray = path.GetDirs();
954 
955  int j = pathDirArray.GetCount() - 1;
956  for (int i = suffixDirArray.GetCount() - 1; i >= 0; i--)
957  {
958  // skip paths like /./././ and ////
959  if (suffixDirArray[i] == _T(".") || suffixDirArray[i] == _T(""))
960  continue;
961 
962  // suffix has more directories than path - cannot represent the same path
963  if (j < 0)
964  return false;
965 
966  // suffix contains ".." - from now on we cannot precisely determine
967  // whether suffix and path match - we assume that they do
968  if (suffixDirArray[i] == _T(".."))
969  return true;
970  // the corresponding directories of the two paths differ
971  else if (suffixDirArray[i] != pathDirArray[j])
972  return false;
973 
974  j--;
975  }
976 
977  if (suffix.IsAbsolute() && (j >= 0 || suffix.GetVolume() != path.GetVolume()))
978  return false;
979 
980  // 'suffix' is a suffix of 'path'
981  return true;
982 }
