944 // Checks whether 'suffix' could be a suffix of 'path' and therefore represents
945 // the same path. This is used to check whether a relative path could represent
946 // the same path as absolute path. For instance, for
947 // suffix = sdk/globals.cpp
948 // path = /home/user/codeblocks/trunk/src/sdk/globals.cpp
949 // it returns true. The function expects that 'path' is normalized and compares
950 // 'path' with 'suffix' starting from the end of the path. When it reaches .. in
951 // 'suffix' it gives up (there is no way to check relative filename names
952 // exactly) and if the path compared so far is identical, it returns true
953 bool IsSuffixOfPath(wxFileName const & suffix, wxFileName const & path)
954 {
955  if (path.GetFullName() != suffix.GetFullName())
956  {
957  return false;
958  }
959 
960  wxArrayString suffixDirArray = suffix.GetDirs();
961  wxArrayString pathDirArray = path.GetDirs();
962 
963  int j = pathDirArray.GetCount() - 1;
964  for (int i = suffixDirArray.GetCount() - 1; i >= 0; i--)
965  {
966  // skip paths like /./././ and ////
967  if (suffixDirArray[i] == _T(".") || suffixDirArray[i] == _T(""))
968  continue;
969 
970  // suffix has more directories than path - cannot represent the same path
971  if (j < 0)
972  return false;
973 
974  // suffix contains ".." - from now on we cannot precisely determine
975  // whether suffix and path match - we assume that they do
976  if (suffixDirArray[i] == _T(".."))
977  return true;
978  // the corresponding directories of the two paths differ
979  else if (suffixDirArray[i] != pathDirArray[j])
980  return false;
981 
982  j--;
983  }
984 
985  if (suffix.IsAbsolute() && (j >= 0 || suffix.GetVolume() != path.GetVolume()))
986  return false;
987 
988  // 'suffix' is a suffix of 'path'
989  return true;
990 }
