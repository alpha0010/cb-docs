386 bool CCOptionsDlg::ValidateReplacementToken(wxString& from, wxString& to)
387 {
388  // cut off any leading / trailing spaces
389  from.Trim(true).Trim(false);
390  to.Trim(true).Trim(false);
391 
392  if (to.IsEmpty())
393  {
394  // Allow removing a token, but ask the user if that's OK.
395  if (cbMessageBox( _("This setup will replace the token with an empty string.\n"
396  "This will *remove* the token and probably break CC for some cases.\n"
397  "Do you really want to *remove* that token?"),
398  _("Confirmation"),
399  wxICON_QUESTION | wxYES_NO ) == wxID_YES)
400  {
401  return true;
402  }
403  }
404  else if (to.Contains(from))
405  {
406  cbMessageBox(_("Replacement token cannot contain search token.\n"
407  "This would cause an infinite loop otherwise."),
408  _("Error"), wxICON_ERROR);
409  return false;
410  }
411 
412  wxRegEx re(_T("[A-Za-z_]+[0-9]*[A-Za-z_]*"));
413  if (!re.Matches(from))
414  {
415  cbMessageBox(_("Search token can only contain alphanumeric characters and underscores."),
416  _("Error"), wxICON_ERROR);
417  return false;
418  }
419  if (!re.Matches(to))
420  {
421  // Allow replacing with special characters only if the user says it's ok.
422  if (cbMessageBox( _("You are replacing a token with a string that contains\n"
423  "characters other than alphanumeric and underscores.\n"
424  "This could make parsing the file impossible.\n"
425  "Are you sure?"),
426  _("Confirmation"),
427  wxICON_QUESTION | wxYES_NO ) != wxID_YES)
428  {
429  return false;
430  }
431  }
432 
433  return true;
434 }
