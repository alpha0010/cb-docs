118 void IncrementalSelectHandler::FilterItems()
119 {
120  m_iterator->Reset();
121 
122  const wxString &inputPattern = m_text->GetValue().Lower();
123  if (inputPattern.empty())
124  {
125  int count = m_iterator->GetTotalCount();
126  for (int ii = 0; ii < count; ++ii)
127  m_iterator->AddIndex(ii);
128 
129  FilterItemsFinalize(*m_list, *m_iterator);
130  return;
131  }
132 
133  // We put a star before and after pattern to find search expression everywhere in path
134  // that is: if user enter "a", it will match "123a", "12a" or "a12".
135  wxString search(wxT("*") + inputPattern + wxT("*"));
136  bool isWord = !inputPattern.empty();
137  for (auto ch : inputPattern)
138  {
139  if (!wxIsalpha(ch))
140  {
141  isWord = false;
142  break;
143  }
144  }
145 
146  std::vector<int> indices, promoted;
147  indices.reserve(100);
148  promoted.reserve(100);
149 
150  for (int i = 0; i < m_iterator->GetTotalCount(); ++i)
151  {
152  wxString const &item = m_iterator->GetItemFilterString(i).Lower();
153  if (item.Matches(search.c_str()))
154  {
155  // If the search pattern doesn't contain non alpha characters and it matches at the start of the word in
156  // the item string then promote these items to the top of the list. The order is preserved.
157  if (isWord)
158  {
159  size_t pos = 0, newPos;
160  bool isPromoted = false;
161 
162  while ((newPos = item.find(inputPattern, pos)) != wxString::npos)
163  {
164  if (newPos == 0)
165  {
166  isPromoted = true;
167  break;
168  }
169 
170  if (!wxIsalpha(item[newPos - 1]))
171  {
172  isPromoted = true;
173  break;
174  }
175 
176  // Move one character forward to prevent the same string to be found again.
177  pos = newPos + 1;
178  }
179  if (isPromoted)
180  promoted.push_back(i);
181  else
182  indices.push_back(i);
183  }
184  else
185  indices.push_back(i);
186  }
187  }
188 
189  for (auto i : promoted)
190  m_iterator->AddIndex(i);
191  for (auto i : indices)
192  m_iterator->AddIndex(i);
193 
194  FilterItemsFinalize(*m_list, *m_iterator);
195 }

201 static wxStandardID KeyDownAction(wxKeyEvent& event, int &selected, int selectedMax)
202 {
203  // now, adjust position from key input
204  switch (event.GetKeyCode())
205  {
206  case WXK_RETURN:
207  case WXK_NUMPAD_ENTER:
208  return wxID_OK;
209 
210  case WXK_ESCAPE:
211  return wxID_CANCEL;
212 
213  case WXK_UP:
214  case WXK_NUMPAD_UP:
215  if (selected)
216  selected--;
217  break;
218 
219  case WXK_DOWN:
220  case WXK_NUMPAD_DOWN:
221  selected++;
222  break;
223 
224  case WXK_PAGEUP:
225  case WXK_NUMPAD_PAGEUP:
226  selected -= 10;
227  break;
228 
229  case WXK_PAGEDOWN:
230  case WXK_NUMPAD_PAGEDOWN:
231  selected += 10;
232  break;
233 
234  case WXK_HOME:
235  if (wxGetKeyState(WXK_CONTROL))
236  selected = 0;
237  else
238  event.Skip();
239  break;
240 
241  case WXK_END:
242  if (wxGetKeyState(WXK_CONTROL))
243  selected = selectedMax;
244  else
245  event.Skip();
246  break;
247 
248  default:
249  event.Skip();
250  break;
251  }
252 
253  // Clamp value below 0 and above Max
254  if (selected < 0)
255  selected = 0;
256  else if (selected > selectedMax)
257  selected = selectedMax;
258 
259  return wxID_LOWEST;
260 }
