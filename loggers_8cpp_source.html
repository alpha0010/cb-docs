27 // Helper function which blends a colour with the default window text colour,
28 // so that text will be readable in bright and dark colour schemes
29 wxColour BlendTextColour(wxColour col)
30 {
31  wxColour fgCol = wxSystemSettings::GetColour(wxSYS_COLOUR_WINDOWTEXT);
32  wxColour bgCol = wxSystemSettings::GetColour(wxSYS_COLOUR_WINDOW);
33 
34  int dist = (fgCol.Red()*fgCol.Red() + fgCol.Green()*fgCol.Green() + fgCol.Blue()*fgCol.Blue())
35  - (bgCol.Red()*bgCol.Red() + bgCol.Green()*bgCol.Green() + bgCol.Blue()*bgCol.Blue());
36  if (dist > 0)
37  {
38  // If foreground color is brighter than background color, this is a dark theme, so
39  // brighten the text colour.
40  // I would use wxColour::changeLightness(), but it's only available in v2.9.0 or later.
41  int d = int(sqrt(dist)/4);
42  int r = col.Red() + d;
43  int g = col.Green() + d;
44  int b = col.Blue() + d;
45  return wxColour( r>255 ? 255 : r, g>255 ? 255 : g, b>255 ? 255 : b );
46  }
47  return col;
48 }

194 void TextCtrlLogger::ToggleWrapMode()
195 {
196  if (!control || !HasFeature(Feature::IsWrappable))
197  return;
198 
199  long ctrl_style = control->GetWindowStyle();
200 
201  // wxTE_DONTWRAP is an equivalent for wxHSCROLL (see <wx/textctrl.h>)
202  bool is_dontwrap = ((ctrl_style & wxTE_DONTWRAP)==wxTE_DONTWRAP);
203  if (is_dontwrap)
204  ctrl_style &= ~wxTE_DONTWRAP; // don't wrap = OFF means wrapping = ON
205  else
206  ctrl_style |= wxTE_DONTWRAP; // don't wrap = ON means wrapping = OFF
207 
208 // On Windows the wrap-style can not easily be changed on the fly, but if the align flags
209 // change wxWidgets recreates the textcontrol, doing this ourselves seems not to be so easy,
210 // so we use this hack here (testing if the centred-flag is set, is not needed for our loggers)
211 #ifdef __WXMSW__
212  bool is_centred = ((ctrl_style & wxALIGN_CENTER)==wxALIGN_CENTER);
213  if (is_centred)
214  {
215  ctrl_style &= ~wxALIGN_CENTRE;
216  control->SetWindowStyleFlag(ctrl_style);
217  ctrl_style |= wxALIGN_CENTRE;
218  }
219  else
220  {
221  ctrl_style |= wxALIGN_CENTRE;
222  control->SetWindowStyleFlag(ctrl_style);
223  ctrl_style &= ~wxALIGN_CENTRE;
224  }
225 #endif
226  control->SetWindowStyleFlag(ctrl_style);
227 }
