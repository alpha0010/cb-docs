195 cbNotebookStack* EditorManager::GetNotebookStack()
196 {
197  bool found = false;
198  wxWindow* wnd;
199  cbNotebookStack* body;
200  cbNotebookStack* prev_body;
201 
202  while (m_nNotebookStackSize != m_pNotebook->GetPageCount()) // Sync stack with Notebook
203  {
204  if (m_nNotebookStackSize < m_pNotebook->GetPageCount())
205  {
206  for (size_t i = 0; i<m_pNotebook->GetPageCount(); ++i)
207  {
208  wnd = m_pNotebook->GetPage(i);
209  found = false;
210  for (body = m_pNotebookStackHead->next; body != NULL; body = body->next)
211  {
212  if (wnd == body->window)
213  {
214  found = true;
215  break;
216  }
217  }
218  if (!found)
219  {
220  m_pNotebookStackTail->next = new cbNotebookStack(wnd);
221  m_pNotebookStackTail = m_pNotebookStackTail->next;
222  ++m_nNotebookStackSize;
223  }
224  }
225  }
226  if (m_nNotebookStackSize > m_pNotebook->GetPageCount())
227  {
228  for (prev_body = m_pNotebookStackHead, body = prev_body->next; body != NULL; prev_body = body, body = body->next)
229  {
230  if (m_pNotebook->GetPageIndex(body->window) == wxNOT_FOUND)
231  {
232  prev_body->next = body->next;
233  delete body;
234  --m_nNotebookStackSize;
235  body = prev_body;
236  }
237  }
238  }
239  }
240 
241  return m_pNotebookStackHead->next;
242 }

357 cbEditor* EditorManager::Open(LoaderBase* fileLdr, const wxString& filename, int /*pos*/, ProjectFile* data)
358 {
359  bool can_updateui = !GetActiveEditor() || !Manager::Get()->GetProjectManager()->IsLoading();
360  wxFileName fn(realpath(filename));
361  NormalizePath(fn, wxEmptyString);
362  wxString fname = UnixFilename(fn.GetFullPath());
363  // Manager::Get()->GetLogManager()->DebugLog("Trying to open '%s'", fname.c_str());
364  if (!wxFileExists(fname))
365  return nullptr;
366  // Manager::Get()->GetLogManager()->DebugLog("File exists '%s'", fname.c_str());
367 
368  // disallow application shutdown while opening files
369  // WARNING: remember to set it to true, when exiting this function!!!
370  s_CanShutdown = false;
371 
372  EditorBase* eb = IsOpen(fname);
373  cbEditor* ed = nullptr;
374  if (eb)
375  {
376  if (eb->IsBuiltinEditor())
377  {
378  ed = (cbEditor*)eb;
379  }
380  else
381  {
382  s_CanShutdown = true;
383  return nullptr; // is open but not a builtin editor
384  }
385  }
386 
387  if (!ed)
388  {
389  if (!fileLdr)
390  fileLdr = Manager::Get()->GetFileManager()->Load(filename);
391  if (fileLdr)
392  {
393  ed = new cbEditor(m_pNotebook, fileLdr, fname, m_Theme);
394  if (ed->IsOK())
395  AddEditorBase(ed);
396  else
397  {
398  ed->Destroy();
399  ed = nullptr;
400  }
401  }
402  }
403 
404  // check for ProjectFile
405  if (ed && !ed->GetProjectFile())
406  {
407  // First checks if we're already being passed a ProjectFile as a parameter
408  if (data)
409  Manager::Get()->GetLogManager()->DebugLog(_T("Project data set for ") + data->file.GetFullPath());
410  else
411  Manager::Get()->GetProjectManager()->FindProjectForFile(ed->GetFilename(), &data, false, false);
412  if (data)
413  ed->SetProjectFile(data,true);
414  }
415 
416  if (can_updateui)
417  {
418  if (ed)
419  {
420  SetActiveEditor(ed); // fires the cbEVT_EDITOR_ACTIVATED event
421  ed->GetControl()->SetFocus();
422  }
423  }
424 
425  // we 're done
426  s_CanShutdown = true;
427 
428  return ed;
429 }
