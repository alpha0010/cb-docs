96 wxString EditorBase::CreateUniqueFilename()
97 {
98  const wxString prefix = _("Untitled");
99  const wxString path = wxGetCwd() + wxFILE_SEP_PATH;
100  wxString tmp;
101  int iter = 0;
102  while (true)
103  {
104  tmp.Clear();
105  tmp << path << prefix << wxString::Format(_T("%d"), iter);
106  if (!Manager::Get()->GetEditorManager()->GetEditor(tmp) &&
107  !wxFileExists(path + tmp))
108  {
109  return tmp;
110  }
111  ++iter;
112  }
113 }

221 wxMenu* EditorBase::CreateContextSubMenu(long id) // For context menus
222 {
223  wxMenu* menu = nullptr;
224 
225  if (id == idSwitchTo)
226  {
227  menu = new wxMenu;
228  m_SwitchTo.clear();
229  for (int i = 0; i < EditorMaxSwitchTo && i < Manager::Get()->GetEditorManager()->GetEditorsCount(); ++i)
230  {
231  EditorBase* other = Manager::Get()->GetEditorManager()->GetEditor(i);
232  if (!other || other == this)
233  continue;
234  id = idSwitchFile1+i;
235  m_SwitchTo[id] = other;
236  menu->Append(id, (other->GetModified() ? wxT("*") : wxEmptyString) + other->GetShortName());
237  }
238  if (!menu->GetMenuItemCount())
239  {
240  delete menu;
241  menu = nullptr;
242  }
243  }
244  return menu;
245 }

273 void EditorBase::DisplayContextMenu(const wxPoint& position, ModuleType type)
274 {
275  bool noeditor = (type != mtEditorManager);
276  // noeditor:
277  // True if context menu belongs to open files tree;
278  // False if belongs to cbEditor
279 
280  // inform the editors we 're just about to create a context menu
281  if (!OnBeforeBuildContextMenu(position, type))
282  return;
283 
284  wxMenu* popup = new wxMenu;
285 
286  if (!noeditor && wxGetKeyState(WXK_CONTROL))
287  {
288  cbStyledTextCtrl* control = Manager::Get()->GetEditorManager()->GetBuiltinActiveEditor()->GetControl();
289  wxString text = control->GetSelectedText();
290  if (text.IsEmpty())
291  {
292  const int pos = control->GetCurrentPos();
293  text = control->GetTextRange(control->WordStartPosition(pos, true), control->WordEndPosition(pos, true));
294  }
295 
296  if (!text.IsEmpty())
297  {
298  popup->Append(idGoogle, _("Search the Internet for \"") + text + _T("\""));
299  popup->Append(idMsdn, _("Search MSDN for \"") + text + _T("\""));
300  popup->Append(idStackOverflow, _("Search StackOverflow for \"") + text + _T("\""));
301  popup->Append(idCodeProject, _("Search CodeProject for \"") + text + _T("\""));
302  popup->Append(idCPlusPlusCom, _("Search CplusPlus.com for \"") + text + _T("\""));
303  }
304  lastWord = text;
305 
306  wxMenu* switchto = CreateContextSubMenu(idSwitchTo);
307  if (switchto)
308  {
309  if (popup->GetMenuItemCount() > 0)
310  popup->AppendSeparator();
311  popup->Append(idSwitchTo, _("Switch to"), switchto);
312  }
313  }
314  else if (!noeditor && wxGetKeyState(WXK_ALT))
315  { // run a script
316  }
317  else
318  {
319  // Basic functions
320  BasicAddToContextMenu(popup, type);
321 
322  // Extended functions, part 1 (virtual)
323  AddToContextMenu(popup, type, false);
324 
325  // ask other editors / plugins if they need to add any entries in this menu...
326  FileTreeData* ftd = new FileTreeData(nullptr, FileTreeData::ftdkUndefined);
327  ftd->SetFolder(m_Filename);
328  Manager::Get()->GetPluginManager()->AskPluginsForModuleMenu(type, popup, ftd);
329  delete ftd;
330 
331  popup->AppendSeparator();
332  // Extended functions, part 2 (virtual)
333  AddToContextMenu(popup, type, true);
334  }
335  // inform the editors we 're done creating a context menu (just about to show it)
336  OnAfterBuildContextMenu(type);
337 
338  // display menu
339  wxPoint clientpos;
340  if (position==wxDefaultPosition) // "context menu" key
341  {
342  // obtain the caret point (on the screen) as we assume
343  // that the user wants to work with the keyboard
344  cbStyledTextCtrl* const control = Manager::Get()->GetEditorManager()->GetBuiltinActiveEditor()->GetControl();
345  clientpos = control->PointFromPosition(control->GetCurrentPos());
346  }
347  else
348  {
349  clientpos = ScreenToClient(position);
350  }
351 
352  m_pData->m_DisplayingPopupMenu = true;
353  PopupMenu(popup, clientpos);
354  delete popup;
355  m_pData->m_DisplayingPopupMenu = false;
356 
357  // this code *must* be the last code executed by this function
358  // because it *will* invalidate 'this'
359  if (m_pData->m_CloseMe)
360  Manager::Get()->GetEditorManager()->Close(this);
361 }

363 void EditorBase::OnContextMenuEntry(wxCommandEvent& event)
364 {
365  // we have a single event handler for all popup menu entries
366  // This was ported from cbEditor and used for the basic operations:
367  // Switch to, close, save, etc.
368 
369  const int id = event.GetId();
370  m_pData->m_CloseMe = false;
371 
372  if (id == idCloseMe)
373  {
374  if (m_pData->m_DisplayingPopupMenu)
375  m_pData->m_CloseMe = true; // defer delete 'this' until after PopupMenu() call returns
376  else
377  Manager::Get()->GetEditorManager()->Close(this);
378  }
379  else if (id == idCloseAll)
380  {
381  if (m_pData->m_DisplayingPopupMenu)
382  {
383  Manager::Get()->GetEditorManager()->CloseAllInTabCtrlExcept(this);
384  m_pData->m_CloseMe = true; // defer delete 'this' until after PopupMenu() call returns
385  }
386  else
387  Manager::Get()->GetEditorManager()->CloseAllInTabCtrl();
388  }
389  else if (id == idCloseAllOthers)
390  {
391  Manager::Get()->GetEditorManager()->CloseAllInTabCtrlExcept(this);
392  }
393  else if (id == idSaveMe)
394  {
395  Save();
396  }
397  else if (id == idSaveAll)
398  {
399  Manager::Get()->GetEditorManager()->SaveAll();
400  }
401  else if (id >= idSwitchFile1 && id <= idSwitchFileMax)
402  {
403  // "Switch to..." item
404  EditorBase* const ed = m_SwitchTo[id];
405  if (ed)
406  Manager::Get()->GetEditorManager()->SetActiveEditor(ed);
407 
408  m_SwitchTo.clear();
409  }
410  else
411  {
412  if (id == idGoogle)
413  wxLaunchDefaultBrowser(wxString(_T("http://www.google.com/search?q=")) << URLEncode(lastWord));
414  else if (id == idMsdn)
415  wxLaunchDefaultBrowser(wxString(_T("http://social.msdn.microsoft.com/Search/en-US/?query=")) << URLEncode(lastWord) << _T("&ac=8"));
416  else if (id == idStackOverflow)
417  wxLaunchDefaultBrowser(wxString(_T("http://stackoverflow.com/search?q=")) << URLEncode(lastWord));
418  else if (id == idCodeProject)
419  wxLaunchDefaultBrowser(wxString(_T("http://www.codeproject.com/search.aspx?q=")) << URLEncode(lastWord));
420  else if (id == idCPlusPlusCom)
421  wxLaunchDefaultBrowser(wxString(_T("http://www.cplusplus.com/search.do?q=")) << URLEncode(lastWord));
422  }
423 }
