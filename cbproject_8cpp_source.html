315 void cbProject::CalculateCommonTopLevelPath()
316 {
317  // find the common toplevel path
318  // for simple projects, this might be the path to the project file
319  // for projects where the project file is in a subdir, files will
320  // have ".." in their paths
321  wxString sep = wxFileName::GetPathSeparator();
322  wxFileName base = GetBasePath() + sep;
323  wxString vol = base.GetVolume();
324  bool prjHasUNCName = base.GetFullPath().StartsWith(_T("\\\\"));
325  bool fileHasUNCName = false;
326 
327  Manager::Get()->GetLogManager()->DebugLog(_T("Project's base path: ") + base.GetFullPath());
328 
329  // This loop takes ~30ms for 1000 project files
330  // it's as fast as it can get, considered that it used to take ~1200ms ;)
331  // don't even bother making it faster - you can't :)
332 #ifdef ctlp_measuring
333  wxStopWatch sw;
334 #endif
335  for (FilesList::iterator it = m_Files.begin(); it != m_Files.end(); ++it)
336  {
337  ProjectFile* f = (*it);
338  if (!f)
339  continue;
340 
341  if ( !vol.IsSameAs(f->file.GetVolume()) )
342  continue;
343 
344  fileHasUNCName = f->file.GetFullPath().StartsWith(_T("\\\\"));
345 
346  if ( (!prjHasUNCName && fileHasUNCName)
347  || ( prjHasUNCName && !fileHasUNCName) )
348  {
349  continue;
350  }
351 
352  wxString tmp = f->relativeFilename;
353  wxString tmpbase = m_BasePath;
354 
355  size_t pos = 0;
356  while ( (pos < tmp.Length())
357  && ( (tmp.GetChar(pos) == _T('.'))
358  || (tmp.GetChar(pos) == _T('/'))
359  || (tmp.GetChar(pos) == _T('\\')) ) )
360  {
361  ++pos;
362  }
363  if ( (pos > 0) && (pos < tmp.Length()) )
364  tmpbase << sep << tmp.Left(pos) << sep;
365 
366  wxFileName tmpbaseF(tmpbase); tmpbaseF.Normalize(wxPATH_NORM_DOTS);
367  if ( (tmpbaseF.GetDirCount() < base.GetDirCount())
368  && (base.GetPath().StartsWith(tmpbaseF.GetPath())) )
369  {
370  base = tmpbaseF;
371  }
372  }
373 #ifdef ctlp_measuring
374  Manager::Get()->GetLogManager()->DebugLogError(F(_T("%s::%s:%d took : %d ms"), cbC2U(__FILE__).c_str(),cbC2U(__PRETTY_FUNCTION__).c_str(), __LINE__, (int)sw.Time()));
375 #endif
376 
377  m_CommonTopLevelPath = base.GetFullPath();
378  Manager::Get()->GetLogManager()->DebugLog(_T("Project's common toplevel path: ") + m_CommonTopLevelPath);
379 
380  for (FilesList::iterator it = m_Files.begin(); it != m_Files.end(); ++it)
381  {
382  ProjectFile* f = (*it);
383  if (!f)
384  continue;
385 
386  wxString fileName = f->file.GetFullPath();
387  fileHasUNCName = fileName.StartsWith(_T("\\\\"));
388 
389  if ( (prjHasUNCName && fileHasUNCName)
390  || ( !prjHasUNCName
391  && !fileHasUNCName
392  && vol.IsSameAs(f->file.GetVolume()) ) )
393  {
394  wxFileName relFileCTLP(f->file);
395  relFileCTLP.MakeRelativeTo( m_CommonTopLevelPath );
396  wxFileName relFileBase(f->file);
397  relFileBase.MakeRelativeTo( GetBasePath() );
398 
399  // The commented (old) method to obtain the relativeToCommonTopLevelPath is fast, but does *not* work, if you save
400  // the project on a different drive in a sub-folder of an existing source file on that (different) drive:
401  // I.e.: Project on C:\Folder\Project.cbp has file C:\Folder\SubFolder\foo.cpp and D:\Folder\bar.cpp
402  // Saved the project under D:\Folder\SubFolder\ProjectNew.cbp would cause a wrong computation of bar.cpp otherwise!!!
403 // f->relativeToCommonTopLevelPath = fileName.Right(fileName.Length() - m_CommonTopLevelPath.Length());
404  // Using wxFileName instead, although its costly:
405  f->relativeToCommonTopLevelPath = relFileCTLP.GetFullPath();
406  f->relativeFilename = relFileBase.GetFullPath();
407  }
408  else
409  {
410  f->relativeToCommonTopLevelPath = fileName;
411  f->relativeFilename = fileName;
412  }
413 
414  f->SetObjName(f->relativeToCommonTopLevelPath);
415  }
416 }

742  || ft == ftResource

1672  {
1673  ++pos;

1675

1677  size_t nextPos = pos;
1678  while (nextPos < stringDesc.Length() && stringDesc.GetChar(++nextPos) != _T('/') && stringDesc.GetChar(nextPos) != _T(':'))
1679  ;

1682  if (current.IsEmpty() || current[0] == _T(':')) // abort on invalid case: "node/:attr=val" (consecutive "/:")

1690  if (!sub)

1697  // last node?
