113 void MultiSelectDlg::SelectWildCard(const wxString& wild, bool select, bool clearOld)
114 {
115  if (wild.IsEmpty())
116  return;
117  wxArrayString wilds = GetArrayFromString(wild, _T(";"));
118  wxCheckListBox* lst = XRCCTRL(*this, "lstItems", wxCheckListBox);
119  for (size_t i = 0; i < lst->GetCount(); ++i)
120  {
121  if (clearOld || !lst->IsChecked(i))
122  {
123  wxString entry = lst->GetString(i).Lower();
124  bool MatchesWildCard = false;
125  for (unsigned int x = 0; x < wilds.GetCount(); ++x)
126  {
127  if (entry.Matches(wilds[x].Lower()))
128  {
129  lst->Check(i, select);
130  MatchesWildCard = true;
131  break;
132  }
133  }
134  // did not match the wildcard and was selected ( == in the old list)
135  // and we want those to be removed (clearOld) -> uncheck
136  if (!MatchesWildCard && clearOld && lst->IsChecked(i))
137  lst->Check(i, false);
138  }
139  }
140  UpdateStatus();
141 }

143 void MultiSelectDlg::OnWildcard(cb_unused wxCommandEvent& event)
144 {
145  wxString wild = cbGetTextFromUser(_("Enter a selection wildcard\n(e.g. \"dlg*.cpp\" "
146  "would select all files starting with \"dlg\" and "
147  "ending in \".cpp\")\nSeparate multiple masks with \";\":"),
148  _("Wildcard selection"));
149  if (wild.IsEmpty())
150  return;
151 
152  // Do not ask to un-select before if there are no items selected
153  bool ask_clear = false;
154  wxCheckListBox* lst = XRCCTRL(*this, "lstItems", wxCheckListBox);
155  for (size_t i = 0; i < lst->GetCount(); ++i)
156  {
157  if (lst->IsChecked(i))
158  {
159  ask_clear = true;
160  break;
161  }
162  }
163 
164  bool clear = false;
165  if (ask_clear)
166  clear = cbMessageBox(_("Do you want to clear the previous selections?"),
167  _("Question"),
168  wxICON_QUESTION | wxYES_NO, this) == wxID_YES;
169  SelectWildCard(wild, true, clear);
170 }
