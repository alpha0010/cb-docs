296 bool IsPointerType(wxString type)
297 {
298  type.Trim(true);
299  type.Trim(false);
300 
301  if (type.Contains(wxT("char *")) || type.Contains(wxT("char const *")))
302  return false;
303  else if (type.EndsWith(wxT("*")))
304  return true;
305  else if (type.EndsWith(wxT("* const")))
306  return true;
307  else if (type.EndsWith(wxT("* volatile")))
308  return true;
309  return false;
310 }
