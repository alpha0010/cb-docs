137 void cbStyledTextCtrl::OnKeyDown(wxKeyEvent& event)
138 {
139  m_lastSelectedText = GetSelectedText();
140  bool emulateDwellStart = false;
141 
142  switch ( event.GetKeyCode() )
143  {
144  case _T('I'):
145  {
146  // todo: this feature is undocumented and unexpected, remove?
147  if (event.GetModifiers() == wxMOD_ALT)
148  m_braceShortcutState = true;
149  break;
150  }
151 
152  case WXK_TAB:
153  {
154  if (m_tabSmartJump && event.GetModifiers() == wxMOD_NONE)
155  {
156  if (!AutoCompActive() && m_bracePosition != wxSCI_INVALID_POSITION)
157  {
158  m_lastPosition = GetCurrentPos();
159  GotoPos(m_bracePosition);
160 
161  // Need judge if it's the final brace
162  HighlightRightBrace();
163  if (!m_tabSmartJump && CallTipActive())
164  CallTipCancel();
165  return;
166  }
167  }
168  }
169  break;
170 
171  case WXK_BACK:
172  {
173  if (m_tabSmartJump)
174  {
175  if (!(event.ControlDown() || event.ShiftDown() || event.AltDown()))
176  {
177  const int pos = GetCurrentPos();
178  const int index = s_leftBrace.Find((wxChar)GetCharAt(pos - 1));
179  if (index != wxNOT_FOUND && (wxChar)GetCharAt(pos) == s_rightBrace.GetChar(index))
180  {
181  CharRight();
182  DeleteBack();
183  }
184  }
185  else if (m_lastPosition != wxSCI_INVALID_POSITION && event.ControlDown())
186  {
187  GotoPos(m_lastPosition);
188  m_lastPosition = wxSCI_INVALID_POSITION;
189  return;
190  }
191  }
192  }
193  break;
194 
195  case WXK_RETURN:
196  case WXK_NUMPAD_ENTER:
197  case WXK_ESCAPE:
198  {
199  if (m_tabSmartJump)
200  m_tabSmartJump = false;
201  }
202  break;
203 
204  case WXK_CONTROL:
205  {
206  EmulateDwellStart();
207  emulateDwellStart = true;
208  }
209  break;
210 
211  case WXK_UP:
212  case WXK_DOWN:
213  {
214  if (event.GetModifiers() == wxMOD_NONE)
215  {
216  if (Manager::Get()->GetCCManager()->ProcessArrow(event.GetKeyCode()))
217  return;
218  }
219  }
220  break;
221  default: break;
222  }
223 
224  if (event.ControlDown() && !emulateDwellStart)
225  EmulateDwellStart();
226 
227  event.Skip();
228 }

230 void cbStyledTextCtrl::OnKeyUp(wxKeyEvent& event)
231 {
232  const int keyCode = event.GetKeyCode();
233  switch (keyCode)
234  {
235  case _T('['): // [ {
236  case _T('\''): // ' "
237 #ifdef __WXMSW__
238  case _T('9'): // ( for wxMSW
239 #else
240  case _T('('): // ( for wxGTK
241 #endif
242  {
243  if ( !AllowTabSmartJump() )
244  break;
245 
246  wxChar ch = keyCode;
247  if (event.ShiftDown())
248  {
249  if (keyCode == _T('\''))
250  ch = _T('"');
251  else if (keyCode == _T('9'))
252  ch = _T('(');
253  else if (keyCode == _T('['))
254  ch = _T('{');
255  }
256 
257  int index = s_leftBrace.Find(ch);
258  if (index != wxNOT_FOUND && (wxChar)GetCharAt(GetCurrentPos()) == s_rightBrace.GetChar(index))
259  {
260  const int pos = GetCurrentPos();
261  if (pos != wxSCI_INVALID_POSITION)
262  {
263  m_tabSmartJump = true;
264  m_bracePosition = pos;
265  }
266  }
267  else if (keyCode == _T('\'')) // ' "
268  m_tabSmartJump = false;
269  }
270  break;
271 
272  case _T(']'): // ] }
273 #ifdef __WXMSW__
274  case _T('0'): // ) for wxMSW
275 #else
276  case _T(')'): // ) for wxGTK
277 #endif
278  {
279  if (!AllowTabSmartJump())
280  break;
281  if (keyCode == _T('0') && !event.ShiftDown())
282  break;
283  m_tabSmartJump = false;
284  }
285  break;
286  default: break;
287  }
288 
289  HighlightRightBrace();
290  event.Skip();
291 }

332 void cbStyledTextCtrl::DoBraceCompletion(const wxChar& ch)
333 {
334  const int pos = GetCurrentPos();
335  const int style = GetStyleAt(pos);
336  if (IsComment(style) || IsComment(GetStyleAt(pos - 2)))
337  return; // do nothing
338  if (ch == wxT('\'') || ch == wxT('"'))
339  {
340  if (GetCharAt(pos) == ch)
341  {
342  DeleteBack();
343  CharRight();
344  }
345  else if (!IsString(GetStyleAt(pos - 2)) && !IsCharacter(GetStyleAt(pos - 2)))
346  InsertText(pos, ch);
347  return; // done
348  }
349 
350  if (IsString(style) || IsCharacter(style))
351  return; // do nothing
352 
353  const wxString opBraces(wxT("([{")); const int opBraceIdx = opBraces.Find(ch);
354  const wxString clBraces(wxT(")]}")); const int clBraceIdx = clBraces.Find(ch);
355  if ( (opBraceIdx != wxNOT_FOUND) || (clBraceIdx != wxNOT_FOUND) )
356  {
357  if ( GetCharAt(pos) == ch )
358  {
359  DeleteBack();
360  CharRight();
361  }
362  else if (opBraceIdx != wxNOT_FOUND)
363  {
364  int nextPos = pos;
365  while ( wxIsspace(GetCharAt(nextPos)) && (nextPos < GetLength()) )
366  ++nextPos;
367 
368  if ( ((wxChar)GetCharAt(nextPos) != clBraces[opBraceIdx])
369  || (BraceMatch(nextPos) != wxNOT_FOUND) )
370  {
371  InsertText(pos, clBraces[opBraceIdx]);
372  }
373  }
374  }
375 }

377 bool cbStyledTextCtrl::DoSelectionBraceCompletion(const wxChar& ch)
378 {
379  if (GetLastSelectedText().IsEmpty())
380  return false; // nothing changed
381  const wxString braces(wxT("([{<'\")]}>'\""));
382  const int braceAIdx = braces.Find(ch, true); // from end (so caret is placed after quotes)
383  if (braceAIdx == wxNOT_FOUND)
384  return false; // nothing changed
385  const int braceBIdx = (braceAIdx + (braces.Length() / 2)) % braces.Length();
386  BeginUndoAction();
387  DeleteBack();
388  if (braceAIdx < braceBIdx)
389  InsertText(GetCurrentPos(), braces[braceAIdx] + GetLastSelectedText() + braces[braceBIdx]);
390  else
391  AddText(braces[braceBIdx] + GetLastSelectedText() + braces[braceAIdx]);
392  EndUndoAction();
393  return true; // succeeded
394 }

408 void cbStyledTextCtrl::HighlightRightBrace()
409 {
410  if (m_bracePosition == wxSCI_INVALID_POSITION)
411  return;
412 
413  int pos = GetCurrentPos();
414  if (pos == wxSCI_INVALID_POSITION)
415  return;
416 
417  const static wxColour caretForeground = GetCaretForeground();
418  const static int caretWidth = GetCaretWidth();
419  const int curLine = GetCurrentLine();
420  const int len = GetLength();
421 
422  if (m_tabSmartJump && (curLine == LineFromPosition(m_bracePosition)))
423  {
424  SetIndicatorCurrent(s_indicHighlight);
425  const int indPos = GetLineIndentPosition(curLine);
426  IndicatorClearRange(indPos, GetLineEndPosition(curLine)-indPos);
427  do
428  {
429  if (pos >= len)
430  break;
431 
432  wxString cur((wxChar)GetCharAt(pos));
433  if (cur == _T("\n"))
434  break;
435 
436  int style = GetStyleAt(pos);
437  if (IsComment(style))
438  continue;
439 
440  if (IsString(style) || IsCharacter(style))
441  {
442  const int nextOne = (pos == len) ? GetStyleAt(pos) : GetStyleAt(pos + 1);
443  if (IsCharacter(nextOne) || IsString(nextOne))
444  continue;
445  }
446 
447  if (s_rightBrace.Contains(cur))
448  {
449  SetCaretForeground(wxColour(255, 0, 0));
450  SetCaretWidth(caretWidth + 1);
451 
452  IndicatorSetForeground(s_indicHighlight, wxColour(80, 236, 120));
453  IndicatorSetStyle(s_indicHighlight, wxSCI_INDIC_ROUNDBOX);
454 #ifndef wxHAVE_RAW_BITMAP
455  IndicatorSetUnder(s_indicHighlight, true);
456 #endif
457  SetIndicatorCurrent(s_indicHighlight);
458  IndicatorFillRange(pos, 1);
459  m_bracePosition = pos + 1;
460  return;
461  }
462  }
463  while (++pos);
464  }
465 
466  m_bracePosition = wxSCI_INVALID_POSITION;
467  m_lastPosition = wxSCI_INVALID_POSITION;
468  m_tabSmartJump = false;
469  SetIndicatorCurrent(s_indicHighlight);
470  IndicatorClearRange(0, len);
471 
472  SetCaretForeground(caretForeground);
473  SetCaretWidth(caretWidth);
474 }
