229 void Compiler::MakeValidID()
230 {
231  // basically, make it XML-element compatible
232  // only allow a-z, 0-9, _, and -
233  // (it is already lowercase)
234  // any non-conformant character will be removed
235 
236  wxString newID;
237  if (m_ID.IsEmpty())
238  m_ID = m_Name;
239 
240  size_t pos = 0;
241  while (pos < m_ID.Length())
242  {
243  wxChar ch = m_ID[pos];
244  if (wxIsalnum(ch) || ch == _T('_') || ch == _T('-')) // valid character
245  newID.Append(ch);
246  else if (wxIsspace(ch)) // convert spaces to underscores
247  newID.Append(_T('_'));
248  ++pos;
249  }
250 
251  // make sure it's not starting with a number or a '-'.
252  // if it is, prepend "cb"
253  if (wxIsdigit(newID.GetChar(0)) || newID.GetChar(0) == _T('-'))
254  newID.Prepend(_T("cb"));
255 
256  if (newID.IsEmpty()) // empty? wtf?
257  cbThrow(_T("Can't create a valid compiler ID for ") + m_Name);
258  m_ID = newID.Lower();
259 
260  // check for unique ID
261  if (!IsUniqueID(m_ID))
262  cbThrow(_T("Compiler ID already exists for ") + m_Name);
263  m_CompilerIDs.Add(m_ID);
264 }

741 CompilerLineType Compiler::CheckForWarningsAndErrors(const wxString& line)
742 {
743  if (!m_MultiLineMessages || (m_MultiLineMessages && !m_Error.IsEmpty()))
744  {
745  m_ErrorFilename.Clear();
746  m_ErrorLine.Clear();
747  m_Error.Clear();
748  }
749 
750  for (size_t i = 0; i < m_RegExes.size(); ++i)
751  {
752  RegExStruct& rs = m_RegExes[i];
753  if (!rs.HasRegEx())
754  continue;
755  const wxRegEx &regex = rs.GetRegEx();
756  if (regex.Matches(line))
757  {
758  if (rs.filename > 0)
759  m_ErrorFilename = UnixFilename(regex.GetMatch(line, rs.filename));
760  if (rs.line > 0)
761  m_ErrorLine = regex.GetMatch(line, rs.line);
762  for (int x = 0; x < 3; ++x)
763  {
764  if (rs.msg[x] > 0)
765  {
766  if (!m_Error.IsEmpty())
767  m_Error << _T(" ");
768  m_Error << regex.GetMatch(line, rs.msg[x]);
769  }
770  }
771  return rs.lt;
772  }
773  }
774  return cltNormal; // default return value
