342 {
343  wxListBox* lstTargets = XRCCTRL(*this, "lstBuildTarget", wxListBox);
344 
345  // if no previously selected target, exit
346  if (m_Current_Sel == -1)
347  return;
348 
349  if (force || lstTargets->GetSelection() != m_Current_Sel)
350  {
351  // selected another build target
352  // save changes to the previously selected target
353  wxArrayString array;
354  ProjectBuildTarget* target = m_Project->GetBuildTarget(m_Current_Sel);
355  if (!target)
356  return;
357 
358  wxString platforms = XRCCTRL(*this, "txtPlatform", wxTextCtrl)->GetValue();

360  target->SetPlatforms(p);
361

369

372 
373  fname.Assign(XRCCTRL(*this, "txtDefinitionFileFilename", wxTextCtrl)->GetValue());
374  target->SetDefinitionFileFilename(fname.GetFullPath());
375 
376  fname.Assign(XRCCTRL(*this, "txtWorkingDir", wxTextCtrl)->GetValue());

378

381

394

396  pf->AddBuildTarget(target->GetTitle());

398  pf->RemoveBuildTarget(target->GetTitle());
399  }
