81 void ExamineMemoryDlg::AddHexByte(const wxString& addr, const wxString& hexbyte)
82 {
83 // m_pDbg->Log(_T("AddHexByte(") + addr + _T(", ") + hexbyte + _T(')'));
84  int bcmod = m_ByteCounter % 16;
85 
86  if (m_LastRowStartingAddress == 0)
87  {
88  // because we 'll be appending each row *after* we have consumed it
89  // and then "addr" will point to the next row's starting address,
90  // we 'll keep the current row's starting address in "m_LastRowStartingAddress".
91 
92  // if it's zero (i.e this is the first row), keep "addr" as starting address for this row.
93  // m_LastRowStartingAddress will be set again when we 've consumed this row...
94  addr.ToULong(&m_LastRowStartingAddress, 16);
95  }
96 
97 #define HEX_OFFSET(a) (a*3)
98 #define CHAR_OFFSET(a) (16*3 + 3 + a)
99 
100  unsigned long hb;
101  hexbyte.ToULong(&hb, 16);
102 // m_pDbg->Log(wxString::Format(_T("hb=%d, [0]=%c, [1]=%c"), hb, hexbyte[0], hexbyte[1]));
103 // m_pDbg->Log(wxString::Format(_T("HEX_OFFSET(bcmod)=%d, CHAR_OFFSET(bcmod)=%d"), HEX_OFFSET(bcmod), CHAR_OFFSET(bcmod)));
104  m_LineText[HEX_OFFSET(bcmod)] = hexbyte[0];
105  m_LineText[HEX_OFFSET(bcmod) + 1] = hexbyte[1];
106  m_LineText[CHAR_OFFSET(bcmod)] = hb >= 32 ? wxChar(hb) : wxChar(_T('.'));
107  ++m_ByteCounter;
108 
109  // flush every 16 bytes
110  if (m_ByteCounter != 0 && m_ByteCounter % 16 == 0)
111  {
112  // filled 16 bytes window; append text and reset accumulator array
113  if (m_ByteCounter != 16) // after the first line,
114  m_pText->AppendText(_T('\n')); // prepend a newline
115  m_LineText[23] = _T('|'); // put a "separator" in the middle (just to ease reading a bit)
116 
117  unsigned long a;
118  addr.ToULong(&a, 16);
119  m_pText->AppendText(wxString::Format(_T("0x%lx: %.67s"), m_LastRowStartingAddress, m_LineText));
120  for (int i = 0; i < 67; ++i)
121  m_LineText[i] = _T(' ');
122  // update starting address for next row
123  // add 8 bytes: addr is the start address of the second 8-byte chunk of this line, so next line is +8
124  m_LastRowStartingAddress = a + 8;
125  }
126 }
