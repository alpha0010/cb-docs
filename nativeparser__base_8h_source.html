399  {

401  int nest = 0;

610  bool IsChildOfUnnamedOrEnum(TokenTree* tree, const int targetIdx, const int parentIdx)
611  {
612  if (targetIdx == parentIdx)
613  return true;
614  if (parentIdx == -1)
615  return false;
616 
617  Token* parent = tree->at(parentIdx);
618  if (parent && (parent->m_TokenKind & tkClass))
619  {
620  // loop all children of the parent token
621  for (TokenIdxSet::const_iterator it = parent->m_Children.begin();
622  it != parent->m_Children.end(); ++it)
623  {
624  Token* token = tree->at(*it);
625  // an unnamed class is much similar like the enum
626  if (token && (((token->m_TokenKind & tkClass)
627  && (token->m_IsAnonymous == true))
628  || (token->m_TokenKind & tkEnum)))
629  {
630  // if target token matches on child, we can return success
631  // other wise, we try to see the target token matches child's child.
632  if ((targetIdx == (*it)) || IsChildOfUnnamedOrEnum(tree, targetIdx, (*it)))
633  return true;
