18 class HashForWxStringMap
19 {
20 public:
21  HashForWxStringMap() {}
22  unsigned long operator()(const wxString& x) const
23  {
24  const size_t len = x.length();
25  const size_t intWxChar = sizeof(unsigned int) / sizeof(wxChar);
26  const size_t shortWxChar = sizeof(unsigned short) / sizeof(wxChar);
27  if (len >= intWxChar)
28  return size_t((128 ^ len) + *((unsigned int*)((const wxChar*)x + len - intWxChar)));
29  else if (len >= shortWxChar)
30  return size_t((256 ^ len) + *((unsigned short*)((const wxChar*)x + len - shortWxChar)));
31  else
32  return size_t((512 ^ len) + *((const wxChar*)x + len - 1));
33  }
34  HashForWxStringMap& operator=(const HashForWxStringMap&) { return *this; }
35 };
