140 void GotoFunctionDlg::FillColumn()
141 {
142  // Remove all previous
143  m_ListColumn->DeleteAllItems();
144  for (size_t i = 0; i < m_Result.Count(); ++i)
145  {
146  int parametersStart;
147  int functionStart;
148 
149  // Example :
150  // __declspec(dllexport) int xxxx(int a,void (*b)(int c,unsigned long d),unsigned long e)
151  //
152  // Search for parameters list start : first parenthesis from end counting braces
153  // It won't work for default string parameters/character with brace in
154  int braceCount = 0;
155  parametersStart = 0;
156  for (int j = m_Result[i].Len()-1; j >=0 ; j--)
157  {
158  // New nested
159  if (m_Result[i][j] == _T(')'))
160  braceCount++;
161  else
162  {
163  if (m_Result[i][j] == _T('('))
164  {
165  if (!--braceCount)
166  {
167  // End of nested
168  parametersStart = j;
169  break;
170  }
171  }
172  }
173  }
174  // Search function name start : void dummy(int param);
175  functionStart = m_Result[i].rfind(_T(' '),parametersStart);
176  if (parametersStart == wxNOT_FOUND)
177  {
178  if (functionStart == wxNOT_FOUND)
179  {
180  // Nothing found, just put in function name
181  m_ListColumn->InsertItem(i, _T(""));
182  m_ListColumn->SetItem(i, 1, m_Result[i]);
183  }
184  else
185  {
186  // Fonction and return found
187  m_ListColumn->InsertItem(i, m_Result[i].Left(functionStart));
188  m_ListColumn->SetItem(i, 1, m_Result[i].Mid(functionStart + 1));
189  }
190  }
191  else
192  {
193  if (functionStart == wxNOT_FOUND)
194  {
195  // Function and parameters found
196  m_ListColumn->InsertItem(i, _T(""));
197  m_ListColumn->SetItem(i, 2, m_Result[i].Mid(parametersStart));
198  m_ListColumn->SetItem(i, 1, m_Result[i].Left(parametersStart));
199  }
200  else
201  {
202  // Fonction, return and parameters found
203  m_ListColumn->InsertItem(i, m_Result[i].Left(functionStart));
204  m_ListColumn->SetItem(i, 2, m_Result[i].Mid(parametersStart));
205  m_ListColumn->SetItem(i, 1, m_Result[i].Mid(functionStart + 1, parametersStart - functionStart - 1));
206  }
207  }
208  m_ListColumn->SetItemData(i, m_Indexes[i]);
209  }
210  // Sort item only if no one has been promoted (text typed on search boc)
211  if (!m_Promoted)
212  m_ListColumn->SortItems(SortAscending,(wxIntPtr)this);
213 
214  // AutoFit columns
215  int count = m_ListColumn->GetColumnCount();
216  for (int j = 0; j < count; ++j)
217  m_ListColumn->SetColumnWidth(j, wxLIST_AUTOSIZE);
218 
219 }
