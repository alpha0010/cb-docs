105 void cbAuiNotebook::UpdateTabControlsArray()
106 {
107  cbAuiTabCtrlArray saveTabCtrls = m_TabCtrls;
108  m_TabCtrls.Clear();
109  // first get all tab-controls
110  const size_t tab_Count = GetPageCount();
111  for (size_t i = 0; i < tab_Count; ++i)
112  {
113  wxAuiTabCtrl* tabCtrl = nullptr;
114  int idx = -1;
115  if (FindTab(GetPage(i), &tabCtrl, &idx))
116  {
117  if (tabCtrl && m_TabCtrls.Index(tabCtrl) == wxNOT_FOUND)
118  m_TabCtrls.Add(tabCtrl);
119  }
120  else
121  continue;
122  }
123  bool needEventRebind = m_TabCtrls.GetCount() != saveTabCtrls.GetCount();
124  if (!needEventRebind)
125  {
126  for (size_t i = 0; i < m_TabCtrls.GetCount(); ++i)
127  {
128  if (saveTabCtrls.Index(m_TabCtrls[i]) == wxNOT_FOUND)
129  {
130  needEventRebind = true;
131  break;
132  }
133  }
134  }
135  if (needEventRebind)
136  {
137  ResetTabCtrlEvents();
138  }
139  // make sure the active page is valid and shown
140  for (size_t i = 0; i < m_TabCtrls.GetCount(); ++i)
141  {
142  int pageCount = m_TabCtrls[i]->GetPageCount();
143  if (m_TabCtrls[i]->GetActivePage() < 0 && pageCount > 0)
144  m_TabCtrls[i]->SetActivePage((size_t)0);
145  if (m_TabCtrls[i]->GetActivePage() >= pageCount && pageCount > 0)
146  m_TabCtrls[i]->SetActivePage(pageCount - 1);
147  m_TabCtrls[i]->DoShowHide();
148  }
149 }

373 void cbAuiNotebook::OnTabCtrlMouseWheel(wxMouseEvent& event)
374 {
375  wxAuiTabCtrl* tabCtrl = (wxAuiTabCtrl*)event.GetEventObject();
376  if (!tabCtrl)
377  return;
378  cbAuiNotebook* nb = (cbAuiNotebook*)tabCtrl->GetParent();
379  if (!nb)
380  return;
381 
382  nb->SetSelection(nb->GetPageIndex(tabCtrl->GetWindowFromIdx(tabCtrl->GetActivePage())));
383 
384  bool modkeys = CheckKeyModifier();
385 
386  bool advance = (!s_modToAdvance && !modkeys) || (s_modToAdvance && modkeys);
387 
388  if (advance)
389  nb->AdvanceSelection((event.GetWheelRotation() * s_advanceDirection) < 0);
390  else
391  {
392  size_t tabOffset = tabCtrl->GetTabOffset();
393  size_t lastTabIdx = tabCtrl->GetPageCount()-1;
394  wxWindow* win = nb->GetPage(nb->GetSelection());
395  if (win)
396  {
397  wxClientDC dc(win);
398  if ((event.GetWheelRotation() * s_moveDirection) > 0)
399  {
400  if (!tabCtrl->IsTabVisible(lastTabIdx,tabOffset,&dc,win))
401  tabOffset++;
402  }
403  else
404  {
405  if (tabOffset > 0)
406  tabOffset--;
407  }
408  tabCtrl->SetTabOffset(tabOffset);
409  nb->Refresh();
410  }
411  }

490 void cbAuiNotebook::MinimizeFreeSpace(wxAuiTabCtrl* tabCtrl)
491 {
492  if (!tabCtrl || tabCtrl->GetPageCount() < 2 || !IsWindowReallyShown(this))
493  return;
494 
495  int ctrl_idx = tabCtrl->GetActivePage();
496  wxWindow* win = GetPage(ctrl_idx);
497  if (win)
498  {
499  int tabOffset = tabCtrl->GetTabOffset();
500 
501  wxClientDC dc(win);
502  size_t lastTabIdx = tabCtrl->GetPageCount() - 1;
503 
504  if (!tabCtrl->IsTabVisible(ctrl_idx, tabOffset, & dc, win))
505  {
506  for (int i = lastTabIdx ; i >= 0; --i)
507  {
508  if (tabCtrl->IsTabVisible(ctrl_idx, i, & dc, win))
509  {
510  tabOffset = i;
511  break;
512  }
513  }
514  }
515  while (tabOffset > 0 && tabCtrl->IsTabVisible(lastTabIdx, tabOffset-1, & dc, win))
516  --tabOffset;
517 
518  tabCtrl->SetTabOffset(tabOffset);
519  }
520  tabCtrl->Refresh();
521 }

585 int cbAuiNotebook::GetTabPositionFromIndex(int index)
586 {
587  if (GetPageCount() <= 0)
588  return wxNOT_FOUND;
589 
590  UpdateTabControlsArray();
591 
592  wxAuiTabCtrl* tabCtrl = nullptr;
593  int idx = -1;
594 
595  if (!FindTab(GetPage(index), &tabCtrl, &idx))
596  return wxNOT_FOUND;
597 
598  if (!tabCtrl || idx < 0)
599  return wxNOT_FOUND;
600 
601  int indexOffset = 0;
602  wxAuiPaneInfoArray& all_panes = m_mgr.GetAllPanes();
603  const size_t pane_count = all_panes.GetCount();
604  for (size_t i = 0; i < pane_count; ++i)
605  {
606  wxAuiPaneInfo& pane = all_panes[i];
607  if (pane.name == wxT("dummy"))
608  continue;
609 
610  if (pane.window == GetTabFrameFromTabCtrl(tabCtrl))
611  break;
612 
613  for (size_t j = 0; j < m_TabCtrls.GetCount(); ++j)
614  {
615  if (pane.window == GetTabFrameFromTabCtrl(m_TabCtrls[j]))
616  {
617  indexOffset += m_TabCtrls[j]->GetPageCount();
618  break;
619  }
620  }
621  }
622  return idx + indexOffset;
623 }

625 void cbAuiNotebook::AdvanceSelection(bool forward)
626 {
627  if (GetPageCount() <= 1)
628  return;
629 
630  int currentSelection = GetSelection();
631 
632  wxAuiTabCtrl* tabCtrl = nullptr;
633  int idx = -1;
634 
635  if (!FindTab(GetPage(currentSelection), &tabCtrl, &idx))
636  return;
637 
638  if (!tabCtrl || idx < 0)
639  return;
640 
641  wxWindow* page = nullptr;
642  size_t maxPages = tabCtrl->GetPageCount();
643 
644  forward?idx++:idx--;
645 
646  if (idx < 0)
647  idx = maxPages - 1;
648 
649  if ((size_t)idx < maxPages)
650  page = tabCtrl->GetPage(idx).window;
651 
652  if (!page && maxPages > 0)
653  page = tabCtrl->GetPage(0).window;
654 
655  if (page)
656  {
657  currentSelection = GetPageIndex(page);
658  SetSelection(currentSelection);
659  }
660 }

683  // Build list of panes/tabs
684  wxString tabs, tabsTmp;
685  wxArrayString panes;
686 
687  // first get all tab-controls
688  UpdateTabControlsArray();
689 
690  wxAuiPaneInfoArray& all_panes = m_mgr.GetAllPanes();
691  const size_t pane_count = all_panes.GetCount();
692  for (size_t i = 0; i < pane_count; ++i)
693  {
694  wxAuiPaneInfo& pane = all_panes.Item(i);
695  if (pane.name == wxT("dummy"))
696  continue;
697 
698  wxAuiTabCtrl* tabCtrl = nullptr;
699  for (size_t j = 0; j < m_TabCtrls.GetCount(); ++j)
700  {
701  if (pane.window == GetTabFrameFromTabCtrl(m_TabCtrls.Item(j)))
702  {
703  tabCtrl = m_TabCtrls.Item(j);
704  break;
705  }
706  }
707  if (tabCtrl)
708  {
709  tabsTmp.Clear();
710  // add tab id's
711  size_t page_count = tabCtrl->GetPageCount();
712  for (size_t p = 0; p < page_count; ++p)
713  {
714  wxAuiNotebookPage& page = tabCtrl->GetPage(p);
715  const size_t page_idx = m_tabs.GetIdxFromWindow(page.window);
716 
717  wxString id = UniqueIdFromTooltip(GetPageToolTip(page_idx));
718 
719  // file does not belong to any project, so don't save
720  if (id.BeforeLast(':').empty())
721  continue;
722 
723  // if we save a project (projectTitle non empty), but file does not belong to the project
724  // skip it
725  if (!projectTitle.empty() && id.BeforeLast(':') != projectTitle)
726  continue;
727 
728  if (!tabsTmp.empty())
729  tabsTmp += wxT(",");
730 
731 #if wxCHECK_VERSION(3, 0, 0)
732  if ((int)page_idx == m_curPage)
733 #else
734  if ((int)page_idx == m_curpage)
735 #endif
736  tabsTmp += wxT("*");
737  else if ((int)p == tabCtrl->GetActivePage())
738  tabsTmp += wxT("+");
739 
740  tabsTmp += wxString::Format(wxT("%lu"), static_cast<unsigned long>(page_idx));
741  tabsTmp += wxT(";");
742  tabsTmp += id;
743  }
744  if (!tabsTmp.empty())
745  {
746  if (!tabs.empty())
747  tabs += wxT("|");
748 
749  panes.Add(pane.name);
750  tabs += pane.name;
751  tabs += wxT("=");
752  tabs += tabsTmp;
753  }
754  }
755  }
756  tabs += wxT("@");
757 
758  tabsTmp = m_mgr.SavePerspective();
759 
760  wxArrayString arTabsTmp = GetArrayFromString(tabsTmp, wxT("|"));
761 
762  for (size_t i = arTabsTmp.GetCount(); i > 0 ; )
763  {
764  if (arTabsTmp.Item(--i).StartsWith(wxT("name=")))
765  {
766  wxString strTmp = arTabsTmp.Item(i).AfterFirst('=').BeforeFirst(';');
767  if (strTmp == wxT("dummy"))
768  continue;
769  if (panes.Index(strTmp) < 0)
770  arTabsTmp.RemoveAt(i);
771  }
772  }
773 
774  tabsTmp = GetStringFromArray(arTabsTmp, wxT("|"));
775 
776  // Add frame perspective
777  tabs += tabsTmp;
778 
779  return tabs;

806 bool cbAuiNotebook::LoadPerspective(const wxString& layout, bool mergeLayouts)
807 {
808  if (layout.IsEmpty())
809  return false;
810 
811  wxString tabs = layout.BeforeFirst (wxT ('@') );
812  // Remove all tab ctrls (but still keep them in main index)
813  const size_t tab_count = m_tabs.GetPageCount();
814  for (size_t i = 0; i < tab_count; ++i)
815  {
816  // only remove tabs that are in the layout-string, do not touch others,
817  // so the layout of an already loaded workspace or project will not get destroyed, if the
818  // current layout did not know the tab. The layout file which is loaded last takes precedence,
819  // if there are two or more layout strings containing the actual tab.
820  if ( tabs.Find( UniqueIdFromTooltip(GetPageToolTip(i)) ) < 0 )
821  continue;
822 
823  wxWindow* wnd = m_tabs.GetWindowFromIdx (i);
824 
825  // find out which onscreen tab ctrl owns this tab
826  wxAuiTabCtrl* ctrl;
827  int ctrl_idx;
828  if ( !FindTab(wnd, &ctrl, &ctrl_idx) )
829  return false;
830 
831  // remove the tab from ctrl
832  if ( !ctrl->RemovePage(wnd) )
833  return false;
834  }
835  RemoveEmptyTabFrames();
836  wxString currentLayout;
837  if (mergeLayouts)
838  {
839  currentLayout = m_mgr.SavePerspective();
840  wxString tempLayout;
841  while (!currentLayout.empty())
842  {
843  if ( currentLayout.BeforeFirst('|').StartsWith(_("layout2")) ||
844  currentLayout.BeforeFirst('|').StartsWith(_("name=dummy")) )
845  {
846  currentLayout = currentLayout.AfterFirst(('|'));
847  currentLayout.Trim();
848  currentLayout.Trim(true);
849  }
850  else
851  {
852  wxString pane_part = currentLayout.BeforeFirst('|');
853  pane_part.Trim();
854  pane_part.Trim(true);
855  if (!pane_part.empty())
856  tempLayout += pane_part + wxT("|");
857 
858  currentLayout = currentLayout.AfterFirst('|');
859  currentLayout.Trim();
860  currentLayout.Trim(true);
861  }
862  }
863  currentLayout = tempLayout;
864  if (currentLayout.empty())
865  mergeLayouts = false;
866  }
867 
868  size_t sel_page = 0;
869  int active_tab = 0;
870  bool found = false;
871 
872  wxString frames = layout.AfterFirst (wxT ('@') );
873  // if we load an additional project to an exiting layout, the first new tab always goes into a new frame
874  bool firstTabInCtrl =! currentLayout.empty();
875  // This creates a new tabframe if none exists; a workaround, because we can not directly access
876  // the needed wxTabFrame class, because it is not exported.
877  // This also takes care of all needed pane-info
878  wxAuiTabCtrl* dest_tabs = GetActiveTabCtrl();
879  while (1)
880  {
881  const wxString tab_part = tabs.BeforeFirst(wxT('|'));
882 
883  // if the string is empty, we're done parsing
884  if (tab_part.empty())
885  break;
886 
887  // Get pane name
888  wxString pane_name = tab_part.BeforeFirst(wxT('='));
889 
890  // create a new tab frame
891 #if wxCHECK_VERSION(3, 0, 0)
892  m_curPage = -1;
893 #else
894  m_curpage = -1;
895 #endif
896 
897  // Get list of tab id's and move them to pane
898  wxString tab_list = tab_part.AfterFirst(wxT('='));
899  while (1)
900  {
901  wxString tab = tab_list.BeforeFirst(wxT(','));
902  wxString name = tab.AfterFirst(wxT(';'));
903  tab = tab.BeforeFirst(wxT(';'));
904 
905  if (tab.empty())
906  break;
907  tab_list = tab_list.AfterFirst(wxT(','));
908 
909  // Check if this page has an 'active' marker
910  const wxChar c = tab[0];
911  if (c == wxT('+') || c == wxT('*'))
912  tab = tab.Mid(1);
913 
914  // Move tab to pane
915  const int index_in_m_tabs = GetTabIndexFromTooltip(name);
916  if (index_in_m_tabs < 0)
917  continue;
918 
919  wxAuiNotebookPage& page = m_tabs.GetPage(index_in_m_tabs);
920 
921  // save the actual active tab, because it will be set to 0 after a Split()
922  active_tab = dest_tabs->GetActivePage();
923  const size_t newpage_idx = dest_tabs->GetPageCount();
924  dest_tabs->InsertPage(page.window, page, newpage_idx);
925 
926  if (c == wxT('+'))
927  dest_tabs->SetActivePage(newpage_idx);
928  else if (c == wxT('*'))
929  sel_page = index_in_m_tabs;
930 
931  // If we should be the first tab in a tab-ctrl we switch to the next existing tab,
932  // or create a new one by calling Split() and update the dest_tabs accordingly.
933  if (firstTabInCtrl)
934  {
935  int nextIndex = m_TabCtrls.Index(dest_tabs) + 1;
936  if (nextIndex == 0 || nextIndex >= static_cast<int>(m_TabCtrls.GetCount()))
937  {
938  Split(index_in_m_tabs, wxRIGHT);
939  // reset the active tab, because a Split() set it to zero
940  dest_tabs->SetActivePage(active_tab);
941  dest_tabs = GetActiveTabCtrl();
942  }
943  else
944  dest_tabs = m_TabCtrls.Item(nextIndex);
945  }
946  // Change the pane name to the one we have stored in the layout-string.
947  wxAuiPaneInfo& pane = m_mgr.GetPane( GetTabFrameFromTabCtrl(dest_tabs) );
948  if (pane.name != pane_name)
949  {
950  tab.Replace(pane_name, pane.name);
951  frames.Replace(pane_name, pane.name);
952  pane_name = pane.name;
953  }
954  firstTabInCtrl = false;
955  found = true;
956  }
957  // We come here after at least one tabctrl is filled, so the next tab should go in a new one
958  firstTabInCtrl = true;
959 
960  tabs = tabs.AfterFirst(wxT('|'));
961  }
962 
963  // Check for windows not readded to the notebook and add the at the end.
964  for (size_t i = 0; i < tab_count; ++i)
965  {
966  wxAuiNotebookPage& page = m_tabs.GetPage(i);
967 
968  // find out which onscreen tab ctrl owns this tab
969  // if none then add it to the last used tabctrl
970  wxAuiTabCtrl* ctrl;
971  int ctrl_idx;
972  if ( !FindTab(page.window, &ctrl, &ctrl_idx) )
973  {
974  const size_t newpage_idx = dest_tabs->GetPageCount();
975  dest_tabs->InsertPage (page.window, page, newpage_idx);
976  }
977 
978  }
979 
980  if (mergeLayouts)
981  {
982  wxRegEx reDockSize(_T("(dock_size[()0-9,]+=)[0-9]+"));
983  const wxString replacement(wxT("\\1-1"));
984  // Make a centered frame left docked
985  frames.Replace(wxString::Format(wxT("dock_size(%d"), wxAUI_DOCK_CENTER), wxString::Format(wxT("dock_size(%d"), wxAUI_DOCK_LEFT));
986  frames.Replace(wxString::Format(wxT("dir=%d"), wxAUI_DOCK_CENTER), wxString::Format(wxT("dir=%d"), wxAUI_DOCK_LEFT));
987  if (reDockSize.Matches(frames))
988  reDockSize.ReplaceAll(&frames,replacement);
989  if (reDockSize.Matches(currentLayout))
990  reDockSize.ReplaceAll(&currentLayout,replacement);
991  while (!currentLayout.empty())
992  {
993  wxString pane_part = currentLayout.BeforeFirst('|');
994  pane_part.Trim();
995  pane_part.Trim(true);
996  if (!pane_part.empty())
997  frames += pane_part + wxT("|");
998  currentLayout = currentLayout.AfterFirst('|');
999  currentLayout.Trim();
1000  currentLayout.Trim(true);
1001  }
1002  }
1003 
1004  if (found)
1005  m_mgr.LoadPerspective(frames);
1006  RemoveEmptyTabFrames();
1007 
1008  // Force refresh of selection
1009 #if wxCHECK_VERSION(3, 0, 0)
1010  m_curPage = -1;
1011 #else
1012  m_curpage = -1;
1013 #endif
1014  SetSelection(sel_page);
1015 
1016  UpdateTabControlsArray();
1017  return true;
1018 }
