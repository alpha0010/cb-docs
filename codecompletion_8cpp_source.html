225  inline bool EditorHasNameUnderCursor(wxString& NameUnderCursor, bool& IsInclude)
226  {
227  bool ReturnValue = false;
228  if (cbEditor* ed = Manager::Get()->GetEditorManager()->GetBuiltinActiveEditor())
229  {
230  cbStyledTextCtrl* control = ed->GetControl();
231  const int pos = control->GetCurrentPos();
232  const wxString line = control->GetLine(control->LineFromPosition(pos));
233  const wxRegEx reg(_T("^[ \t]*#[ \t]*include[ \t]+[\"<]([^\">]+)[\">]"));
234  wxString inc;
235  if (reg.Matches(line))
236  inc = reg.GetMatch(line, 1);
237 
238  if (!inc.IsEmpty())
239  {
240  NameUnderCursor = inc;
241  ReturnValue = true;
242  IsInclude = true;
243  }
244  else
245  {
246  const int start = control->WordStartPosition(pos, true);
247  const int end = control->WordEndPosition(pos, true);
248  const wxString word = control->GetTextRange(start, end);
249  if (!word.IsEmpty())
250  {
251  NameUnderCursor.Clear();
252  NameUnderCursor << word;
253  ReturnValue = true;
254  IsInclude = false;
255  }
256  }
257  }
258  return ReturnValue;

949  cbStyledTextCtrl* stc = ed->GetControl();
950 
951  TokenIdxSet result;
952  if ( m_NativeParser.MarkItemsByAI(result, m_NativeParser.GetParser().Options().useSmartSense, true, caseSens, caretPos)
953  || m_NativeParser.LastAISearchWasGlobal() ) // enter even if no match (code-complete C++ keywords)
954  {
955  if (s_DebugSmartSense)
956  CCLogger::Get()->DebugLog(F(wxT("%lu results"), static_cast<unsigned long>(result.size())));
957  TRACE(F(wxT("%lu results"), static_cast<unsigned long>(result.size())));
958 
959  if (result.size() <= m_CCMaxMatches)
960  {
961  if (s_DebugSmartSense)
962  CCLogger::Get()->DebugLog(wxT("Generating tokens list..."));
963 
964  wxImageList* ilist = m_NativeParser.GetImageList();
965  stc->ClearRegisteredImages();

967  tokens.reserve(result.size());
968  std::set<int> alreadyRegistered;
969  StringSet uniqueStrings; // ignore keywords with same name as parsed tokens
970

972 
973  CC_LOCKER_TRACK_TT_MTX_LOCK(s_TokenTreeMutex)
974 
975  for (TokenIdxSet::const_iterator it = result.begin(); it != result.end(); ++it)
976  {
977  const Token* token = tree->at(*it);
978  if (!token || token->m_Name.IsEmpty())
979  continue;
980 
981  if (preprocessorOnly && token->m_TokenKind != tkMacroDef)
982  continue;
983 
984  int iidx = m_NativeParser.GetTokenKindImage(token);
985  if (alreadyRegistered.find(iidx) == alreadyRegistered.end())
986  {
987  stc->RegisterImage(iidx, ilist->GetBitmap(iidx));
988  alreadyRegistered.insert(iidx);
989  }
990 
991  wxString dispStr;
992  if (token->m_TokenKind & tkAnyFunction)
993  {
994  if (m_DocHelper.IsEnabled())
995  dispStr = wxT("(): ") + token->m_FullType;
996  else
997  dispStr = token->GetFormattedArgs() << _T(": ") << token->m_FullType;
998  }
999  else if (token->m_TokenKind == tkVariable)
1000  dispStr = wxT(": ") + token->m_FullType;
1001  tokens.push_back(CCToken(token->m_Index, token->m_Name + dispStr, token->m_Name, token->m_IsTemp ? 0 : 5, iidx));
1002  uniqueStrings.insert(token->m_Name);
1003 
1004  if (token->m_TokenKind == tkNamespace && token->m_Aliases.size())
1005  {
1006  for (size_t i = 0; i < token->m_Aliases.size(); ++i)
1007  {
1008  // dispStr will currently be empty, but contain something in the future...
1009  tokens.push_back(CCToken(token->m_Index, token->m_Aliases[i] + dispStr, token->m_Aliases[i], 5, iidx));
1010  uniqueStrings.insert(token->m_Aliases[i]);
1011  }
1012  }
1013  }
1014 
1015  CC_LOCKER_TRACK_TT_MTX_UNLOCK(s_TokenTreeMutex)
1016 
1017  if (m_NativeParser.LastAISearchWasGlobal() && !preprocessorOnly)
1018  {
1019  // empty or partial search phrase: add theme keywords in search list
1020  if (s_DebugSmartSense)
1021  CCLogger::Get()->DebugLog(_T("Last AI search was global: adding theme keywords in list"));
1022 
1023  EditorColourSet* colour_set = ed->GetColourSet();
1024  if (colour_set)
1025  {
1026  wxString lastSearch = m_NativeParser.LastAIGlobalSearch().Lower();
1027  int iidx = ilist->GetImageCount();
1028  FileType fTp = FileTypeOf(ed->GetShortName());
1029  bool isC = (fTp == ftHeader || fTp == ftSource|| fTp == ftTemplateSource);
1030  // theme keywords
1031  HighlightLanguage lang = ed->GetLanguage();
1032  if (lang == HL_NONE)
1033  lang = colour_set->GetLanguageForFilename(ed->GetFilename());
1034  wxString strLang = colour_set->GetLanguageName(lang);
1035  // if its sourcecode/header file and a known fileformat, show the corresponding icon
1036  if (isC && strLang == wxT("C/C++"))
1037  stc->RegisterImage(iidx, wxBitmap(cpp_keyword_xpm));
1038  else if (isC && strLang == wxT("D"))
1039  stc->RegisterImage(iidx, wxBitmap(d_keyword_xpm));
1040  else
1041  stc->RegisterImage(iidx, wxBitmap(unknown_keyword_xpm));
1042  // the first two keyword sets are the primary and secondary keywords (for most lexers at least)
1043  // but this is now configurable in global settings
1044  for (int i = 0; i <= wxSCI_KEYWORDSET_MAX; ++i)
1045  {
1046  if (!m_LexerKeywordsToInclude[i])
1047  continue;
1048 
1049  wxString keywords = colour_set->GetKeywords(lang, i);

1052  {
1053  wxString kw = tkz.GetNextToken();
1054  if ( kw.Lower().StartsWith(lastSearch)
1055  && uniqueStrings.find(kw) == uniqueStrings.end() )
1056  {
1057  tokens.push_back(CCToken(wxNOT_FOUND, kw, iidx));
1058  }
1059  }
1060  }
1061  }
1062  }
1063  }
1064  else if (!stc->CallTipActive())
1065  {
1066  wxString msg = _("Too many results.\n"
1067  "Please edit results' limit in code-completion options,\n"
1068  "or type at least one more character to narrow the scope down.");
1069  stc->CallTipShow(stc->GetCurrentPos(), msg);
1070  }
1071  }
1072  else if (!stc->CallTipActive())
1073  {
1074  if (s_DebugSmartSense)
1075  CCLogger::Get()->DebugLog(wxT("0 results"));

1078  {

1081  msg += m_NativeParser.GetParser().NotDoneReason();
1082  CCLogger::Get()->DebugLog(msg);

1363  addComment = false;
1364  break;
1365  }

1373  {

1376  if (ppIf.Matches(stc->GetLine(ppLine))) // ignore else's, elif's, ...
1377  ++depth;
1378  else if (ppEnd.Matches(stc->GetLine(ppLine)))

1384  pp.Matches(stc->GetLine(ppLine));

1386  itemText.Append(wxT(" // ") + pp.GetMatch(stc->GetLine(ppLine), 2));
1387  break;
1388  }
1389  }
1390  }

1392

1395  while (ch != _T('<') && ch != _T('"') && ch != _T('#') && (pos>0))
1396  ch = stc->GetCharAt(--pos);

1402  else if (ch == _T('<'))
1403  itemText << _T('>');

1408  const wxString& alreadyText = stc->GetTextRange(curPos, endPos);

1414  bool insideParentheses = false;

1434 
1435  CC_LOCKER_TRACK_TT_MTX_UNLOCK(s_TokenTreeMutex)
1436 
1437  if (addParentheses)
1438  {
1439  bool insideFunction = true;

1441  {
1442  ccSearchData searchData = { stc, ed->GetFilename() };
1443  int funcToken;
1444  if (m_NativeParser.FindCurrentFunctionStart(&searchData, 0, 0, &funcToken) == -1)

1447  itemText += tokenArgs;

1449  }
1450  else // Found something, but result may be false positive.

1452  CC_LOCKER_TRACK_TT_MTX_LOCK(s_TokenTreeMutex)
1453

1455  // Make sure that parent is not container (class, etc)
1456  if (parent && (parent->m_TokenKind & tkAnyFunction) == 0)

1461  }
1462

1479  }
1480  else

1483  }

1488  stc->SetTargetEnd(curPos);
1489

1493  stc->GotoPos(startPos + itemText.Length() + positionModificator);
