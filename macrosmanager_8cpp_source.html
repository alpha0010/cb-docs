435 void MacrosManager::ReplaceMacros(wxString& buffer, ProjectBuildTarget* target, bool subrequest)
436 {
437  if (buffer.IsEmpty())
438  return;
439 
440  static const wxString delim(_T("$%["));
441  if ( buffer.find_first_of(delim) == wxString::npos )
442  return;
443 
444  cbProject* project = target
445  ? target->GetParentProject()
446  : Manager::Get()->GetProjectManager()->GetActiveProject();
447  EditorBase* editor = Manager::Get()->GetEditorManager()->GetActiveEditor();
448 
449  if (!target)
450  {
451  if (project)
452  {
453  // use the currently compiling target
454  target = project->GetCurrentlyCompilingTarget();
455  // if none,
456  if (!target)
457  // use the last known active target
458  target = project->GetBuildTarget(project->GetActiveBuildTarget());
459  }
460  }
461  if (project != m_LastProject || target != m_LastTarget || (editor && (editor->GetFilename() != m_ActiveEditorFilename))
462  || (project && (UnixFilename(project->GetBasePath()) != m_ProjectDir))
463  || (UnixFilename(m_ProjectWxFileName.GetFullName()) != m_ProjectFilename)
464  || (target && (target->GetTitle() != m_TargetName)) )
465  RecalcVars(project, editor, target);
466 
467  wxString search;
468  wxString replace;
469 
470  if (buffer.find(_T("$if")) != wxString::npos)
471  {
472  while (m_RE_If.Matches(buffer))
473  {
474  search = m_RE_If.GetMatch(buffer, 0);
475  replace = EvalCondition(m_RE_If.GetMatch(buffer, 1), m_RE_If.GetMatch(buffer, 3), m_RE_If.GetMatch(buffer, 5), target);
476  buffer.Replace(search, replace, false);
477  }
478  }
479 
480  while (m_RE_Script.Matches(buffer))
481  {
482  search = m_RE_Script.GetMatch(buffer, 1);
483  replace = Manager::Get()->GetScriptingManager()->LoadBufferRedirectOutput(m_RE_Script.GetMatch(buffer, 2));
484  buffer.Replace(search, replace, false);
485  }
486 
487  while (m_RE_ToAbsolutePath.Matches(buffer))
488  {
489  search = m_RE_ToAbsolutePath.GetMatch(buffer, 0);
490  const wxString relativePath = m_RE_ToAbsolutePath.GetMatch(buffer, 1);
491  wxFileName fn(relativePath);
492  fn.MakeAbsolute();
493  replace = fn.GetFullPath();
494  buffer.Replace(search, replace, false);
495  }
496 
497  while (m_RE_To83Path.Matches(buffer))
498  {
499  search = m_RE_To83Path.GetMatch(buffer, 0);
500  const wxString path = m_RE_To83Path.GetMatch(buffer, 1);
501  wxFileName fn(path);
502  fn.MakeAbsolute(); // make absolute before translating to 8.3 notation
503  replace = fn.GetShortPath();
504  buffer.Replace(search, replace, false);
505  }
506 
507  int index = wxNOT_FOUND;
508  while ((index = buffer.Index(toNativePath)) != wxNOT_FOUND)
509  {
510  int end = MatchBrace(buffer, index + toNativePath.Length() - 1);
511  wxString content = buffer.Mid(index + toNativePath.Length(), end - index - toNativePath.Length());
512  ReplaceMacros(content, target, true);
513  buffer.Replace(buffer.Mid(index, end - index + 1), UnixFilename(content), false);
514  }
515 
516  while ((index = buffer.Index(toUnixPath)) != wxNOT_FOUND)
517  {
518  int end = MatchBrace(buffer, index + toUnixPath.Length() - 1);
519  wxString content = buffer.Mid(index + toUnixPath.Length(), end - index - toUnixPath.Length());
520  ReplaceMacros(content, target, true);
521  buffer.Replace(buffer.Mid(index, end - index + 1), UnixFilename(content, wxPATH_UNIX), false);
522  }
523 
524  while ((index = buffer.Index(toWindowsPath)) != wxNOT_FOUND)
525  {
526  int end = MatchBrace(buffer, index + toWindowsPath.Length() - 1);

616 wxString MacrosManager::EvalCondition(const wxString& in_cond, const wxString& true_clause, const wxString& false_clause, ProjectBuildTarget* target)
617 {
618  enum condition_codes {EQ = 1, LT = 2, GT = 4, NE = 8};
619 
620  wxString cond(in_cond);
621  wxString result;
622 
623  ReplaceMacros(cond, target, true);
624 
625  if (!m_RE_IfSp.Matches(in_cond))
626  return false_clause;
627 
628 
629  wxString cmpToken(m_RE_IfSp.GetMatch(in_cond, 3).Strip(wxString::both));
630  wxString left(m_RE_IfSp.GetMatch(in_cond, 2).Strip(wxString::both));
631  wxString right(m_RE_IfSp.GetMatch(in_cond, 4).Strip(wxString::both));
632 
633 
634  int compare = left.Cmp(right);
635  if (compare == 0)
636  compare = EQ;
637  else if (compare < 0)
638  compare = LT | NE;
639  else if (compare > 0)
640  compare = GT | NE;
641 
642 
643  if (cmpToken.IsEmpty())
644  {
645  if (cond.IsEmpty() || cond.IsSameAs(_T("0")) || cond.IsSameAs(_T("false")))
646  return false_clause;
647  return true_clause;
648  }
649 
650  int condCode = 0;
651 
652  if (cmpToken.IsSameAs(_T("==")) || cmpToken.IsSameAs(_T("=")))
653  condCode = EQ;
654  if (cmpToken.IsSameAs(_T("<")))
655  condCode = LT;
656  if (cmpToken.IsSameAs(_T(">")))
657  condCode = GT;
658  if (cmpToken.IsSameAs(_T("<=")))
659  condCode = EQ | LT;
660  if (cmpToken.IsSameAs(_T(">=")))
661  condCode = EQ | GT;
662  if (cmpToken.IsSameAs(_T("!=")))
663  condCode = NE;
664 
665  return condCode & compare ? true_clause : false_clause;
666 }
