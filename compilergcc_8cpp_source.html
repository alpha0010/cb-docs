688 void CompilerGCC::SetupEnvironment()
689 {
690  // Special case so "No Compiler" is valid, but I'm not sure there is
691  // any valid reason to continue with this function.
692  // If we do continue there are wx3 asserts, because of empty paths.
693  if (m_CompilerId == wxT("null"))
694  return;
695 
696  Compiler* compiler = CompilerFactory::GetCompiler(m_CompilerId);
697  if (!compiler)
698  return;
699 
700  wxString currentPath;
701  if ( !wxGetEnv(_T("PATH"), &currentPath) )
702  {
703  InfoWindow::Display(_("Environment error"),
704  _("Could not read the PATH environment variable!\n"
705  "This can't be good. There may be problems running\n"
706  "system commands and the application might not behave\n"
707  "the way it was designed to..."),
708  15000, 3000);
709  return;
710  }
711 
712 // Manager::Get()->GetLogManager()->DebugLogError(_T("PATH environment:"));
713 // Manager::Get()->GetLogManager()->DebugLogError(currentPath);
714 
715  const wxString pathApp = platform::windows ? _T(";") : _T(":");
716  const wxString pathSep = wxFileName::GetPathSeparator(); // "\" or "/"
717  const bool caseSens = !(platform::windows);
718 
719  wxString cApp = compiler->GetPrograms().C;
720  Manager::Get()->GetMacrosManager()->ReplaceMacros(cApp);
721  wxArrayString extraPaths = compiler->GetExtraPaths();
722  wxString extraPathsBinPath(wxEmptyString);
723 
724  // Get configured masterpath, expand macros and remove trailing separators
725  wxString masterPath = compiler->GetMasterPath();
726 
727  Manager::Get()->GetMacrosManager()->ReplaceMacros(masterPath);
728  while ( !masterPath.IsEmpty()
729  && ((masterPath.Last() == '\\') || (masterPath.Last() == '/')) )
730  masterPath.RemoveLast();
731 
732  // Compile new PATH list...
733  wxPathList pathList;
734  // [1] Pre-pend "master path" and "master path\bin"...
735  if ( !masterPath.Trim().IsEmpty() ) // Would be very bad, if it *is* empty
736  {
737  pathList.Add(masterPath + pathSep + _T("bin"));
738  pathList.Add(masterPath); // in case there is no "bin" sub-folder
739  }
740  // [2] Get configured extrapath(s), expand macros and remove trailing separators
741  for (size_t i=0; i<extraPaths.GetCount(); ++i)
742  {
743  wxString extraPath = extraPaths[i];
744  Manager::Get()->GetMacrosManager()->ReplaceMacros(extraPath);
745  while (extraPath.Last() == '\\' || extraPath.Last() == '/')
746  extraPath.RemoveLast();
747  if (!extraPath.Trim().IsEmpty())
748  {
749  // Remember, if we found the C application in the extra path's:
750  if ( extraPathsBinPath.IsEmpty()
751  && wxFileExists(extraPath + pathSep + cApp ) )
752  extraPathsBinPath = extraPath;
753  pathList.Add(extraPath);
754  }
755  }
756  // [3] Append what has already been in the PATH envvar...
757  // If we do it this way, paths are automatically normalized and doubles are removed
758  wxPathList pathArray;
759  pathArray.AddEnvList(_T("PATH"));
760  pathList.Add(pathArray);
761 
762  // Try to locate the path to the C compiler:
763  wxString binPath = pathList.FindAbsoluteValidPath(cApp);
764 
765  // It seems, under Win32, the above command doesn't search in paths with spaces...
766  // Look directly for the file in question in masterPath if it is not already found.
767  if ( binPath.IsEmpty()
768  || (pathList.Index(wxPathOnly(binPath), caseSens)==wxNOT_FOUND) )
769  {
770  if (wxFileExists(masterPath + pathSep + _T("bin") + pathSep + cApp))
771  binPath = masterPath + pathSep + _T("bin");
772  else if (wxFileExists(masterPath + pathSep + cApp))
773  binPath = masterPath;
774  else if (!extraPathsBinPath.IsEmpty())
775  binPath = extraPathsBinPath;
776  }
777  else
778  binPath = wxPathOnly(binPath);
779 
780  /* TODO (jens#1#): Is the above correct ?
781  Or should we search in the whole systempath (pathList in this case) for the executable? */
782  // Try again...
783  if ((binPath.IsEmpty() || (pathList.Index(binPath, caseSens)==wxNOT_FOUND)))
784  {
785  InfoWindow::Display(_("Environment error"),
786  _("Can't find compiler executable in your configured search path's for ") + compiler->GetName() + _T('\n'));
787  Manager::Get()->GetLogManager()->DebugLogError(F(_T("Can't find compiler executable in your configured search path's (for %s)..."), compiler->GetName().wx_str()));
788 
789  return; // Failed to locate compiler executable in path's as provided!
790  }
791 
792  // Convert the pathList into a string to apply.
793  wxString envPath(binPath); // make sure the bin-path we found is in front
794  // and remove it from pathList
795  pathList.Remove(binPath);
796  for (size_t i=0; i<pathList.GetCount(); ++i)
797  envPath += ( pathApp + pathList[i] );
798 
799 // Manager::Get()->GetLogManager()->DebugLogError(_T("Updating compiler PATH environment:"));
800 // Manager::Get()->GetLogManager()->DebugLogError(envPath);
801 
802  if ( !wxSetEnv(_T("PATH"), envPath) )
803  {
804  InfoWindow::Display(_("Environment error"),
805  _("Can't set PATH environment variable! That's bad and the compiler might not work."));
806  Manager::Get()->GetLogManager()->DebugLog(_T("Can't set PATH environment variable! That's bad and the compiler might not work.\n"));
807  }

1221  return -3;

1223

1225  if (!cmd)
1226  {
1227  if (IsProcessRunning())
1228  return 0;
1229 
1230  while (1)
1231  {
1232  // keep switching build states until we have commands to run or reach end of states
1233  BuildStateManagement();
1234  cmd = m_CommandQueue.Next();
1235  if (!cmd && m_BuildState == bsNone && m_NextBuildState == bsNone)
1236  {
1237  NotifyJobDone(true);
