183 wxString cbDebuggerPlugin::GetEditorWordAtCaret(const wxPoint* mousePosition)
184 {
185  cbEditor* ed = Manager::Get()->GetEditorManager()->GetBuiltinActiveEditor();
186  if (!ed)
187  return wxEmptyString;
188  cbStyledTextCtrl* stc = ed->GetControl();
189  if (!stc)
190  return wxEmptyString;
191 
192  wxString selected_text = stc->GetSelectedText();
193  if (selected_text != wxEmptyString)
194  {
195  selected_text.Trim(true);
196  selected_text.Trim(false);
197 
198  wxString::size_type pos = selected_text.find(wxT('\n'));
199  if (pos != wxString::npos)
200  {
201  selected_text.Remove(pos, selected_text.length() - pos);
202  selected_text.Trim(true);
203  selected_text.Trim(false);
204  }
205  // check if the mouse is over the selected text
206  if (mousePosition)
207  {
208  int startPos = stc->GetSelectionStart();
209  int endPos = stc->GetSelectionEnd();
210  int mousePos = stc->PositionFromPointClose(mousePosition->x, mousePosition->y);
211  if (mousePos == wxSCI_INVALID_POSITION)
212  return wxEmptyString;
213  else if (startPos <= mousePos && mousePos <= endPos)
214  return selected_text;
215  else
216  return wxEmptyString;
217  }
218  else
219  return selected_text;
220  }
221 
222  if (mousePosition)
223  {
224  int pos = stc->PositionFromPoint(*mousePosition);
225  int start = stc->WordStartPosition(pos, true);
226  int end = stc->WordEndPosition(pos, true);
227  selected_text = stc->GetTextRange(start, end);
228  }
229  else
230  {
231  int start = stc->WordStartPosition(stc->GetCurrentPos(), true);
232  int end = stc->WordEndPosition(stc->GetCurrentPos(), true);
233  selected_text = stc->GetTextRange(start, end);
234  }
235  return selected_text;

379 void cbDebuggerPlugin::EditorLinesAddedOrRemoved(cbEditor* editor, int startline, int lines)
380 {
381  // here we keep the breakpoints in sync with the editors
382  // (whenever lines are added or removed)
383  if (!editor || lines == 0)
384  return;
385 
386  const wxString& filename = editor->GetFilename();
387 
388  std::vector<int> breakpoints_for_file;
389  int count = GetBreakpointsCount();
390  for (int ii = 0; ii < count; ++ii)
391  {
392  const cb::shared_ptr<cbBreakpoint> &b = GetBreakpoint(ii);
393 
394  if (b->GetLocation() == filename)
395  {
396  breakpoints_for_file.push_back(ii);
397  }
398  }
399 
400  if (lines < 0)
401  {
402  // removed lines
403  // make "lines" positive, for easier reading below
404  lines = -lines;
405  int endline = startline + lines - 1;
406 
407  std::vector<cb::shared_ptr<cbBreakpoint> > to_remove;
408 
409  for (std::vector<int>::iterator it = breakpoints_for_file.begin(); it != breakpoints_for_file.end(); ++it)
410  {
411  const cb::shared_ptr<cbBreakpoint> &b = GetBreakpoint(*it);
412  if (b->GetLine() > endline)
413  ShiftBreakpoint(*it, -lines);
414  else if (b->GetLine() >= startline && b->GetLine() <= endline)
415  to_remove.push_back(b);
416  }
417 
418  for (std::vector<cb::shared_ptr<cbBreakpoint> >::iterator it = to_remove.begin(); it != to_remove.end(); ++it)
419  DeleteBreakpoint(*it);
420  }
421  else
422  {
423  for (std::vector<int>::iterator it = breakpoints_for_file.begin(); it != breakpoints_for_file.end(); ++it)
424  {
425  const cb::shared_ptr<cbBreakpoint> &b = GetBreakpoint(*it);
426  if (b->GetLine() > startline)
427  ShiftBreakpoint(*it, lines);
428  }
429  }
430 }

633 
634  wxString out;
635  switch (target->GetTargetType())
636  {
637  case ttExecutable:
638  case ttConsoleOnly:
639  case ttNative:
640  {
641  out = UnixFilename(target->GetOutputFilename());
642  Manager::Get()->GetMacrosManager()->ReplaceEnvVars(out); // apply env vars
643  wxFileName f(out);
644  f.MakeAbsolute(target->GetParentProject()->GetBasePath());
645  out = f.GetFullPath();
646  Log(_("Adding file: ") + out);
647  ConvertDirectory(out);
648  }
649  break;
650 
651  case ttStaticLib:
652  case ttDynamicLib:
653  // check for hostapp
654  if (target->GetHostApplication().IsEmpty())
655  {
656  cbMessageBox(_("You must select a host application to \"run\" a library..."));
657  return false;
658  }
659  out = UnixFilename(target->GetHostApplication());
660  Manager::Get()->GetMacrosManager()->ReplaceEnvVars(out); // apply env vars
661  Log(_("Adding file: ") + out);

663  break;
664 
665  case ttCommandsOnly: // fall through:
666  default:
667  Log(_("Unsupported target type (Project -> Properties -> Build Targets -> Type)"), Logger::error);
668  return false;
669  }
670  if (out.empty())
671  {
672  Log(_("Couldn't find the path to the debuggee!"), Logger::error);
673  return false;
674  }
675 
676  workingDirectory = target->GetWorkingDir();
677  Manager::Get()->GetMacrosManager()->ReplaceEnvVars(workingDirectory);
678  wxFileName cd(workingDirectory);
679  if (cd.IsRelative())
680  cd.MakeAbsolute(target->GetParentProject()->GetBasePath());
681  workingDirectory = cd.GetFullPath();
682 
683  pathToDebuggee = out;

687 bool cbDebuggerPlugin::EnsureBuildUpToDate(StartType startType)
688 {
689  m_StartType = startType;
690  m_WaitingCompilerToFinish = false;
691 
692  // compile project/target (if not attaching to a PID)
693  if (!IsAttachedToProcess())
694  {
695  // should we build to make sure project is up-to-date?
696  if (!cbDebuggerCommonConfig::GetFlag(cbDebuggerCommonConfig::AutoBuild))
697  {
698  m_WaitingCompilerToFinish = false;
699  m_pCompiler = nullptr;
700  return true;
701  }
702

704  PluginsArray plugins = Manager::Get()->GetPluginManager()->GetCompilerOffers();
705  if (plugins.GetCount())
706  m_pCompiler = (cbCompilerPlugin*)plugins[0];
707  else
708  m_pCompiler = nullptr;
709  if (m_pCompiler)
710  {
711  // is the compiler already running?
712  if (m_pCompiler->IsRunning())
713  {
714  Log(_("Compiler in use..."));
715  Log(_("Aborting debugging session"));
716  cbMessageBox(_("The compiler is currently in use. Aborting debugging session..."),
717  _("Compiler running"), wxICON_WARNING);
718  return false;
719  }
720 
721  Log(_("Building to ensure sources are up-to-date"));
722  m_WaitingCompilerToFinish = true;
723  m_pCompiler->Build();
724  // now, when the build is finished, DoDebug will be launched in OnCompilerFinished()
725  }
726  }

853  // Fetch the terminal's tty, so we can tell the debugger what TTY to use,
854  // thus redirecting program's stdin/stdout/stderr to the terminal.
855 
856  wxString cmd;
857  int consolePid = 0;
858  // Use the terminal specified by the user in the Settings -> Environment.
859  wxString term = Manager::Get()->GetConfigManager(_T("app"))->Read(_T("/console_terminal"), DEFAULT_CONSOLE_TERM);
860 
861  term.Replace(_T("$TITLE"), wxString(wxT("'"))+_("Program Console")+wxT("'"));
862  cmd << term << _T(" ");
863 
864  const wxString &sleepCommand = MakeSleepCommand();
865  cmd << sleepCommand;
866 
867  Manager::Get()->GetMacrosManager()->ReplaceEnvVars(cmd);
868 
869  // The lifetime of wxProcess objects is very uncertain, so we are using a shared pointer to
870  // prevent us accessing deleted objects.
871  cb::shared_ptr<ConsoleProcessTerminationInfo> processInfo(new ConsoleProcessTerminationInfo);
872  ConsoleProcess *process = new ConsoleProcess(processInfo);
873  consolePid = wxExecute(cmd, wxEXEC_ASYNC, process);
874  if (consolePid <= 0)
875  return -1;
876 
877  // Try to find the TTY. We're using a loop, because some slow machines might make the check fail due
878  // to a slow starting terminal.
879  for (int ii = 0; ii < 100; ++ii)
880  {
881  // First, wait for the terminal to settle down, else PS won't see the sleep task
882  Manager::Yield();
883  ::wxMilliSleep(200);
884 
885  // Try to detect if the terminal command is present or its parameters are valid.
886  if (processInfo->FailedToStart() /*&& ii > 0*/)
887  {
888  Log(F(wxT("Failed to execute terminal command: '%s' (exit code: %d)"),
889  cmd.wx_str(), processInfo->status), Logger::error);
890  break;
891  }
892 
893  // Try to find tty path and pid for the sleep command we've just executed.
894  const ConsoleInfo &info = GetConsoleTty(consolePid);
895

897  if (!info.IsValid())
898  continue;

901  // try to be easier on resources and use a shared server process. For these terminals the
902  // spawned terminal process exits immediately, but the sleep command is still executed.
903  // If we detect such case we will return the PID for the sleep command instead of the PID
904  // for the terminal.
905  if (kill(consolePid, 0) == -1 && errno == ESRCH) {
906  DebugLog(F(wxT("Using sleep command's PID as console PID %d, TTY %s"),

909  return info.sleepPID;

915  return consolePid;
916  }
917  }
