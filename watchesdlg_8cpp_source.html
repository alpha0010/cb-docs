286  static void WatchToString(wxString &result, const cbWatch &watch, const wxString &indent = wxEmptyString)
287  {
288  wxString sym, value;
289  watch.GetSymbol(sym);
290  watch.GetValue(value);
291 
292  result += indent + wxT("[symbol = ") + sym + wxT("]\n");
293  result += indent + wxT("[value = ") + value + wxT("]\n");
294  result += indent + wxString::Format(wxT("[children = %d]\n"), watch.GetChildCount());
295 
296  for(int child_index = 0; child_index < watch.GetChildCount(); ++child_index)
297  {
298  cb::shared_ptr<const cbWatch> child = watch.GetChild(child_index);
299 
300  result += indent + wxString::Format(wxT("[child %d]\n"), child_index);
301  WatchToString(result, *child, indent + wxT(" "));
302  }
303  }
304  private:
305  DECLARE_EVENT_TABLE()
306  private:
307  typedef std::map<cbWatch const*, WatchRawDialog*> Map;
308 
309  static Map s_dialogs;
310 
311  wxTextCtrl *m_text;
312  Type m_type;

411 inline void AppendChildren(wxPropertyGrid &grid, wxPGProperty &property, cbWatch &watch,
412  bool readonly, const wxColour &changedColour)
413 {
414  for(int ii = 0; ii < watch.GetChildCount(); ++ii)
415  {
416  cb::shared_ptr<cbWatch> child = watch.GetChild(ii);
417 
418  wxString symbol, value, type;
419  child->GetSymbol(symbol);
420  child->GetValue(value);
421  child->GetType(type);
422 
423  wxPGProperty *prop = new WatchesProperty(symbol, value, child, readonly);
424  prop->SetExpanded(child->IsExpanded());
425  wxPGProperty *new_prop = grid.AppendIn(&property, prop);
426  grid.SetPropertyAttribute(new_prop, wxT("Units"), type);
427  if (value.empty())
428  grid.SetPropertyHelpString(new_prop, wxEmptyString);
429  else
430  grid.SetPropertyHelpString(new_prop, symbol + wxT("=") + value);
431  grid.EnableProperty(new_prop, grid.IsPropertyEnabled(&property));
432 
433  if (child->IsChanged())
434  {
435  grid.SetPropertyTextColour(prop, changedColour);
436  WatchRawDialog::UpdateValue(static_cast<const WatchesProperty*>(prop));
437  }
438  else
439  {
440 #if wxCHECK_VERSION(3, 0, 0)
441  grid.SetPropertyColoursToDefault(prop);
442 #else
443  grid.SetPropertyColourToDefault(prop);
444 #endif
445  }
446 
447  AppendChildren(grid, *prop, *child.get(), readonly, changedColour);
448  }
449 }

451 inline void UpdateWatch(wxPropertyGrid *grid, wxPGProperty *property, cb::shared_ptr<cbWatch> watch, bool readonly)
452 {
453  if (!property)
454  return;
455  const wxColour &changedColour = Manager::Get()->GetColourManager()->GetColour(wxT("dbg_watches_changed"));
456 
457  wxString value, symbol, type;
458  watch->GetSymbol(symbol);
459  watch->GetValue(value);
460  property->SetLabel(symbol);
461  property->SetValue(value);
462  property->SetExpanded(watch->IsExpanded());
463  watch->GetType(type);
464  if (watch->IsChanged())
465  grid->SetPropertyTextColour(property, changedColour);
466  else
467  {
468 #if wxCHECK_VERSION(3, 0, 0)
469  grid->SetPropertyColoursToDefault(property);
470 #else
471  grid->SetPropertyColourToDefault(property);
472 #endif
473  }
474  grid->SetPropertyAttribute(property, wxT("Units"), type);
475  if (value.empty())
476  grid->SetPropertyHelpString(property, wxEmptyString);
477  else
478  grid->SetPropertyHelpString(property, symbol + wxT("=") + value);
479 
480  property->DeleteChildren();
481 
482  if (property->GetName() != symbol)
483  {
484  grid->SetPropertyName(property, symbol);
485  grid->SetPropertyLabel(property, symbol);
486  }
487 
488  AppendChildren(*grid, *property, *watch, readonly, changedColour);
489 
490  WatchRawDialog::UpdateValue(static_cast<const WatchesProperty*>(property));
491 }
