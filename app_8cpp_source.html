113 #if wxCHECK_VERSION(3, 0, 0)
114 bool DDEConnection::OnExecute(cb_unused const wxString& topic, const void *data, cb_unused size_t size, cb_unused wxIPCFormat format)
115 #else
116 bool DDEConnection::OnExecute(cb_unused const wxString& topic, wxChar *data, cb_unused int size, cb_unused wxIPCFormat format)
117 #endif
118 {
119  wxString strData((wxChar*)data);
120 
121  if (strData.StartsWith(_T("[IfExec_Open(\"")))
122  return false; // let Shell Open handle the request as we *know* that we have registered the Shell Open command, too
123 
124  if (strData.StartsWith(_T("[Open(\"")))
125  {
126  wxRegEx reCmd(_T("\"(.*)\""));
127  if (reCmd.Matches(strData))
128  {
129  const wxString file = reCmd.GetMatch(strData, 1);
130  // always put files in the delayed queue, the will either be loaded in OnDisconnect, or after creating of MainFrame
131  // if we open the files directly it can lead to an applicaton hang (at least when opening C::B's project-file on linux)
132  CodeBlocksApp* cb = (CodeBlocksApp*)wxTheApp;
133  if (cb)
134  cb->AddFileToOpenDelayed(file);
135  }
136  return true;
137  }
138  else if (strData.StartsWith(_T("[OpenLine(\"")))
139  {
140  wxRegEx reCmd(_T("\"(.*)\""));
141  if (reCmd.Matches(strData))
142  {
143  wxString file = reCmd.GetMatch(strData, 1);
144  CodeBlocksApp* cb = (CodeBlocksApp*)wxTheApp;
145  cb->SetAutoFile(file);
146  }
147  return true;
148  }
149  else if (strData.StartsWith(_T("[Raise]")))
150  {
151  if (m_Frame)
152  {
153  if (m_Frame->IsIconized())
154  m_Frame->Iconize(false);
155  m_Frame->Raise();
156  }
157  return true;
158  }
159  else if (strData.StartsWith(_T("[CmdLine({")))
160  {
161  int pos = strData.Find(_T("})]"));
162  if (pos!=wxNOT_FOUND)
163  {
164  wxString line = strData.Mid(10, pos-10);
165  line.Replace(_T("\\)"), _T(")"));
166  line.Replace(_T("\\("), _T("("));
167  CodeBlocksApp* cb = (CodeBlocksApp*)wxTheApp;
168  if (m_Frame && !line.empty())
169  {
170  // Manager::Get()->GetLogManager()->Log(wxString::Format(_T("DDEConnection::OnExecute line = ") + line));
171  cb->ParseCmdLine(m_Frame, line);
172  CodeBlocksEvent event(cbEVT_APP_CMDLINE);
173  Manager::Get()->ProcessEvent(event);
174  }
175 
176  }
177  return true;
178  }
179  wxSafeShowMessage(wxT("Warning"),wxString::Format(wxT("DDE topic %s not handled."),strData.wx_str()));
180  return false;
181 }

895 {

928  if (idx == -1)
929  return 0; // no target selected: just abort

932  }
933

935  PlaceWindow(m_pBatchBuildDialog);
936 
937  wxString title = _("Building '") + wxFileNameFromPath(wxString(argv[argc-1])) + _("' (target '") + m_BatchTarget + _T("')");

945  title);
946

975  // If it has not, it's still compiling.
976  if (m_pBatchBuildDialog)
977  {
978  // If operation is "--clean", there is no need to display the dialog

1091 int CodeBlocksApp::ParseCmdLine(MainFrame* handlerFrame, const wxString& CmdLineString)
1092 {
1093  // code shamelessely taken from the console wxWindows sample :)
1094  bool filesInCmdLine = false;
1095 
1096 #if wxUSE_CMDLINE_PARSER
1097  wxCmdLineParser& parser = *Manager::GetCmdLineParser();
1098  if ( CmdLineString.IsEmpty() )
1099  parser.SetCmdLine(argc, argv);
1100  else
1101  parser.SetCmdLine(CmdLineString);
1102  // wxApp::argc is a wxChar**
1103 
1104  // don't display errors as plugins will have the chance to parse the command-line
1105  // too, so we don't know here what exactly are the supported options
1106  int res = parser.Parse(false);
1107  if (res == -1)
1108  {
1109  return -1;
1110  }
1111  else
1112  {
1113  if (handlerFrame)
1114  {
1115  m_HasProject = false;
1116  m_HasWorkSpace = false;
1117  int count = parser.GetParamCount();
1118 
1119  parser.Found(_T("file"), &m_AutoFile);
1120 
1121  filesInCmdLine = (count != 0) || (!m_AutoFile.empty());
1122 
1123  for (int param = 0; param < count; ++param)
1124  {
1125  // is it a project/workspace?
1126  FileType ft = FileTypeOf(parser.GetParam(param));
1127  wxFileName fn(parser.GetParam(param));
1128  fn.Normalize(); // really important so that two same files with different names are not loaded twice
1129  if (ft == ftCodeBlocksProject)
1130  {
1131  m_HasProject = true;
1132  m_DelayedFilesToOpen.Add(parser.GetParam(param));
1133  }
1134  else if (ft == ftCodeBlocksWorkspace)
1135  {
1136  // only one workspace can be opened
1137  m_HasWorkSpace = true;
1138  m_DelayedFilesToOpen.Clear(); // remove all other files
1139  m_DelayedFilesToOpen.Add(fn.GetFullPath()); // and add only the workspace
1140  break; // and stop processing any more files
1141  }
1142  //else if (ft == ftSource || ft == ftHeader || ft == ftResource)
1143  else if (wxFile::Exists(fn.GetFullPath())) //also try to open non source, header and resource files
1144  {
1145  m_DelayedFilesToOpen.Add(fn.GetFullPath());
1146  }
1147  }
1148 
1149  // batch jobs
1150  m_Batch = m_HasProject || m_HasWorkSpace;
1151  m_Batch = m_Batch && (m_Build || m_ReBuild || m_Clean);
1152  }
1153  else
1154  {
1155  wxString val;
1156  parser.Found(_T("prefix"), &m_Prefix);
1157  parser.Found(_T("user-data-dir"), &m_UserDataDir);
1158 #ifdef __WXMSW__
1159  m_DDE = !parser.Found(_T("no-dde"));
1160  m_Assocs = !parser.Found(_T("no-check-associations"));
1161 #else
1162  m_DDE = !parser.Found(_T("no-ipc"));
1163 #endif
1164  m_SafeMode = parser.Found(_T("safe-mode"));
1165  m_Splash = !parser.Found(_T("no-splash-screen"));
1166  m_HasDebugLog = parser.Found(_T("debug-log"));
1167  m_CrashHandler = !parser.Found(_T("no-crash-handler"));
1168 
1169  wxLog::EnableLogging(parser.Found(_T("verbose")));
1170 
1171  if ( parser.Found(_T("personality"), &val)
1172  || parser.Found(_T("profile"), &val) )
1173  {
1174  SetupPersonality(val);
1175  }
1176 
1177  // batch jobs
1178  m_BatchNotify = parser.Found(_T("batch-build-notify"));
1179  m_BatchWindowAutoClose = !parser.Found(_T("no-batch-window-close"));
1180  m_Build = parser.Found(_T("build"));
1181  m_ReBuild = parser.Found(_T("rebuild"));
1182  m_Clean = parser.Found(_T("clean"));
1183  parser.Found(_T("target"), &m_BatchTarget);
1184  parser.Found(_T("script"), &m_Script);
1185  // initial setting for batch flag (will be reset when ParseCmdLine() is called again).
1186  m_Batch = m_Build || m_ReBuild || m_Clean;
1187 
1188 
1189  if (parser.Found(_T("no-log")) == false)
1190  Manager::Get()->GetLogManager()->SetLog(new TextCtrlLogger, LogManager::app_log);
1191  if (parser.Found(_T("log-to-file")))
1192  Manager::Get()->GetLogManager()->SetLog(new FileLogger(_T("codeblocks.log")), LogManager::app_log);
1193  if (m_HasDebugLog)
1194  Manager::Get()->GetLogManager()->SetLog(new TextCtrlLogger, LogManager::debug_log);
1195  if (parser.Found(_T("debug-log-to-file")))
1196  Manager::Get()->GetLogManager()->SetLog(new FileLogger(_T("codeblocks-debug.log")), LogManager::debug_log);
1197  }
1198 
1199  }
1200 #endif // wxUSE_CMDLINE_PARSER
1201  return filesInCmdLine ? 1 : 0;
1202 }

1222 {

1224  for (std::set<wxString>::const_iterator it = uniqueFilesToOpen.begin(); it != uniqueFilesToOpen.end(); ++it)
1225  frame->Open(*it, true);
1226  m_DelayedFilesToOpen.Clear();
1227 
1228  // --file foo.cpp[:line]
1229  if (!m_AutoFile.IsEmpty())
1230  {
1231  wxString linePart;
1232  wxString filePart;
1233  // wxString::Last is deprecated
1234  long linePos = m_AutoFile.Find(_T(':'), true);
1235  if (linePos != wxNOT_FOUND)
1236  {
1237  linePart = m_AutoFile.Mid(linePos + 1, wxString::npos);
1238  filePart = m_AutoFile;
1239  filePart.Remove(linePos);
1240  }
1241 
1242  long line = -1;
1243  if (linePos != wxNOT_FOUND)
1244  {
1245  // on windows, if ":line" is omitted:
1246  // assuming drive letter before the colon if ToLong fails
1247  // c:\foo\bar.h gives \foo\bar.h
1248  if ( !linePart.ToLong(&line) )
1249  {
1250  // on windows, if :line is omitted: c:\foo\bar.h -> \foo\bar.h is not the line number!
1251  filePart = m_AutoFile;
1252  }
1253  }
1254  wxFileName fn(filePart);
1255  fn.Normalize(); // really important so that two same files with different names are not loaded twice
1256  if (frame->Open(fn.GetFullPath(), false))
1257  {
1258  EditorBase* eb = Manager::Get()->GetEditorManager()->GetEditor(fn.GetFullPath());
1259  if (eb && (line != -1))
1260  eb->GotoLine(line - 1, true);
1261  }
