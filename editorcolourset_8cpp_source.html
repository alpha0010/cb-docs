100 void EditorColourSet::LoadAvailableSets()
101 {
102  // no need for syntax highlighting if batch building
103  if (Manager::IsBatchBuild())
104  return;
105 
106  wxDir dir;
107  wxString filename;
108  FileManager *fm = FileManager::Get();
109  std::list<LoaderBase*> loaders;
110  int count = 0;
111 
112  // user paths first
113  wxString path = ConfigManager::GetFolder(sdDataUser) + _T("/lexers/");
114  if (wxDirExists(path) && dir.Open(path))
115  {
116  Manager::Get()->GetLogManager()->Log(F(_("Scanning for lexers in %s..."), path.wx_str()));
117  bool ok = dir.GetFirst(&filename, _T("lexer_*.xml"), wxDIR_FILES);
118  while (ok)
119  {
120  loaders.push_back(fm->Load(path + filename));
121  ok = dir.GetNext(&filename);
122  ++count;
123  }
124  Manager::Get()->GetLogManager()->Log(F(_("Found %d lexers"), count));
125  count = 0;
126  }
127 
128  // global paths next
129  path = ConfigManager::GetFolder(sdDataGlobal) + _T("/lexers/");
130  if (wxDirExists(path) && dir.Open(path))
131  {
132  Manager::Get()->GetLogManager()->Log(F(_("Scanning for lexers in %s..."), path.wx_str()));
133  bool ok = dir.GetFirst(&filename, _T("lexer_*.xml"), wxDIR_FILES);
134  while (ok)
135  {
136  loaders.push_back(fm->Load(path + filename));
137  ok = dir.GetNext(&filename);
138  ++count;
139  }
140  Manager::Get()->GetLogManager()->Log(F(_("Found %d lexers"), count));
141  }
142 
143  EditorLexerLoader lex(this);
144  for (std::list<LoaderBase*>::iterator it = loaders.begin(); it != loaders.end(); ++it)
145  lex.Load(*it);
146 
147  ::Delete(loaders);
148 
149 
150  for (OptionSetsMap::iterator it = m_Sets.begin(); it != m_Sets.end(); ++it)
151  {
152  wxString lang = it->second.m_Langs;
153  if (lang.IsEmpty())
154  continue;
155 
156  // keep the original filemasks and keywords, so we know what needs saving later
157  for (int i = 0; i <= wxSCI_KEYWORDSET_MAX; ++i)
158  {
159  it->second.m_originalKeywords[i] = it->second.m_Keywords[i];
160  }
161  it->second.m_originalFileMasks = it->second.m_FileMasks;
162 
163  // remove old settings, no longer used
164  unsigned int i = 0;
165  while (i < it->second.m_Colours.GetCount())
166  {
167  OptionColour* opt = it->second.m_Colours.Item(i);
168  // valid values are:
169  if (opt->value < 0 && // styles >= 0
170  opt->value != cbSELECTION && // cbSELECTION
171  opt->value != cbHIGHLIGHT_LINE) // cbHIGHLIGHT_LINE
172  {
173  it->second.m_Colours.Remove(opt);
174  delete opt;
175  }
176  else
177  ++i;
178  }
179  }
180 }

182 HighlightLanguage EditorColourSet::AddHighlightLanguage(int lexer, const wxString& name)
183 {
184  if ( lexer <= wxSCI_LEX_NULL
185  || lexer > wxSCI_LEX_LAST // this is a C::B extension to wxscintilla.h
186  || name.IsEmpty() )
187  {
188  return HL_NONE;
189  }
190 
191  // fix name to be XML compliant
192  wxString newID;
193  size_t pos = 0;
194  while (pos < name.Length())
195  {
196  wxChar ch = name[pos];
197  if (wxIsalnum(ch) || ch == _T('_'))
198  newID.Append(ch); // valid character
199  else if (wxIsspace(ch))
200  newID.Append(_T('_')); // convert spaces to underscores
201  ++pos;
202  }
203  // make sure it's not starting with a number or underscore.
204  // if it is, prepend an 'A'
205  if (wxIsdigit(newID.GetChar(0)) || newID.GetChar(0) == _T('_'))
206  newID.Prepend(_T('A'));
207 
208  if (GetHighlightLanguage(newID) != HL_NONE)
209  return HL_NONE;
210 
211  m_Sets[newID].m_Langs = name;
212  m_Sets[newID].m_Lexers = lexer;
213  return newID;
214 }

256 void EditorColourSet::UpdateOptionsWithSameName(HighlightLanguage lang, OptionColour* base)
257 {
258  if (!base)
259  return;
260  // first find the index of this option
261  int idx = -1;
262  OptionSet& mset = m_Sets[lang];
263  for (unsigned int i = 0; i < mset.m_Colours.GetCount(); ++i)
264  {
265  OptionColour* opt = mset.m_Colours.Item(i);
266  if (opt == base)
267  {
268  idx = i;
269  break;
270  }
271  }
272  if (idx == -1)
273  return;
274 
275  // now loop again, but update the other options with the same name
276  for (unsigned int i = 0; i < mset.m_Colours.GetCount(); ++i)
277  {
278  if ((int)i == idx)
279  continue; // skip the base option
280  OptionColour* opt = mset.m_Colours.Item(i);
281  if (!opt->name.Matches(base->name))
282  continue;
283  opt->fore = base->fore;
284  opt->back = base->back;
285  opt->bold = base->bold;
286  opt->italics = base->italics;
287  opt->underlined = base->underlined;
288  }

401 HighlightLanguage EditorColourSet::GetLanguageForFilename(const wxString& filename)
402 {
403  // convert filename to lowercase first (m_FileMasks already contains
404  // lowercase-only strings) and allow for filemasks like Makefile.*:
405  wxString lfname = wxFileName(filename.Lower()).GetFullName();
406 
407  // first search in filemasks
408  for (OptionSetsMap::iterator it = m_Sets.begin(); it != m_Sets.end(); ++it)
409  {
410  for (unsigned int x = 0; x < it->second.m_FileMasks.GetCount(); ++x)
411  {
412  if (lfname.Matches(it->second.m_FileMasks.Item(x)))
413  return it->first;
414  }
415  }
416  // parse #! directive
417  if ( wxFileExists(filename) )
418  {
419  wxFileInputStream input(filename);
420  wxTextInputStream text(input);
421  wxString line;
422  if (input.IsOk() && !input.Eof() )
423  line = text.ReadLine();
424  if (!line.IsEmpty())
425  {
426  wxRegEx reSheBang(wxT("#![ \t]*([a-zA-Z/]+)[ \t]*([a-zA-Z/]*)"));
427  if (reSheBang.Matches(line))
428  {
429  wxString prog = reSheBang.GetMatch(line, 1);
430  if (prog.EndsWith(wxT("env")))
431  prog = reSheBang.GetMatch(line, 2);
432  if (prog.Find(wxT('/')) != wxNOT_FOUND)
433  prog = prog.AfterLast(wxT('/'));
434  if (prog == wxT("sh"))
435  prog = wxT("bash");
436  HighlightLanguage lang = GetHighlightLanguage(prog);
437  if (lang != HL_NONE)
438  return lang;
439  }
440  else if (line.Trim().StartsWith(wxT("<?xml")))
441  return GetHighlightLanguage(wxT("XML"));
442  }
443  }
444  // standard headers
445  const wxString cppNames = wxT("|"
446  "algorithm|" "atomic|" "array|" "bitset|"
447  "chrono|" "complex|" "condition_variable|" "deque|"
448  "exception|" "fstream|" "forward_list|" "future|"
449  "functional|" "hash_map|" "hash_set|" "initializer_list|"
450  "iomanip|" "ios|" "iostream|" "istream|"
451  "iterator|" "limits|" "list|" "locale|"
452  "map|" "memory|" "mutex|" "new|"
453  "numeric|" "ostream|" "queue|" "random|"
454  "ratio|" "regex|" "set|" "sstream|"
455  "stack|" "stdexcept|" "streambuf|" "string|"
456  "strstream|" "system_error|" "thread|" "tuple|"
457  "typeinfo|" "type_traits|" "unordered_map|" "unordered_set|"
458  "utility|" "valarray|" "vector|"
459 
460  "cassert|" "cctype|" "cerrno|" "cfenv|" "cfloat|"
461  "cinttypes|" "ciso646|" "climits|" "clocale|" "cmath|"
462  "csetjmp|" "csignal|" "cstdarg|" "cstdbool|" "cstddef|"
463  "cstdint|" "cstdio|" "cstdlib|" "cstring|" "ctgmath|"
464  "ctime|" "cuchar|" "cwchar|" "cwctype|" );
465  if (cppNames.Find(wxT("|") + lfname + wxT("|")) != wxNOT_FOUND)
466  return GetHighlightLanguage(wxT("C/C++"));
467  return HL_NONE;
468 }

514 void EditorColourSet::Apply(HighlightLanguage lang, cbStyledTextCtrl* control, bool isC)
515 {
516  if (!control)
517  return;
518  control->StyleClearAll();
519 
520  if (lang == HL_NONE)
521  return;
522 
523  // first load the default colours to all styles used by the actual lexer (ignoring some built-in styles)
524  OptionColour* defaults = GetDefaultOption(lang);
525  OptionSet& mset = m_Sets[lang];
526  control->SetLexer(mset.m_Lexers);
527  control->SetStyleBits(control->GetStyleBitsNeeded());
528  if (defaults)
529  {
530  int countStyles = 1 << control->GetStyleBits();
531  // walk until countStyles, otherwise the background-colour is only set for characters,
532  // not for empty background
533  for (int i = 0; i <= countStyles; ++i)
534  {
535  if (i < 33 || (i > 39 && i < wxSCI_STYLE_MAX))
536  DoApplyStyle(control, i, defaults);
537  }
538  }
539 
540  // Calling StyleClearAll above clears the style for the line numbers, so we have to re-apply it.
541  ColourManager *colours = Manager::Get()->GetColourManager();
542  control->StyleSetForeground(wxSCI_STYLE_LINENUMBER, colours->GetColour(wxT("editor_linenumbers_fg")));
543  control->StyleSetBackground(wxSCI_STYLE_LINENUMBER, colours->GetColour(wxT("editor_linenumbers_bg")));
544 
545  for (unsigned int i = 0; i < mset.m_Colours.GetCount(); ++i)
546  {
547  OptionColour* opt = mset.m_Colours.Item(i);
548 
549  if (opt->isStyle)
550  {
551  DoApplyStyle(control, opt->value, opt);
552  }
553  else
554  {
555  if (opt->value == cbHIGHLIGHT_LINE)
556  {
557  control->SetCaretLineBackground(opt->back);
558  Manager::Get()->GetConfigManager(_T("editor"))->Write(_T("/highlight_caret_line_colour"), opt->back);
559  }
560  else if (opt->value == cbSELECTION)
561  {
562  if (opt->back != wxNullColour)
563  {
564  control->SetSelBackground(true, opt->back);
565 // Manager::Get()->GetConfigManager(_T("editor"))->Write(_T("/selection_colour"), opt->back);
566  }
567  else
568  control->SetSelBackground(false, wxColour(0xC0, 0xC0, 0xC0));
569 
570  if (opt->fore != wxNullColour)
571  {
572  control->SetSelForeground(true, opt->fore);
573 // Manager::Get()->GetConfigManager(_T("editor"))->Write(_T("/selection_fgcolour"), opt->fore);
574  }
575  else
576  control->SetSelForeground(false, *wxBLACK);
577  }
578 // else
579 // {
580 // control->MarkerDefine(-opt->value, 1);
581 // control->MarkerSetBackground(-opt->value, opt->back);
582 // }
583  }
584  }
585  for (int i = 0; i <= wxSCI_KEYWORDSET_MAX; ++i)
586  {
587  if (!isC || i != 1) // exclude stl highlights for C
588  control->SetKeyWords(i, mset.m_Keywords[i]);
589  }
590 
591  control->Colourise(0, -1); // the *most* important part!
592 }

594 void EditorColourSet::Save()
595 {
596  // no need for syntax highlighting if batch building
597  if (Manager::IsBatchBuild())
598  return;
599 
600  wxString key;
601  ConfigManager* cfg = Manager::Get()->GetConfigManager(_T("editor"));
602 
603  //FIXME: Commenting out the following line is no definite cure, but it hides the annoying disappearing colourset for now
604  //NOTE (mandrav): uncommenting it doesn't seem to cause any trouble (at least now). What was the problem?
605  cfg->DeleteSubPath(_T("/colour_sets/") + m_Name);
606 
607  // write the theme name
608  cfg->Write(_T("/colour_sets/") + m_Name + _T("/name"), m_Name);
609 
610  for (OptionSetsMap::iterator it = m_Sets.begin(); it != m_Sets.end(); ++it)
611  {
612  if (it->first == HL_NONE || it->first == HL_AUTO)
613  continue;
614  wxString lang = it->first;
615 
616  bool gsaved = false;
617 
618  key.Clear();
619  key << _T("/colour_sets/") << m_Name << _T('/') << lang;
620  for (unsigned int i = 0; i < it->second.m_Colours.GetCount(); ++i)
621  {
622  OptionColour* opt = it->second.m_Colours.Item(i);
623  wxString tmpKey;
624  tmpKey << key << _T("/style") << wxString::Format(_T("%u"), i);
625 
626  bool saved = false;
627 
628  if (opt->fore != opt->originalfore)
629  {
630  cfg->Write(tmpKey + _T("/fore"), opt->fore);
631  saved = true;
632  }
633  if (opt->back != opt->originalback)
634  {
635  cfg->Write(tmpKey + _T("/back"), opt->back);
636  saved = true;
637  }
638  if (opt->bold != opt->originalbold)
639  {
640  cfg->Write(tmpKey + _T("/bold"), opt->bold);
641  saved = true;
642  }
643  if (opt->italics != opt->originalitalics)
644  {
645  cfg->Write(tmpKey + _T("/italics"), opt->italics);
646  saved = true;
647  }
648  if (opt->underlined != opt->originalunderlined)
649  {
650  cfg->Write(tmpKey + _T("/underlined"), opt->underlined);
651  saved = true;
652  }
653  if (opt->isStyle != opt->originalisStyle)
654  {
655  cfg->Write(tmpKey + _T("/isStyle"), opt->isStyle);
656  saved = true;
657  }
658

689 void EditorColourSet::Load()
690 {

693  return;

695  static bool s_notifiedUser = false;

699 
700  // read the theme name

702 
703  int x = 0;

707  continue;

711  key << _T("/colour_sets/") << m_Name << _T("/set") << wxString::Format(_T("%d"), x++);

716  cfg->DeleteSubPath(key);
717  if (!s_notifiedUser)

724  s_notifiedUser = true;
725  }
726  continue;
727  }
728  // make sure we didn't create it accidentally
729  cfg->DeleteSubPath(key);
730 
731  // new-style configuration key
732  key.Clear();
733  key << _T("/colour_sets/") << m_Name << _T('/') << it->first;
734  if (!cfg->Exists(key + _T("/name")))
735  {
736  // make sure we didn't create it accidentally
737  cfg->DeleteSubPath(key);
738  continue;
739  }
740

742  {
743  wxString tmpKey;

746  {
747  // make sure we didn't create it accidentally
748  cfg->DeleteSubPath(tmpKey);

753  {
754  OptionColour* opt = it->second.m_Colours.Item(j);
755  if (!opt || opt->name != name)
756  continue;

764  if (cfg->Exists(tmpKey + _T("/italics")))
765  opt->italics = cfg->ReadBool(tmpKey + _T("/italics"), opt->italics);
766  if (cfg->Exists(tmpKey + _T("/underlined")))
767  opt->underlined = cfg->ReadBool(tmpKey + _T("/underlined"), opt->underlined);
768  if (cfg->Exists(tmpKey + _T("/isStyle")))
769  opt->isStyle = cfg->ReadBool(tmpKey + _T("/isStyle"), opt->isStyle);
770  }
771  }
772  wxString tmpkey;
773  for (int i = 0; i <= wxSCI_KEYWORDSET_MAX; ++i)
774  {
775  tmpkey.Printf(_T("%s/editor/keywords/set%d"), key.c_str(), i);

808 void EditorColourSet::SetKeywords(HighlightLanguage lang, int idx, const wxString& keywords)
809 {
810  if (lang != HL_NONE && idx >=0 && idx <= wxSCI_KEYWORDSET_MAX)
811  {
812  wxString tmp(_T(' '), keywords.length()); // faster than using Alloc()
813 
814  const wxChar *src = keywords.c_str();
815  #if wxCHECK_VERSION(3, 0, 0)
816  wxStringCharType *dst = const_cast<wxStringCharType*>(tmp.wx_str());
817  #else
818  wxChar *dst = (wxChar *) tmp.c_str();
819  #endif
820  wxChar c;
821  size_t len = 0;
822 
823  while ((c = *src))
824  {
825  ++src;
826  if (c > _T(' '))
827  *dst = c;
828  else // white space
829  {
830  *dst = _T(' ');
831  while (*src && *src < _T(' '))
832  ++src;
833  }
834 
835  ++dst;
836  ++len;
837  }
838 
839  tmp.Truncate(len);
840 
841  OptionSet& mset = m_Sets[lang];
842  mset.m_Keywords[idx] = tmp;
843  }
