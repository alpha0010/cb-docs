531 int DebuggerGDB::LaunchProcess(const wxString& cmd, const wxString& cwd)
532 {
533  if (m_pProcess)
534  return -1;
535 
536  // start the gdb process
537  m_pProcess = new PipedProcess(&m_pProcess, this, idGDBProcess, true, cwd);
538  Log(_("Starting debugger: ") + cmd);
539  m_Pid = wxExecute(cmd, wxEXEC_ASYNC, m_pProcess);
540 
541 #ifdef __WXMAC__
542  if (m_Pid == -1)
543  {
544  // Great! We got a fake PID. Time to Go Fish with our "ps" rod:
545 
546  m_Pid = 0;
547  pid_t mypid = getpid();
548  wxString mypidStr;
549  mypidStr << mypid;
550 
551  long pspid = 0;
552  wxString psCmd;
553  wxArrayString psOutput;
554  wxArrayString psErrors;
555 
556  psCmd << wxT("/bin/ps -o ppid,pid,command");
557  DebugLog(wxString::Format( _("Executing: %s"), psCmd.wx_str()) );
558  int result = wxExecute(psCmd, psOutput, psErrors, wxEXEC_SYNC);
559 
560  mypidStr << wxT(" ");
561 
562  for (int i = 0; i < psOutput.GetCount(); ++i)
563  { // PPID PID COMMAND
564  wxString psLine = psOutput.Item(i);
565  if (psLine.StartsWith(mypidStr) && psLine.Contains(wxT("gdb")))
566  {
567  wxString pidStr = psLine.Mid(mypidStr.Length());
568  pidStr = pidStr.BeforeFirst(' ');
569  if (pidStr.ToLong(&pspid))
570  {
571  m_Pid = pspid;
572  break;
573  }
574  }
575  }
576 
577  for (int i = 0; i < psErrors.GetCount(); ++i)
578  DebugLog(wxString::Format( _("PS Error:%s"), psErrors.Item(i).wx_str()) );
579  }
580 #endif
581 
582  if (!m_Pid)
583  {
584  delete m_pProcess;
585  m_pProcess = 0;
586  Log(_("failed"), Logger::error);
587  return -1;
588  }
589  else if (!m_pProcess->GetOutputStream())
590  {
591  delete m_pProcess;
592  m_pProcess = 0;
593  Log(_("failed (to get debugger's stdin)"), Logger::error);
594  return -2;
595  }
596  else if (!m_pProcess->GetInputStream())
597  {
598  delete m_pProcess;
599  m_pProcess = 0;
600  Log(_("failed (to get debugger's stdout)"), Logger::error);
601  return -2;
602  }
603  else if (!m_pProcess->GetErrorStream())
604  {
605  delete m_pProcess;
606  m_pProcess = 0;
607  Log(_("failed (to get debugger's stderr)"), Logger::error);
608  return -2;
609  }
610  Log(_("done"));
611  return 0;
612 }

993 void DebuggerGDB::ConvertToGDBDirectory(wxString& str, wxString base, bool relative)
994 {

996  return;
997 
998  ConvertToGDBFriendly(str);
999  ConvertToGDBFriendly(base);

1001  StripQuotes(base);
1002 
1003  if (platform::windows)
1004  {
1005  int ColonLocation = str.Find(_T(':'));
1006  bool convert_path_83 = false;
1007  if (ColonLocation != wxNOT_FOUND)
1008  convert_path_83 = true;
1009  else if (!base.IsEmpty() && str.GetChar(0) != _T('/'))
1010  {
1011  if (base.GetChar(base.Length()) == _T('/'))
1012  base = base.Mid(0, base.Length() - 2);
1013 
1014  while (!str.IsEmpty())
1015  {
1016  base += _T("/") + str.BeforeFirst(_T('/'));
1017  if (str.Find(_T('/')) != wxNOT_FOUND) str = str.AfterFirst(_T('/'));
1018  else str.Clear();
1019  }
1020  convert_path_83 = true;
1021  }
1022 
1023  // If can, get 8.3 name for path (Windows only)
1024  if (convert_path_83 && str.Contains(_T(' '))) // only if has spaces
1025  {
1026  wxFileName fn(str); // might contain a file name, too
1027  wxString path_83 = fn.GetShortPath();
1028  if (!path_83.IsEmpty())
1029  str = path_83; // construct filename again
1030  }
1031 
1032  if (ColonLocation == wxNOT_FOUND || base.IsEmpty())
1033  relative = false; // Can't do it
1034  }
1035  else
1036  {
1037  if ((str.GetChar(0) != _T('/') && str.GetChar(0) != _T('~')) || base.IsEmpty())
1038  relative = false;
1039  }
1040 
1041  if (relative)
1042  {
1043  if (platform::windows)
1044  {
1045  if (str.Find(_T(':')) != wxNOT_FOUND)
1046  str = str.Mid(str.Find(_T(':')) + 2, str.Length());
1047  if (base.Find(_T(':')) != wxNOT_FOUND)
1048  base = base.Mid(base.Find(_T(':')) + 2, base.Length());
1049  }
1050  else
1051  {
1052  if (str.GetChar(0) == _T('/')) str = str.Mid(1, str.Length());
1053  else if (str.GetChar(0) == _T('~')) str = str.Mid(2, str.Length());
1054

1056  else if (base.GetChar(0) == _T('~')) base = base.Mid(2, base.Length());
1057  }

1060  {
1061  if (str.BeforeFirst(_T('/')) == base.BeforeFirst(_T('/')))
1062  {
1063  if (str.Find(_T('/')) == wxNOT_FOUND) str.Clear();
1064  else str = str.AfterFirst(_T('/'));
1065 
1066  if (base.Find(_T('/')) == wxNOT_FOUND) base.Clear();
1067  else base = base.AfterFirst(_T('/'));
1068  }
1069  else break;
1070  }
1071  while (!base.IsEmpty())
1072  {
1073  str = _T("../") + str;
1074  if (base.Find(_T('/')) == wxNOT_FOUND) base.Clear();
1075  else base = base.AfterFirst(_T('/'));
1076  }

1573 void DebuggerGDB::DoBreak(bool temporary)
1574 {
1575  m_TemporaryBreak = temporary;
1576 
1577  // m_Process is PipedProcess I/O; m_Pid is debugger pid
1578  if (m_pProcess && m_Pid && !IsStopped())
1579  {
1580  long childPid = m_State.GetDriver()->GetChildPID();
1581  long pid = childPid;
1582  #ifndef __WXMSW__
1583  if (pid > 0 && !wxProcess::Exists(pid))
1584  {
1585  DebugLog(wxString::Format(_("Child process (pid:%ld) doesn't exists"), pid), Logger::warning);
1586  pid = 0;
1587  }
1588  if (pid <= 0)
1589  pid = m_Pid; // try poking gdb directly
1590  // non-windows gdb can interrupt the running process. yay!
1591  if (pid <= 0) // look out for the "fake" PIDs (killall)
1592  cbMessageBox(_("Unable to stop the debug process!"), _("Error"), wxOK | wxICON_WARNING);
1593  else
1594  {
1595  if (!wxProcess::Exists(pid))
1596  DebugLog(wxString::Format(_("GDB process (pid:%ld) doesn't exists"), pid), Logger::error);
1597 
1598  Log(F(_("Trying to interrupt process with pid: %ld; child pid: %ld gdb pid: %ld"),
1599  pid, childPid, static_cast<long>(m_Pid)));
1600  wxKillError error;
1601  if (wxKill(pid, wxSIGINT, &error) != 0)
1602  DebugLog(wxString::Format(_("Can't kill process (%ld) %d"), pid, (int)(error)));
1603  }
1604  #else
1605  // windows gdb can interrupt the running process too. yay!
1606  if ( (pid <=0)
1607  && (CreateToolhelp32SnapshotFunc!=NULL)
1608  && (Process32FirstFunc!=NULL)
1609  && (Process32NextFunc!=NULL) )
1610  {
1611  HANDLE snap = CreateToolhelp32SnapshotFunc(TH32CS_SNAPALL,0);
1612  if (snap!=INVALID_HANDLE_VALUE)
1613  {
1614  PROCESSENTRY32 lppe;
1615  lppe.dwSize = sizeof(PROCESSENTRY32);
1616  BOOL ok = Process32FirstFunc(snap, &lppe);
1617  while ( ok == TRUE)
1618  {
1619  if (static_cast<int>(lppe.th32ParentProcessID) == m_Pid) // Have my Child...
1620  {
1621  pid = lppe.th32ProcessID;
1622  DebugLog(F(_("Found child: %ld"), pid));
1623  }
1624  lppe.dwSize = sizeof(PROCESSENTRY32);
1625  ok = Process32NextFunc(snap, &lppe);
1626  }
1627  CloseHandle(snap);
1628  }
1629  else
1630  Log(_("No handle created. Trying to pause directly with cbd.exe..."), Logger::warning);
1631  }
1632 
1633  if (m_State.GetDriver()->UseDebugBreakProcess())
1634  {
1635  if (!DebugBreakProcessFunc)
1636  Log(_("DebugBreakProcess is not supported, you need Windows XP or newer..."), Logger::error);
1637  else if (pid > 0)
1638  {
1639  Log(F(_("Trying to interrupt process with pid: %ld; child pid: %ld gdb pid: %ld"),
1640  pid, childPid, static_cast<long>(m_Pid)));
1641  HANDLE proc = OpenProcess(PROCESS_ALL_ACCESS, FALSE, (DWORD)pid);
1642  if (proc)
1643  {
1644  DebugBreakProcessFunc(proc); // yay!
1645  CloseHandle(proc);
1646  }
1647  else
1648  Log(wxT("Interrupting debugger failed :("), Logger::error);
1649  }
1650  }
1651  else
1652  {
1653  if (m_Pid > 0)
1654  {
1655  Log(_("Trying to interrupt the process by sending CTRL-C event to the console!"));
1656  if (GenerateConsoleCtrlEvent(CTRL_C_EVENT, 0) == 0)
1657  {
1658  Log(wxT("Interrupting debugger failed :("), Logger::error);
1659  return;
1660  }
1661  }
1662  }
1663  #endif
1664  // Notify debugger plugins for end of debug session
1665  PluginManager *plm = Manager::Get()->GetPluginManager();
1666  CodeBlocksEvent evt(cbEVT_DEBUGGER_PAUSED);
1667  plm->NotifyPlugins(evt);
1668  }
1669 }
