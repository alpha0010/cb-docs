83 void InfoPane::LoadTabOrder(wxString layout)
84 {
85  wxString theLayout(layout);
86 
87  long selectedTab;
88  theLayout.AfterLast('=').ToLong(&selectedTab);
89  theLayout.Remove(theLayout.Find('|',true));
90  wxStringTokenizer strTok(theLayout, _T(";"));
91  wxString title;
92  while (strTok.HasMoreTokens())
93  {
94  wxString theToken = strTok.GetNextToken();
95  title = theToken.BeforeFirst('=');
96  for (size_t j = 0; j < m_Pages.GetCount();++j)
97  {
98  if (m_Pages.Item(j)->title == title)
99  {
100  long theIndex;
101  if (theToken.AfterFirst('=').ToLong(&theIndex))
102  m_Pages.Item(j)->indexInNB = theIndex;
103  break;
104  }
105  }
106  }
107  ReorderTabs(&CompareIndexes);
108  SetSelection(selectedTab);
109 }

118 void InfoPane::ReorderTabs(CompareFunction cmp_f)
119 {
120  if (m_Pages.GetCount() == 0)
121  return;
122  m_Pages.Sort(cmp_f);
123 
124  cbAuiNotebook::Hide();
125  int index = 0;
126  for (size_t i = 0 ; i < m_Pages.GetCount(); ++i)
127  {
128  int pageIndex = GetPageIndex(m_Pages.Item(i)->window);
129  if (m_Pages.Item(i)->indexInNB < 0)
130  {
131  if (pageIndex >= 0)
132  RemovePage(pageIndex);
133  if (m_Pages.Item(i)->window)
134  m_Pages.Item(i)->window->Hide();
135  }
136  else
137  {
138  if (pageIndex < 0)
139  AddPagePrivate(m_Pages.Item(i)->window, m_Pages.Item(i)->title, m_Pages.Item(i)->icon);
140  if (index++ != pageIndex)
141  MovePage(m_Pages.Item(i)->window, index );
142  }
143  }
144  cbAuiNotebook::Show();
145 }

173 void InfoPane::Toggle(size_t i)
174 {
175  UpdateEffectiveTabOrder();
176  if (m_Pages.Item(i)->indexInNB < 0)

179  m_Pages.Item(i)->indexInNB = ~m_Pages.Item(i)->indexInNB ;
180  else
181  m_Pages.Item(i)->indexInNB = AddPagePrivate(m_Pages.Item(i)->window, m_Pages.Item(i)->title, m_Pages.Item(i)->icon);

183  else

187  if (m_Pages.Item(i)->window)
188  m_Pages.Item(i)->window->Hide();
189  RemovePage(GetPageIndex(m_Pages.Item(i)->window));

363 void InfoPane::DoShowContextMenu()
364 {
365  UpdateEffectiveTabOrder();
366  m_Pages.Sort(&CompareIndexes);
367 
368  wxMenu menu;
369 
370  int selection = GetSelection();
371  if ( (selection >= 0)
372  && (selection < static_cast<int>(GetPageCount()))
373  && (m_Pages.Item(GetPageIndexByWindow( GetPage(GetSelection()) ))->islogger) )
374  {
375  Logger* l = m_Pages.Item(GetPageIndexByWindow( GetPage(GetSelection()) ))->logger;
376 
377  if (l->HasFeature(Logger::Feature::CanCopy))
378  {
379  menu.Append(idCopyAllToClipboard, _("Copy contents to clipboard"));
380  menu.Append(idCopySelectedToClipboard, _("Copy selection to clipboard"));
381  }
382  bool needSeparator = true;
383  if (l->HasFeature(Logger::Feature::IsWrappable))
384  {
385  if (menu.GetMenuItemCount() > 0)
386  {
387  needSeparator = false;
388  menu.AppendSeparator();
389  }
390  menu.AppendCheckItem(idWrapMode, _("Toggle wrap mode"));
391  menu.Check(idWrapMode, l->GetWrapMode());
392  }
393  if (l->HasFeature(Logger::Feature::CanClear))
394  {
395  if (needSeparator && menu.GetMenuItemCount() > 0)
396  menu.AppendSeparator();
397  menu.Append(idClear, _("Clear contents"));
398  }
399  if (l->HasFeature(Logger::Feature::Additional))
400  l->AppendAdditionalMenuItems(menu);
401  }
402 
403  if (menu.GetMenuItemCount() > 0)
404  menu.AppendSeparator();
405 
406  if (Manager::Get()->GetConfigManager(_T("app"))->ReadBool(_T("/environment/infopane_tabs_bottom"), false))
407  menu.Append(idNB_TabTop, _("Tabs at top"));
408  else
409  menu.Append(idNB_TabBottom, _("Tabs at bottom"));
410 
411  // add toggle sub-menu
412  wxMenu* view = new wxMenu;
413  for (size_t i = 0; i < m_Pages.GetCount(); ++i)
414  {
415  if (m_Pages.Item(i)->window)
416  {
417  view->Append(m_Pages.Item(i)->eventID, m_Pages.Item(i)->title, wxEmptyString, wxITEM_CHECK);
418  view->Check(m_Pages.Item(i)->eventID, m_Pages.Item(i)->indexInNB >= 0);
419  }
420  }
421 
422  if (view->GetMenuItemCount() > 0)
423  menu.AppendSubMenu(view, _("Toggle..."));
424  else
425  delete view;
426 
427  PopupMenu(&menu);
428 }
